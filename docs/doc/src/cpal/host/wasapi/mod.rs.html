16/// The WASAPI host, the default windows host type.
17///
18/// Note: If you use a WASAPI output device as an input device it will
19/// transparently enable loopback mode (see
20/// https://docs.microsoft.com/en-us/windows/win32/coreaudio/loopback-recording).
21#[derive(Debug)]
22pub struct Host;

88fn windows_err_to_cpal_err<E: ErrDeviceNotAvailable>(e: windows::core::Error) -> E {

92fn windows_err_to_cpal_err_message<E: ErrDeviceNotAvailable>(
93    e: windows::core::Error,
94    message: &str,
95) -> E {
96    match e.code() {
97        Audio::AUDCLNT_E_DEVICE_INVALIDATED => E::device_not_available(),
98        _ => {
99            let description = format!("{}{}", message, e);
100            let err = BackendSpecificError { description };
101            err.into()
102        }
103    }
