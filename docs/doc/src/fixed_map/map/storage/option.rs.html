41/// [`MapStorage`] for [`Option`] types.
42///
43/// # Examples
44///
45/// ```
46/// use fixed_map::{Key, Map};
47///
48/// #[derive(Debug, Clone, Copy, PartialEq, Key)]
49/// enum Part {
50///     A,
51///     B,
52/// }
53///
54/// #[derive(Debug, Clone, Copy, PartialEq, Key)]
55/// enum MyKey {
56///     First(Option<Part>),
57///     Second,
58/// }
59///
60/// let mut a = Map::new();
61/// a.insert(MyKey::First(None), 1);
62/// a.insert(MyKey::First(Some(Part::A)), 2);
63///
64/// assert_eq!(a.get(MyKey::First(Some(Part::A))), Some(&2));
65/// assert_eq!(a.get(MyKey::First(Some(Part::B))), None);
66/// assert_eq!(a.get(MyKey::First(None)), Some(&1));
67/// assert_eq!(a.get(MyKey::Second), None);
68///
69/// assert!(a.iter().eq([(MyKey::First(Some(Part::A)), &2), (MyKey::First(None), &1)]));
70/// assert!(a.values().copied().eq([2, 1]));
71/// assert!(a.keys().eq([MyKey::First(Some(Part::A)), MyKey::First(None)]));
72/// ```
73pub struct OptionMapStorage<K, V>
74where
75    K: Key,
76{
77    some: K::MapStorage<V>,
78    none: Option<V>,
79}

214impl<K, V> MapStorage<Option<K>, V> for OptionMapStorage<K, V>
215where
216    K: Key,
217{
218    type Iter<'this> = Iter<'this, K, V> where K: 'this, V: 'this;
219    type Keys<'this> = Keys<'this, K, V> where K: 'this, V: 'this;
220    type Values<'this> = Values<'this, K, V> where K: 'this, V: 'this;
221    type IterMut<'this> = IterMut<'this, K, V> where K: 'this, V: 'this;
222    type ValuesMut<'this> = ValuesMut<'this, K, V> where K: 'this, V: 'this;
223    type IntoIter = IntoIter<K, V>;
224    type Occupied<'this> = Occupied<'this, K, V> where K: 'this, V: 'this;
225    type Vacant<'this> = Vacant<'this, K, V> where K: 'this, V: 'this;
226
227    #[inline]
228    fn empty() -> Self {
229        Self {
230            some: K::MapStorage::empty(),
231            none: Option::default(),
232        }
233    }
234
235    #[inline]
236    fn len(&self) -> usize {
237        self.some.len() + usize::from(self.none.is_some())
238    }
239
240    #[inline]
241    fn is_empty(&self) -> bool {
242        self.some.is_empty() && self.none.is_none()
243    }
244
245    #[inline]
246    fn insert(&mut self, key: Option<K>, value: V) -> Option<V> {
247        match key {
248            Some(key) => self.some.insert(key, value),
249            None => mem::replace(&mut self.none, Some(value)),
250        }
251    }
252
253    #[inline]
254    fn contains_key(&self, key: Option<K>) -> bool {
255        match key {
256            Some(key) => self.some.contains_key(key),
257            None => self.none.is_some(),
258        }
259    }
260
261    #[inline]
262    fn get(&self, key: Option<K>) -> Option<&V> {
263        match key {
264            Some(key) => self.some.get(key),
265            None => self.none.as_ref(),
266        }
267    }
268
269    #[inline]
270    fn get_mut(&mut self, key: Option<K>) -> Option<&mut V> {
271        match key {
272            Some(key) => self.some.get_mut(key),
273            None => self.none.as_mut(),
274        }
275    }
276
277    #[inline]
278    fn remove(&mut self, key: Option<K>) -> Option<V> {
279        match key {
280            Some(key) => self.some.remove(key),
281            None => self.none.take(),
282        }
283    }
284
285    #[inline]
286    fn retain<F>(&mut self, mut func: F)
287    where
288        F: FnMut(Option<K>, &mut V) -> bool,
289    {
290        self.some.retain(|k, v| func(Some(k), v));
291        if let Some(none) = self.none.as_mut() {
292            if !func(None, none) {
293                self.none = None;
294            }
295        }
296    }
297
298    #[inline]
299    fn clear(&mut self) {
300        self.some.clear();
301        self.none = None;
302    }
303
304    #[inline]
305    fn iter(&self) -> Self::Iter<'_> {
306        let map: fn(_) -> _ = |(k, b)| (Some(k), b);
307        let a = self.some.iter().map(map);
308        let map: fn(_) -> _ = |v| (None, v);
309        let b = self.none.iter().map(map);
310        a.chain(b)
311    }
312
313    #[inline]
314    fn keys(&self) -> Self::Keys<'_> {
315        let map: fn(_) -> _ = |k| Some(k);
316        self.some
317            .keys()
318            .map(map)
319            .chain(self.none.is_some().then_some(None::<K>))
320    }
321
322    #[inline]
323    fn values(&self) -> Self::Values<'_> {
324        self.some.values().chain(self.none.iter())
325    }
326
327    #[inline]
328    fn iter_mut(&mut self) -> Self::IterMut<'_> {
329        let map: fn(_) -> _ = |(k, b)| (Some(k), b);
330        let a = self.some.iter_mut().map(map);
331        let map: fn(_) -> _ = |v| (None, v);
332        let b = self.none.iter_mut().map(map);
333        a.chain(b)
334    }
335
336    #[inline]
337    fn values_mut(&mut self) -> Self::ValuesMut<'_> {
338        self.some.values_mut().chain(self.none.iter_mut())
339    }
340
341    #[inline]
342    fn into_iter(self) -> Self::IntoIter {
343        let map: fn(_) -> _ = |(k, b)| (Some(k), b);
344        let a = self.some.into_iter().map(map);
345        let map: fn(_) -> _ = |v| (None, v);
346        let b = self.none.into_iter().map(map);
347        a.chain(b)
348    }
349
350    #[inline]
351    fn entry(&mut self, key: Option<K>) -> Entry<'_, Self, Option<K>, V> {
352        match key {
353            Some(key) => match self.some.entry(key) {
354                Entry::Occupied(entry) => Entry::Occupied(Occupied::Some(entry)),
355                Entry::Vacant(entry) => Entry::Vacant(Vacant::Some(entry)),
356            },
357            None => match OptionBucket::new(&mut self.none) {
358                OptionBucket::Some(some) => Entry::Occupied(Occupied::None(some)),
359                OptionBucket::None(none) => Entry::Vacant(Vacant::None(none)),
360            },
361        }
362    }
