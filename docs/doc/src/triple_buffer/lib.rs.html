1//! In this crate, we propose a Rust implementation of triple buffering. This is
2//! a non-blocking thread synchronization mechanism that can be used when a
3//! single producer thread is frequently updating a shared data block, and a
4//! single consumer thread wants to be able to read the latest available version
5//! of the shared data whenever it feels like it.
6//!
7//! # Examples
8//!
9//! For many use cases, you can use the ergonomic write/read interface, where
10//! the producer moves values into the buffer and the consumer accesses the
11//! latest buffer by shared reference:
12//!
13//! ```
14//! // Create a triple buffer
15//! use triple_buffer::triple_buffer;
16//! let (mut buf_input, mut buf_output) = triple_buffer(&0);
17//!
18//! // The producer thread can move a value into the buffer at any time
19//! let producer = std::thread::spawn(move || buf_input.write(42));
20//!
21//! // The consumer thread can read the latest value at any time
22//! let consumer = std::thread::spawn(move || {
23//!     let latest = buf_output.read();
24//!     assert!(*latest == 42 || *latest == 0);
25//! });
26//!
27//! # producer.join().unwrap();
28//! # consumer.join().unwrap();
29//! ```
30//!
31//! In situations where moving the original value away and being unable to
32//! modify it on the consumer's side is too costly, such as if creating a new
33//! value involves dynamic memory allocation, you can use a lower-level API
34//! which allows you to access the producer and consumer's buffers in place
35//! and to precisely control when updates are propagated:
36//!
37//! ```
38//! // Create and split a triple buffer
39//! use triple_buffer::triple_buffer;
40//! let (mut buf_input, mut buf_output) = triple_buffer(&String::with_capacity(42));
41//!
42//! // --- PRODUCER SIDE ---
43//!
44//! // Mutate the input buffer in place
45//! {
46//!     // Acquire a reference to the input buffer
47//!     let input = buf_input.input_buffer_mut();
48//!
49//!     // In general, you don't know what's inside of the buffer, so you should
50//!     // always reset the value before use (this is a type-specific process).
51//!     input.clear();
52//!
53//!     // Perform an in-place update
54//!     input.push_str("Hello, ");
55//! }
56//!
57//! // Publish the above input buffer update
58//! buf_input.publish();
59//!
60//! // --- CONSUMER SIDE ---
61//!
62//! // Manually fetch the buffer update from the consumer interface
63//! buf_output.update();
64//!
65//! // Acquire read-only reference to the output buffer
66//! let output = buf_output.peek_output_buffer();
67//! assert_eq!(*output, "Hello, ");
68//!
69//! // Or acquire mutable reference if necessary
70//! let output_mut = buf_output.output_buffer_mut();
71//!
72//! // Post-process the output value before use
73//! output_mut.push_str("world!");
74//! ```
75//!
76//! Finally, as a middle ground before the maximal ergonomics of the
77//! [`write()`](Input::write) API and the maximal control of the
78//! [`input_buffer_mut()`](Input::input_buffer_mut)/[`publish()`](Input::publish)
79//! API, you can also use the
80//! [`input_buffer_publisher()`](Input::input_buffer_publisher) RAII API on the
81//! producer side, which ensures that `publish()` is automatically called when
82//! the resulting input buffer handle goes out of scope:
83//!
84//! ```
85//! // Create and split a triple buffer
86//! use triple_buffer::triple_buffer;
87//! let (mut buf_input, _) = triple_buffer(&String::with_capacity(42));
88//!
89//! // Mutate the input buffer in place and publish it
90//! {
91//!     // Acquire a reference to the input buffer
92//!     let mut input = buf_input.input_buffer_publisher();
93//!
94//!     // In general, you don't know what's inside of the buffer, so you should
95//!     // always reset the value before use (this is a type-specific process).
96//!     input.clear();
97//!
98//!     // Perform an in-place update
99//!     input.push_str("Hello world!");
100//!
101//!     // Input buffer is automatically published at the end of the scope of
102//!     // the "input" RAII guard
103//! }
104//!
105//! // From this point on, the consumer can see the updated version
106//! ```

123/// A triple buffer, useful for nonblocking and thread-safe data sharing
124///
125/// A triple buffer is a single-producer single-consumer nonblocking
126/// communication channel which behaves like a shared variable: the producer
127/// submits regular updates, and the consumer accesses the latest available
128/// value whenever it feels like it.
129#[derive(Debug)]
130pub struct TripleBuffer<T: Send> {
131    /// Input object used by producers to send updates
132    input: Input<T>,
133
134    /// Output object used by consumers to read the current value
135    output: Output<T>,
136}
137//
138impl<T: Clone + Send> TripleBuffer<T> {
139    /// Construct a triple buffer with a certain initial value
140    pub fn new(initial: &T) -> Self {
141        Self::new_impl(|| initial.clone())
142    }
143}
144//
145impl<T: Default + Send> Default for TripleBuffer<T> {
146    /// Construct a triple buffer with a default-constructed value
147    fn default() -> Self {
148        Self::new_impl(T::default)
149    }
150}
151//
152impl<T: Send> TripleBuffer<T> {
153    /// Construct a triple buffer, using a functor to generate initial values
154    fn new_impl(mut generator: impl FnMut() -> T) -> Self {
155        // Start with the shared state...
156        let shared_state = Arc::new(SharedState::new(|_i| generator(), 0));
157
158        // ...then construct the input and output structs
159        TripleBuffer {
160            input: Input {
161                shared: shared_state.clone(),
162                input_idx: 1,
163            },
164            output: Output {
165                shared: shared_state,
166                output_idx: 2,
167            },
168        }
169    }
170
171    /// Extract input and output of the triple buffer
172    //
173    // NOTE: Although it would be nicer to directly return `Input` and `Output`
174    //       from `new()`, the `split()` design gives some API evolution
175    //       headroom towards future allocation-free modes of operation where
176    //       the SharedState is a static variable, or a stack-allocated variable
177    //       used through scoped threads or other unsafe thread synchronization.
178    //
179    //       See https://github.com/HadrienG2/triple-buffer/issues/8 .
180    //
181    pub fn split(self) -> (Input<T>, Output<T>) {
182        (self.input, self.output)
183    }
184}
185//
186/// Shorthand for `TripleBuffer::new(initial).split()`
187pub fn triple_buffer<T: Clone + Send>(initial: &T) -> (Input<T>, Output<T>) {
188    TripleBuffer::new(initial).split()
189}
190//
191// The Clone and PartialEq traits are used internally for testing and I don't
192// want to commit to supporting them publicly for now.
193//
194#[doc(hidden)]
195impl<T: Clone + Send> Clone for TripleBuffer<T> {
196    fn clone(&self) -> Self {
197        // Clone the shared state. This is safe because at this layer of the
198        // interface, one needs an Input/Output &mut to mutate the shared state.
199        let shared_state = Arc::new(unsafe { (*self.input.shared).clone() });
200
201        // ...then the input and output structs
202        TripleBuffer {
203            input: Input {
204                shared: shared_state.clone(),
205                input_idx: self.input.input_idx,
206            },
207            output: Output {
208                shared: shared_state,
209                output_idx: self.output.output_idx,
210            },
211        }
212    }
213}
214//
215#[doc(hidden)]
216impl<T: PartialEq + Send> PartialEq for TripleBuffer<T> {
217    fn eq(&self, other: &Self) -> bool {
218        // Compare the shared states. This is safe because at this layer of the
219        // interface, one needs an Input/Output &mut to mutate the shared state.
220        let shared_states_equal = unsafe { (*self.input.shared).eq(&*other.input.shared) };
221
222        // Compare the rest of the triple buffer states
223        shared_states_equal
224            && (self.input.input_idx == other.input.input_idx)
225            && (self.output.output_idx == other.output.output_idx)
226    }
227}

229/// Producer interface to the triple buffer
230///
231/// The producer of data can use this struct to submit updates to the triple
232/// buffer whenever he likes. These updates are nonblocking: a collision between
233/// the producer and the consumer will result in cache contention, but deadlocks
234/// and scheduling-induced slowdowns cannot happen.
235#[derive(Debug)]
236pub struct Input<T: Send> {
237    /// Reference-counted shared state
238    shared: Arc<SharedState<T>>,
239
240    /// Index of the input buffer (which is private to the producer)
241    input_idx: BufferIndex,
242}
243//
244// Public interface
245impl<T: Send> Input<T> {
246    /// Write a new value into the triple buffer
247    pub fn write(&mut self, value: T) {
248        // Update the input buffer
249        *self.input_buffer_mut() = value;
250
251        // Publish our update to the consumer
252        self.publish();
253    }
254
255    /// Check if the consumer has fetched our latest submission yet
256    ///
257    /// This method is only intended for diagnostics purposes. Please do not let
258    /// it inform your decision of sending or not sending a value, as that would
259    /// effectively be building a very poor spinlock-based double buffer
260    /// implementation. If what you truly need is a double buffer, build
261    /// yourself a proper blocking one instead of wasting CPU time.
262    pub fn consumed(&self) -> bool {
263        let back_info = self.shared.back_info.load(Ordering::Relaxed);
264        back_info & BACK_DIRTY_BIT == 0
265    }
266
267    /// Query the current value of the input buffer
268    ///
269    /// This is simply a read-only version of
270    /// [`input_buffer_mut()`](Input::input_buffer_mut). Please read the
271    /// documentation of that method for more information on the precautions
272    /// that need to be taken when accessing the input buffer in place.
273    fn peek_input_buffer(&self) -> &T {
274        // Access the input buffer directly
275        let input_ptr = self.shared.buffers[self.input_idx as usize].get();
276        unsafe { &*input_ptr }
277    }
278
279    /// Access the input buffer directly
280    ///
281    /// This is, for now, a deprecated alias to
282    /// [`input_buffer_mut()`](Input::input_buffer_mut). Please use this method
283    /// instead.
284    ///
285    /// In a future major release of this crate, `input_buffer()` will
286    /// undergo a breaking change to instead provide read-only access.
287    ///
288    /// The aim of this process is to eventually migrate towards the standard
289    /// `accessor()`/`accessor_mut()` method naming convention that most Rust
290    /// libraries follow.
291    #[deprecated = "Please use input_buffer_mut() instead"]
292    pub fn input_buffer(&mut self) -> &mut T {
293        self.input_buffer_mut()
294    }
295
296    /// Access the input buffer directly
297    ///
298    /// This advanced interface allows you to update the input buffer in place,
299    /// so that you can avoid creating values of type T repeatedy just to push
300    /// them into the triple buffer when doing so is expensive.
301    ///
302    /// However, by using it, you force yourself to take into account some
303    /// implementation subtleties that you could otherwise ignore.
304    ///
305    /// First, the buffer does not contain the last value that you published
306    /// (which is now available to the consumer thread). In fact, what you get
307    /// may not match _any_ value that you sent in the past, but rather be a new
308    /// value that was written in there by the consumer thread. All you can
309    /// safely assume is that the buffer contains a valid value of type T, which
310    /// you may need to "clean up" before use using a type-specific process
311    /// (like calling the `clear()` method of a `Vec`/`String`).
312    ///
313    /// Second, we do not send updates automatically. You need to call
314    /// [`publish()`](Input::publish) in order to propagate a buffer update to
315    /// the consumer. If you would prefer this to be done automatically when the
316    /// input buffer reference goes out of scope, consider using the
317    /// [`input_buffer_publisher()`](Input::input_buffer_publisher) RAII
318    /// interface instead.
319    pub fn input_buffer_mut(&mut self) -> &mut T {
320        // This is safe because the synchronization protocol ensures that we
321        // have exclusive access to this buffer.
322        let input_ptr = self.shared.buffers[self.input_idx as usize].get();
323        unsafe { &mut *input_ptr }
324    }
325
326    /// Publish the current input buffer, checking for overwrites
327    ///
328    /// After updating the input buffer in-place using
329    /// [`input_buffer_mut()`](Input::input_buffer_mut), you can use this method
330    /// to publish your updates to the consumer. Beware that this will replace
331    /// the current input buffer with another one that has totally unrelated
332    /// contents.
333    ///
334    /// It will also tell you whether you overwrote a value which was not read
335    /// by the consumer thread.
336    pub fn publish(&mut self) -> bool {
337        // Swap the input buffer and the back buffer, setting the dirty bit
338        //
339        // The ordering must be AcqRel, because...
340        //
341        // - Our accesses to the old buffer must not be reordered after this
342        //   operation (which mandates Release ordering), otherwise they could
343        //   race with the consumer accessing the freshly published buffer.
344        // - Our accesses from the buffer must not be reordered before this
345        //   operation (which mandates Consume ordering, that is best
346        //   approximated by Acquire in Rust), otherwise they would race with
347        //   the consumer accessing the buffer as well before switching to
348        //   another buffer.
349        //   * This reordering may seem paradoxical, but could happen if the
350        //     compiler or CPU correctly speculated the new buffer's index
351        //     before that index is actually read, as well as on weird hardware
352        //     with incoherent caches like GPUs or old DEC Alpha where keeping
353        //     data in sync across cores requires manual action.
354        //
355        let former_back_info = self
356            .shared
357            .back_info
358            .swap(self.input_idx | BACK_DIRTY_BIT, Ordering::AcqRel);
359
360        // The old back buffer becomes our new input buffer
361        self.input_idx = former_back_info & BACK_INDEX_MASK;
362
363        // Tell whether we have overwritten unread data
364        former_back_info & BACK_DIRTY_BIT != 0
365    }
366
367    /// Access the input buffer wrapped in the `InputPublishGuard`
368    ///
369    /// This is an RAII alternative to the [`input_buffer_mut()`]/[`publish()`]
370    /// workflow where the [`publish()`] transaction happens automatically when
371    /// the input buffer handle goes out of scope.
372    ///
373    /// Please check out the documentation of [`input_buffer_mut()`] and
374    /// [`publish()`] to know more about the precautions that you need to take
375    /// when using the lower-level in-place buffer access interface.
376    ///
377    /// [`input_buffer_mut()`]: Input::input_buffer_mut
378    /// [`publish()`]: Input::publish
379    pub fn input_buffer_publisher(&mut self) -> InputPublishGuard<T> {
380        InputPublishGuard { reference: self }
381    }
382}

384/// RAII Guard to the buffer provided by an [`Input`].
385///
386/// The current buffer of the [`Input`] can be accessed through this guard via
387/// its [`Deref`] and [`DerefMut`] implementations.
388///
389/// When the guard is dropped, [`Input::publish()`] will be called
390/// automatically.
391///
392/// This structure is created by the [`Input::input_buffer_publisher()`] method.
393pub struct InputPublishGuard<'a, T: 'a + Send> {
394    reference: &'a mut Input<T>,
395}
396
397impl<T: Send> Deref for InputPublishGuard<'_, T> {
398    type Target = T;
399
400    fn deref(&self) -> &T {
401        self.reference.peek_input_buffer()
402    }
403}
404
405impl<T: Send> DerefMut for InputPublishGuard<'_, T> {
406    fn deref_mut(&mut self) -> &mut T {
407        self.reference.input_buffer_mut()
408    }
409}
410
411impl<T: Send> Drop for InputPublishGuard<'_, T> {
412    #[inline]
413    fn drop(&mut self) {
414        self.reference.publish();
415    }
416}
417
418impl<T: Send + fmt::Debug> fmt::Debug for InputPublishGuard<'_, T> {
419    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
420        fmt::Debug::fmt(&**self, f)
421    }
422}
423
424impl<T: fmt::Display + Send> fmt::Display for InputPublishGuard<'_, T> {
425    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
426        (**self).fmt(f)
427    }
428}

430/// Consumer interface to the triple buffer
431///
432/// The consumer of data can use this struct to access the latest published
433/// update from the producer whenever he likes. Readout is nonblocking: a
434/// collision between the producer and consumer will result in cache contention,
435/// but deadlocks and scheduling-induced slowdowns cannot happen.
436#[derive(Debug)]
437pub struct Output<T: Send> {
438    /// Reference-counted shared state
439    shared: Arc<SharedState<T>>,
440
441    /// Index of the output buffer (which is private to the consumer)
442    output_idx: BufferIndex,
443}
444//
445// Public interface
446impl<T: Send> Output<T> {
447    /// Access the latest value from the triple buffer
448    pub fn read(&mut self) -> &T {
449        // Fetch updates from the producer
450        self.update();
451
452        // Give access to the output buffer
453        self.output_buffer_mut()
454    }
455
456    /// Tell whether an updated value has been submitted by the producer
457    ///
458    /// This method is mainly intended for diagnostics purposes. Please do not
459    /// let it inform your decision of reading a value or not, as that would
460    /// effectively be building a very poor spinlock-based double buffer
461    /// implementation. If what you truly need is a double buffer, build
462    /// yourself a proper blocking one instead of wasting CPU time.
463    pub fn updated(&self) -> bool {
464        let back_info = self.shared.back_info.load(Ordering::Relaxed);
465        back_info & BACK_DIRTY_BIT != 0
466    }
467
468    /// Query the current value of the output buffer
469    ///
470    /// This is simply a read-only version of
471    /// [`output_buffer_mut()`](Output::output_buffer_mut). Please read the
472    /// documentation of that method for more information on the precautions
473    /// that need to be taken when accessing the output buffer in place.
474    ///
475    /// In particular, remember that this method does not update the output
476    /// buffer automatically. You need to call [`update()`](Output::update) in
477    /// order to fetch buffer updates from the producer.
478    pub fn peek_output_buffer(&self) -> &T {
479        // Access the output buffer directly
480        let output_ptr = self.shared.buffers[self.output_idx as usize].get();
481        unsafe { &*output_ptr }
482    }
483
484    /// Access the input buffer directly
485    ///
486    /// This is, for now, a deprecated alias to [`output_buffer_mut()`]. Please
487    /// use this method instead.
488    ///
489    /// In a future major release of this crate, `output_buffer()` will undergo
490    /// a breaking change to instead provide read-only access, like
491    /// [`peek_output_buffer()`] currently does. At that point,
492    /// [`peek_output_buffer()`] will be deprecated.
493    ///
494    /// Finally, in a later major release [`peek_output_buffer()`] will be
495    /// removed.
496    ///
497    /// The aim of this process is to eventually migrate towards the standard
498    /// `accessor()`/`accessor_mut()` method naming convention that most Rust
499    /// libraries follow.
500    ///
501    /// [`output_buffer_mut()`]: Output::output_buffer_mut
502    /// [`peek_output_buffer()`]: Output::peek_output_buffer
503    #[deprecated = "Please use output_buffer_mut() instead"]
504    pub fn output_buffer(&mut self) -> &mut T {
505        self.output_buffer_mut()
506    }
507
508    /// Access the output buffer directly
509    ///
510    /// This advanced interface allows you to modify the contents of the output
511    /// buffer, so that you can avoid copying the output value when this is an
512    /// expensive process. One possible application, for example, is to
513    /// post-process values from the producer before use.
514    ///
515    /// However, by using it, you force yourself to take into account some
516    /// implementation subtleties that you could normally ignore.
517    ///
518    /// First, keep in mind that you can lose access to the current output
519    /// buffer any time [`read()`] or [`update()`] is called, as it may be
520    /// replaced by an updated buffer from the producer automatically.
521    ///
522    /// Second, to reduce the potential for the aforementioned usage error, this
523    /// method does not update the output buffer automatically. You need to call
524    /// [`update()`] in order to fetch buffer updates from the producer.
525    ///
526    /// [`read()`]: Output::read
527    /// [`update()`]: Output::update
528    pub fn output_buffer_mut(&mut self) -> &mut T {
529        // This is safe because the synchronization protocol ensures that we
530        // have exclusive access to this buffer.
531        let output_ptr = self.shared.buffers[self.output_idx as usize].get();
532        unsafe { &mut *output_ptr }
533    }
534
535    /// Update the output buffer
536    ///
537    /// Check if the producer submitted a new data version, and if one is
538    /// available, update our output buffer to use it. Return a flag that tells
539    /// you whether such an update was carried out.
540    ///
541    /// Bear in mind that when this happens, you will lose any change that you
542    /// performed to the output buffer via the
543    /// [`output_buffer_mut()`](Output::output_buffer_mut) interface.
544    pub fn update(&mut self) -> bool {
545        // Check if an update is present in the back-buffer
546        let updated = self.updated();
547        if updated {
548            // Access the shared state
549            let shared_state = &(*self.shared);
550
551            // If so, exchange our output buffer with the back-buffer, thusly
552            // acquiring exclusive access to the old back buffer while giving
553            // the producer a new back-buffer to write to.
554            //
555            // The ordering must be AcqRel, because...
556            //
557            // - Our accesses to the previous buffer must not be reordered after
558            //   this operation (which mandates Release ordering), otherwise
559            //   they could race with the producer accessing the freshly
560            //   liberated buffer.
561            // - Our accesses from the buffer must not be reordered before this
562            //   operation (which mandates Consume ordering, that is best
563            //   approximated by Acquire in Rust), otherwise they would race
564            //   with the producer writing into the buffer before publishing it.
565            //   * This reordering may seem paradoxical, but could happen if the
566            //     compiler or CPU correctly speculated the new buffer's index
567            //     before that index is actually read, as well as on weird hardware
568            //     like GPUs where CPU caches require manual synchronization.
569            //
570            let former_back_info = shared_state
571                .back_info
572                .swap(self.output_idx, Ordering::AcqRel);
573
574            // Make the old back-buffer our new output buffer
575            self.output_idx = former_back_info & BACK_INDEX_MASK;
576        }
577
578        // Tell whether an update was carried out
579        updated
580    }
581}

583/// Triple buffer shared state
584///
585/// In a triple buffering communication protocol, the producer and consumer
586/// share the following storage:
587///
588/// - Three memory buffers suitable for storing the data at hand
589/// - Information about the back-buffer: which buffer is the current back-buffer
590///   and whether an update was published since the last readout.
591#[derive(Debug)]
592struct SharedState<T: Send> {
593    /// Data storage buffers
594    buffers: [CachePadded<UnsafeCell<T>>; 3],
595
596    /// Information about the current back-buffer state
597    back_info: CachePadded<AtomicBackBufferInfo>,
598}
599//
600#[doc(hidden)]
601impl<T: Send> SharedState<T> {
602    /// Given (a way to generate) buffer contents and the back info, build the shared state
603    fn new(mut gen_buf_data: impl FnMut(usize) -> T, back_info: BackBufferInfo) -> Self {
604        let mut make_buf = |i| -> CachePadded<UnsafeCell<T>> {
605            CachePadded::new(UnsafeCell::new(gen_buf_data(i)))
606        };
607        Self {
608            buffers: [make_buf(0), make_buf(1), make_buf(2)],
609            back_info: CachePadded::new(AtomicBackBufferInfo::new(back_info)),
610        }
611    }
612}
613//
614#[doc(hidden)]
615impl<T: Clone + Send> SharedState<T> {
616    /// Cloning the shared state is unsafe because you must ensure that no one
617    /// is concurrently accessing it, since &self is enough for writing.
618    unsafe fn clone(&self) -> Self {
619        Self::new(
620            |i| (*self.buffers[i].get()).clone(),
621            self.back_info.load(Ordering::Relaxed),
622        )
623    }
624}
625//
626#[doc(hidden)]
627impl<T: PartialEq + Send> SharedState<T> {
628    /// Equality is unsafe for the same reason as cloning: you must ensure that
629    /// no one is concurrently accessing the triple buffer to avoid data races.
630    unsafe fn eq(&self, other: &Self) -> bool {
631        // Check whether the contents of all buffers are equal...
632        let buffers_equal = self
633            .buffers
634            .iter()
635            .zip(other.buffers.iter())
636            .all(|tuple| -> bool {
637                let (cell1, cell2) = tuple;
638                *cell1.get() == *cell2.get()
639            });
640
641        // ...then check whether the rest of the shared state is equal
642        buffers_equal
643            && (self.back_info.load(Ordering::Relaxed) == other.back_info.load(Ordering::Relaxed))
644    }
645}

649// Index types used for triple buffering
650//
651// These types are used to index into triple buffers. In addition, the
652// BackBufferInfo type is actually a bitfield, whose third bit (numerical
653// value: 4) is set to 1 to indicate that the producer published an update into
654// the back-buffer, and reset to 0 when the consumer fetches the update.
655//

1201    /// Check the state of a buffer, and the effect of queries on it
1202    fn check_buf_state<T>(buf: &mut TripleBuffer<T>, expected_dirty_bit: bool)
1203    where
1204        T: Clone + Debug + PartialEq + Send,
1205    {
1206        // Make a backup of the buffer's initial state
1207        let initial_buf = buf.clone();
1208
1209        // Check that the input and output point to the same shared state
1210        assert_eq!(as_ptr(&buf.input.shared), as_ptr(&buf.output.shared));
1211
1212        // Access the shared state and decode back-buffer information
1213        let back_info = buf.input.shared.back_info.load(Ordering::Relaxed);
1214        let back_idx = back_info & BACK_INDEX_MASK;
1215        let back_buffer_dirty = back_info & BACK_DIRTY_BIT != 0;
1216
1217        // Input-/output-/back-buffer indexes must be in range
1218        assert!(index_in_range(buf.input.input_idx));
1219        assert!(index_in_range(buf.output.output_idx));
1220        assert!(index_in_range(back_idx));
1221
1222        // Input-/output-/back-buffer indexes must be distinct
1223        assert!(buf.input.input_idx != buf.output.output_idx);
1224        assert!(buf.input.input_idx != back_idx);
1225        assert!(buf.output.output_idx != back_idx);
1226
1227        // Back-buffer must have the expected dirty bit
1228        assert_eq!(back_buffer_dirty, expected_dirty_bit);
1229
1230        // Check that the "input buffer" query behaves as expected
1231        assert_eq!(
1232            as_ptr(&buf.input.input_buffer_mut()),
1233            buf.input.shared.buffers[buf.input.input_idx as usize].get()
1234        );
1235        assert_eq!(*buf, initial_buf);
1236
1237        // Check that the "consumed" query behaves as expected
1238        assert_eq!(!buf.input.consumed(), expected_dirty_bit);
1239        assert_eq!(*buf, initial_buf);
1240
1241        // Check that the output_buffer query works in the initial state
1242        assert_eq!(
1243            as_ptr(&buf.output.peek_output_buffer()),
1244            buf.output.shared.buffers[buf.output.output_idx as usize].get()
1245        );
1246        assert_eq!(*buf, initial_buf);
1247
1248        // Check that the output buffer query works in the initial state
1249        assert_eq!(buf.output.updated(), expected_dirty_bit);
1250        assert_eq!(*buf, initial_buf);
1251    }
