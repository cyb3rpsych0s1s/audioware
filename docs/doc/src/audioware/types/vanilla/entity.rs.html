81pub trait AsEntity {

90impl AsEntity for Ref<Entity> {
91    fn get_entity_id(&self) -> EntityId {
92        if self.is_null() {
93            return EntityId::default();
94        }
95        unsafe { self.instance() }.unwrap().entity_id
96    }
97    fn get_world_position(&self) -> Vector4 {
98        let rtti = RttiSystem::get();
99        let cls = rtti.get_class(CName::new(Entity::NAME)).unwrap();
100        let method: &Method = cls.get_method(CName::new("GetWorldPosition")).ok().unwrap();
101        match unsafe { self.instance() } {
102            Some(x) if x.status == EntityStatus::Attached => method
103                .as_function()
104                .execute::<_, Vector4>(Some(x.as_ref()), ())
105                .unwrap(),
106            _ => Vec3::ZERO.into(),
107        }
108    }
109
110    fn get_world_forward(&self) -> Vector4 {
111        let rtti = RttiSystem::get();
112        let cls = rtti.get_class(CName::new(Entity::NAME)).unwrap();
113        let method: &Method = cls.get_method(CName::new("GetWorldForward")).ok().unwrap();
114        match unsafe { self.instance() } {
115            Some(x) if x.status == EntityStatus::Attached => method
116                .as_function()
117                .execute::<_, Vector4>(Some(x.as_ref()), ())
118                .unwrap(),
119            _ => Vec3::NEG_Z.into(),
120        }
121    }
122
123    fn get_world_orientation(&self) -> Quaternion {
124        let rtti = RttiSystem::get();
125        let cls = rtti.get_class(CName::new(Entity::NAME)).unwrap();
126        let method: &Method = cls
127            .get_method(CName::new("GetWorldOrientation"))
128            .ok()
129            .unwrap();
130        match unsafe { self.instance() } {
131            Some(x) if x.status == EntityStatus::Attached => method
132                .as_function()
133                .execute::<_, Quaternion>(Some(x.as_ref()), ())
134                .unwrap(),
135            _ => Quat::IDENTITY.into(),
136        }
137    }
138
139    fn get_world_transform(&self) -> WorldTransform {
140        let rtti = RttiSystem::get();
141        let cls = rtti.get_class(CName::new(Entity::NAME)).unwrap();
142        let method: &Method = cls
143            .get_method(CName::new("GetWorldTransform;"))
144            .ok()
145            .unwrap();
146        match unsafe { self.instance() } {
147            Some(x) if x.status == EntityStatus::Attached => method
148                .as_function()
149                .execute::<_, WorldTransform>(Some(x.as_ref()), ())
150                .unwrap(),
151            _ => WorldTransform::default(),
152        }
153    }
154
155    fn is_in_workspot(&self) -> bool {
156        self.clone()
157            .cast::<GameObject>()
158            .map(AIActionHelper::is_in_workspot)
159            .unwrap_or(false)
160    }
