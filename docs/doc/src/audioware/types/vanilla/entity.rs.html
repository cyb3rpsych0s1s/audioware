83pub trait AsEntity {

92impl AsEntity for Ref<Entity> {
93    fn get_entity_id(&self) -> EntityId {
94        if self.is_null() {
95            return EntityId::default();
96        }
97        unsafe { self.instance() }.unwrap().entity_id
98    }
99    fn get_world_position(&self) -> Vector4 {
100        let rtti = RttiSystem::get();
101        let cls = rtti.get_class(CName::new(Entity::NAME)).unwrap();
102        let method: &Method = cls.get_method(CName::new("GetWorldPosition")).ok().unwrap();
103        match unsafe { self.instance() } {
104            Some(x) if x.status == EntityStatus::Attached => method
105                .as_function()
106                .execute::<_, Vector4>(Some(x.as_ref()), ())
107                .unwrap(),
108            _ => Vec3::ZERO.into(),
109        }
110    }
111
112    fn get_world_forward(&self) -> Vector4 {
113        let rtti = RttiSystem::get();
114        let cls = rtti.get_class(CName::new(Entity::NAME)).unwrap();
115        let method: &Method = cls.get_method(CName::new("GetWorldForward")).ok().unwrap();
116        match unsafe { self.instance() } {
117            Some(x) if x.status == EntityStatus::Attached => method
118                .as_function()
119                .execute::<_, Vector4>(Some(x.as_ref()), ())
120                .unwrap(),
121            _ => Vec3::NEG_Z.into(),
122        }
123    }
124
125    fn get_world_orientation(&self) -> Quaternion {
126        let rtti = RttiSystem::get();
127        let cls = rtti.get_class(CName::new(Entity::NAME)).unwrap();
128        let method: &Method = cls
129            .get_method(CName::new("GetWorldOrientation"))
130            .ok()
131            .unwrap();
132        match unsafe { self.instance() } {
133            Some(x) if x.status == EntityStatus::Attached => method
134                .as_function()
135                .execute::<_, Quaternion>(Some(x.as_ref()), ())
136                .unwrap(),
137            _ => Quat::IDENTITY.into(),
138        }
139    }
140
141    fn get_world_transform(&self) -> WorldTransform {
142        let rtti = RttiSystem::get();
143        let cls = rtti.get_class(CName::new(Entity::NAME)).unwrap();
144        let method: &Method = cls
145            .get_method(CName::new("GetWorldTransform;"))
146            .ok()
147            .unwrap();
148        match unsafe { self.instance() } {
149            Some(x) if x.status == EntityStatus::Attached => method
150                .as_function()
151                .execute::<_, WorldTransform>(Some(x.as_ref()), ())
152                .unwrap(),
153            _ => WorldTransform::default(),
154        }
155    }
156
157    fn is_in_workspot(&self) -> bool {
158        self.clone()
159            .cast::<GameObject>()
160            .map(AIActionHelper::is_in_workspot)
161            .unwrap_or(false)
162    }
