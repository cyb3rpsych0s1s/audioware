1use audioware_manifest::ScnDialogLineType;
2use red4ext_rs::{
3    RttiSystem,
4    types::{CName, EntityId},
5};
6
7/// Trigger subtitles display in Redscript from Rust.
8pub fn propagate_subtitles(
9    reaction: CName,
10    entity_id: EntityId,
11    emitter_name: CName,
12    line_type: ScnDialogLineType,
13    duration: f32,
14) {
15    let rtti = RttiSystem::get();
16    let methods = rtti.get_global_functions();
17    let method = methods
18        .iter()
19        .find(|x| {
20            x.name()
21                == CName::new(
22                    "Audioware.PropagateSubtitle;CNameEntityIDCNamescnDialogLineTypeFloat",
23                )
24        })
25        .unwrap();
26    method
27        .execute::<_, ()>(
28            None,
29            (reaction, entity_id, emitter_name, line_type, duration),
30        )
31        .unwrap();
32}
