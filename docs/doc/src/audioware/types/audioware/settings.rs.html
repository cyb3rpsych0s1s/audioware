10/// Interop type for [SpatialTrackSettings](audioware_core::SpatialTrackSettings).

39impl Hash for EmitterSettings {
40    fn hash<H: std::hash::Hasher>(&self, state: &mut H) {
41        let distance = self
42            .distances
43            .is_null()
44            .not()
45            .then(|| unsafe { self.distances.fields() }.unwrap().clone());
46        distance.hash(state);
47        match self
48            .attenuation_function
49            .is_null()
50            .not()
51            .then(|| self.attenuation_function.clone())
52        {
53            None => {
54                None::<crate::Tween>.hash(state);
55            }
56            Some(x) if x.is_a::<LinearTween>() => {
57                let x = unsafe { std::mem::transmute::<&Ref<Tween>, &Ref<LinearTween>>(&x) };
58                unsafe { x.fields() }.unwrap().hash(state);
59            }
60            Some(x) if x.is_a::<ElasticTween>() => {
61                let x = unsafe { std::mem::transmute::<&Ref<Tween>, &Ref<ElasticTween>>(&x) };
62                unsafe { x.fields() }.unwrap().hash(state);
63            }
64            _ => {
65                unreachable!("unknown attenuation function");
66            }
67        };
68
69        self.enable_spatialization.hash(state);
70        self.persist_until_sounds_finish.hash(state);
71    }
72}

74impl PartialEq for EmitterSettings {
75    fn eq(&self, other: &Self) -> bool {
76        match (
77            self.attenuation_function.clone().into_easing(),
78            other.attenuation_function.clone().into_easing(),
79        ) {
80            (Some(_), None) | (None, Some(_)) => return false,
81            (Some(x), Some(y)) if x != y => return false,
82            _ => {}
83        };
84        match (
85            self.distances
86                .clone()
87                .is_null()
88                .not()
89                .then(|| unsafe { self.distances.fields() })
90                .flatten(),
91            other
92                .distances
93                .clone()
94                .is_null()
95                .not()
96                .then(|| unsafe { self.distances.fields() })
97                .flatten(),
98        ) {
99            (None, Some(_)) | (Some(_), None) => return false,
100            (Some(x), Some(y)) if x != y => return false,
101            _ => {}
102        };
103        self.enable_spatialization == other.enable_spatialization
104    }
105}

152impl Hash for EmitterDistances {
