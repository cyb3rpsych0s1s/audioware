56pub trait AsGameInstance {
57    /// `public static native func FindEntityByID(self: GameInstance, entityId: EntityID) -> ref<Entity>`
58    fn find_entity_by_id(game: GameInstance, entity_id: EntityId) -> Ref<Entity>;
59    fn get_audio_system() -> Ref<AudioSystem>;
60    fn get_scene_system() -> Ref<SceneSystem>;
61    fn get_world_state_system() -> Ref<WorldStateSystem>;
62    fn get_static_entity_system() -> Ref<StaticEntitySystem>;
63    fn get_dynamic_entity_system() -> Ref<DynamicEntitySystem>;
64}
65
66impl AsGameInstance for GameInstance {
67    fn find_entity_by_id(game: GameInstance, entity_id: EntityId) -> Ref<Entity> {
68        let rtti = RttiSystem::get();
69        let cls = rtti.get_class(CName::new(GameInstance::NAME)).unwrap();
70        let methods = cls.static_methods();
71        let method = methods
72            .iter()
73            .find(|x| x.as_function().name() == CName::new("FindEntityByID"))
74            .unwrap();
75        method
76            .as_function()
77            .execute::<_, Ref<Entity>>(None, (game, entity_id))
78            .unwrap()
79    }
80
81    fn get_audio_system() -> Ref<AudioSystem> {
82        let rtti = RttiSystem::get();
83        let class = rtti.get_class(CName::new(AudioSystem::NAME)).unwrap();
84        let engine = GameEngine::get();
85        let game = engine.game_instance();
86        game.get_system(class.as_type())
87            .cast::<AudioSystem>()
88            .unwrap_or_default()
89    }
90
91    fn get_scene_system() -> Ref<SceneSystem> {
92        let rtti = RttiSystem::get();
93        let class = rtti.get_class(CName::new(SceneSystem::NAME)).unwrap();

124        let rtti = RttiSystem::get();

133    }
134}
135
136pub fn resolve_any_entity(entity_id: EntityId) -> Ref<Entity> {
137    let game = GameInstance::new();
138    let entity = GameInstance::find_entity_by_id(game, entity_id);
139    if !entity.is_null() {
140        return entity;
141    }
142    let statics = GameInstance::get_static_entity_system();
143    if statics.is_ready() {
144        let entity = statics.get_entity(entity_id);
145        if !entity.is_null() {
146            return entity;
147        }
148    }
149    let dynamics = GameInstance::get_dynamic_entity_system();
150    if dynamics.is_ready() {
151        let entity = dynamics.get_entity(entity_id);
152        if !entity.is_null() {
153            return entity;
154        }
155    }
156    Ref::default()
157}
