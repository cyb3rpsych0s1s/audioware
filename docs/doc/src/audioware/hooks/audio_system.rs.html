39unsafe extern "C" fn detour_play(
40    i: *mut IScriptable,
41    f: *mut StackFrame,
42    a3: VoidPtr,
43    a4: VoidPtr,
44    cb: unsafe extern "C" fn(i: *mut IScriptable, f: *mut StackFrame, a3: VoidPtr, a4: VoidPtr),
45) {
46    unsafe {
47        let frame = &mut *f;
48        let state = frame.args_state();
49
50        let event_name: CName = StackFrame::get_arg(frame);
51        let entity_id: EntityId = StackFrame::get_arg(frame);
52        let emitter_name: CName = StackFrame::get_arg(frame);
53
54        if Engine::<CpalBackend>::exists(&event_name) {
55            crate::utils::intercept!("AudioSystem.Play: intercepted {event_name}");
56            crate::engine::queue::send(Command::PlayVanilla {
57                event_name,
58                entity_id: entity_id.into(),
59                emitter_name: emitter_name.into(),
60            });
61        } else {
62            frame.restore_args(state);
63            cb(i, f, a3, a4);
64        }
65    }
66}

97unsafe extern "C" fn detour_switch(
98    i: *mut IScriptable,
99    f: *mut StackFrame,
100    a3: VoidPtr,
101    a4: VoidPtr,
102    cb: unsafe extern "C" fn(i: *mut IScriptable, f: *mut StackFrame, a3: VoidPtr, a4: VoidPtr),
103) {
104    unsafe {
105        let frame = &mut *f;
106        let state = frame.args_state();
107
108        let switch_name: CName = StackFrame::get_arg(frame);
109        let switch_value: CName = StackFrame::get_arg(frame);
110        let entity_id: EntityId = StackFrame::get_arg(frame);
111        let emitter_name: CName = StackFrame::get_arg(frame);
112
113        let prev = Engine::<CpalBackend>::exists(&switch_name);
114        let next = Engine::<CpalBackend>::exists(&switch_value);
115
116        if prev || next {
117            crate::utils::intercept!(
118                "AudioSystem.Switch: intercepted {switch_name}/{switch_value}"
119            );
120
121            crate::engine::queue::send(Command::SwitchVanilla {
122                switch_name,
123                switch_value,
124                entity_id: entity_id.into(),
125                emitter_name: emitter_name.into(),
126            });
127        } else {
128            frame.restore_args(state);
129            cb(i, f, a3, a4);
130        }
131    }
