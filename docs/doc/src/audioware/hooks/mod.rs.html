138#[macro_export]
139macro_rules! attach_native_func {
140    ($name:literal, $offset:path, $hook: ident, $me:ident, $to:ident $(, $v:vis)?) => {
141        ::red4ext_rs::hooks! {
142            static $hook: fn(
143                i: *mut ::red4ext_rs::types::IScriptable,
144                f: *mut ::red4ext_rs::types::StackFrame,
145                a3: ::red4ext_rs::VoidPtr,
146                a4: ::red4ext_rs::VoidPtr) -> ();
147        }
148
149        #[allow(clippy::missing_transmute_annotations)]
150        $($v)? fn $me(env: &::red4ext_rs::SdkEnv) {
151            let addr = ::red4ext_rs::addr_hashes::resolve($offset);
152            let addr = unsafe { ::std::mem::transmute(addr) };
153            unsafe { env.attach_hook($hook, addr, $to) };
154            $crate::utils::intercept!("attached native func hook for {}", $name);
155        }
156    };
157    ($name:literal, $offset:path) => {
158        attach_native_func!($name, $offset, HOOK, attach_hook, detour, pub);
159    };
160}

162#[macro_export]
163macro_rules! attach_native_event {
164    ($offset:path, $class:path, $to:ident $(, $v:vis)?) => {
165        ::red4ext_rs::hooks! {
166            static HOOK: fn(
167                a1: *mut ::red4ext_rs::types::IScriptable,
168                a2: *mut $class) -> ();
169        }
170
171        #[allow(clippy::missing_transmute_annotations)]
172        $($v)? fn attach_hook(env: &::red4ext_rs::SdkEnv) {
173            let addr = ::red4ext_rs::addr_hashes::resolve($offset);
174            let addr = unsafe { ::std::mem::transmute(addr) };
175            unsafe { env.attach_hook(HOOK, addr, $to) };
176            $crate::utils::intercept!("attached native event hook for {}", <$class as ::red4ext_rs::ScriptClass>::NAME);
177        }
178    };
179    ($offset:path, $class:path) => {
180        attach_native_event!($offset, $class, detour, pub);
181    };
