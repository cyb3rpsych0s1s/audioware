181#[macro_export]
182macro_rules! attach_native_func {
183    ($name:literal, $offset:path, $hook: ident, $me:ident, $em:ident, $to:ident $(, $v:vis)?) => {
184        ::red4ext_rs::hooks! {
185            static $hook: fn(
186                i: *mut ::red4ext_rs::types::IScriptable,
187                f: *mut ::red4ext_rs::types::StackFrame,
188                a3: ::red4ext_rs::VoidPtr,
189                a4: ::red4ext_rs::VoidPtr) -> ();
190        }
191
192        #[allow(clippy::missing_transmute_annotations)]
193        $($v)? fn $me(env: &::red4ext_rs::SdkEnv) {
194            let addr = ::red4ext_rs::addr_hashes::resolve($offset);
195            let addr = unsafe { ::std::mem::transmute(addr) };
196            unsafe { env.attach_hook($hook, addr, $to) };
197            $crate::utils::intercept!("attached native func hook for {}", $name);
198        }
199        $($v)? fn $em(env: &::red4ext_rs::SdkEnv) {
200            let addr = ::red4ext_rs::addr_hashes::resolve($offset);
201            let addr: unsafe extern "C" fn(
202                i: *mut ::red4ext_rs::types::IScriptable,
203                f: *mut ::red4ext_rs::types::StackFrame,
204                a3: ::red4ext_rs::VoidPtr,
205                a4: ::red4ext_rs::VoidPtr) -> () = unsafe { ::std::mem::transmute(addr) };
206            unsafe { env.detach_hook(addr) };
207            $crate::utils::intercept!("detached native func hook for {}", $name);
208        }
209    };
210    ($name:literal, $offset:path) => {
211        attach_native_func!($name, $offset, HOOK, attach_hook, detach_hook, detour, pub);
212    };
213}

215#[macro_export]
216macro_rules! attach_native_event {
217    ($offset:path, $class:path, $to:ident $(, $v:vis)?) => {
218        ::red4ext_rs::hooks! {
219            static HOOK: fn(
220                a1: *mut ::red4ext_rs::types::IScriptable,
221                a2: *mut $class) -> ();
222        }
223
224        #[allow(clippy::missing_transmute_annotations)]
225        $($v)? fn attach_hook(env: &::red4ext_rs::SdkEnv) {
226            let addr = ::red4ext_rs::addr_hashes::resolve($offset);
227            let addr = unsafe { ::std::mem::transmute(addr) };
228            unsafe { env.attach_hook(HOOK, addr, $to) };
229            $crate::utils::intercept!("attached native event hook for {}", <$class as ::red4ext_rs::ScriptClass>::NAME);
230        }
231        $($v)? fn detach_hook(env: &::red4ext_rs::SdkEnv) {
232            let addr = ::red4ext_rs::addr_hashes::resolve($offset);
233            let addr: unsafe extern "C" fn(
234                *mut ::red4ext_rs::types::IScriptable,
235                *mut $class) -> () = unsafe { ::std::mem::transmute(addr) };
236            unsafe { env.detach_hook(addr) };
237            $crate::utils::intercept!("detached native event hook for {}", <$class as ::red4ext_rs::ScriptClass>::NAME);
238        }
239    };
240    ($offset:path, $class:path) => {
241        attach_native_event!($offset, $class, detour, pub);
242    };
