117#[macro_export]
118macro_rules! attach_native_func {
119    ($name:literal, $offset:path, $hook: ident, $me:ident, $to:ident $(, $v:vis)?) => {
120        ::red4ext_rs::hooks! {
121            static $hook: fn(
122                i: *mut ::red4ext_rs::types::IScriptable,
123                f: *mut ::red4ext_rs::types::StackFrame,
124                a3: ::red4ext_rs::VoidPtr,
125                a4: ::red4ext_rs::VoidPtr) -> ();
126        }
127
128        #[allow(clippy::missing_transmute_annotations)]
129        $($v)? fn $me(env: &::red4ext_rs::SdkEnv) {
130            let addr = ::red4ext_rs::addr_hashes::resolve($offset);
131            let addr = unsafe { ::std::mem::transmute(addr) };
132            unsafe { env.attach_hook($hook, addr, $to) };
133            $crate::utils::intercept!("attached native func hook for {}", $name);
134        }
135    };
136    ($name:literal, $offset:path) => {
137        attach_native_func!($name, $offset, HOOK, attach_hook, detour, pub);
138    };
139}

141#[macro_export]
142macro_rules! attach_native_event {
143    ($offset:path, $class:path, $to:ident $(, $v:vis)?) => {
144        ::red4ext_rs::hooks! {
145            static HOOK: fn(
146                a1: *mut ::red4ext_rs::types::IScriptable,
147                a2: *mut $class) -> ();
148        }
149
150        #[allow(clippy::missing_transmute_annotations)]
151        $($v)? fn attach_hook(env: &::red4ext_rs::SdkEnv) {
152            let addr = ::red4ext_rs::addr_hashes::resolve($offset);
153            let addr = unsafe { ::std::mem::transmute(addr) };
154            unsafe { env.attach_hook(HOOK, addr, $to) };
155            $crate::utils::intercept!("attached native event hook for {}", <$class as ::red4ext_rs::ScriptClass>::NAME);
156        }
157    };
158    ($offset:path, $class:path) => {
159        attach_native_event!($offset, $class, detour, pub);
160    };
