10/// Reverb mix parameter.
11pub struct ReverbMix(TweenerHandle);
12impl Parameter for ReverbMix {
13    type Value = f32;
14
15    fn try_new<B: Backend>(
16        manager: &mut kira::AudioManager<B>,
17    ) -> Result<Self, crate::error::Error> {
18        let handle = manager.add_modulator(TweenerBuilder { initial_value: 0.0 })?;
19        Ok(Self(handle))
20    }
21
22    fn try_effect(&self) -> Result<impl EffectBuilder, crate::error::Error> {
23        Ok(ReverbBuilder::new()
24            .stereo_width(1.0)
25            .mix(Value::from_modulator(
26                &self.0,
27                Mapping {
28                    input_range: (0.0, 1.0),
29                    output_range: (Mix::DRY, Mix::WET),
30                    easing: kira::Easing::Linear,
31                },
32            )))
33    }
34
35    fn update(&mut self, value: Self::Value, tween: Tween) {
36        self.0.set(value as f64, tween);
37    }
