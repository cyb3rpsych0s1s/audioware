15impl Emitter {
16    pub fn infos(entity_id: EntityId) -> Result<(Vector4, bool), Error> {
17        let entity = resolve_any_entity(entity_id);
18        if entity.is_null() {
19            return Err(Error::Scene {
20                source: SceneError::MissingEmitter { entity_id },
21            });
22        }
23        let busy = entity
24            .clone()
25            .cast::<GameObject>()
26            .map(AIActionHelper::is_in_workspot)
27            .unwrap_or(false);
28        let position = entity.get_world_position();
29        Ok((position, busy))
30    }
31
32    #[allow(clippy::type_complexity)]
33    pub fn full_infos(
34        entity_id: EntityId,
35    ) -> Result<(Vector4, bool, Option<f32>, Option<SpatialTrackDistances>), Error> {
36        let (position, busy) = Self::infos(entity_id)?;
37        let entity = resolve_any_entity(entity_id);
38        if entity.is_null() {
39            return Err(Error::Scene {
40                source: SceneError::MissingEmitter { entity_id },
41            });
42        }
43        let distances = entity.get_emitter_distances();
44        if !entity.is_a::<TimeDilatable>() {
45            return Ok((position, busy, None, distances));
46        }
47        let dilation = entity
48            .clone()
49            .cast::<TimeDilatable>()
50            .as_ref()
51            .map(AsTimeDilatable::get_time_dilation_value);
52        Ok((position, busy, dilation, distances))
53    }
54
55    #[allow(clippy::type_complexity)]
56    pub fn actor_infos(

59        let entity = resolve_any_entity(entity_id);
60        if entity.is_null() {
61            return Err(Error::Scene {
62                source: SceneError::MissingEmitter { entity_id },
63            });
64        }
65        let position = entity.get_world_position();
66        let distances = entity.get_emitter_distances();
67        Ok((position, distances))
68    }
69    pub fn position(entity_id: EntityId) -> Result<Vector4, Error> {
