66    pub fn exists_tag(&self, entity_id: &EntityId, tag_name: &CName) -> bool {

72    #[allow(clippy::too_many_arguments)]
73    pub fn add_emitter(
74        &mut self,
75        handle: Spatial,
76        entity_id: EntityId,
77        tag_name: CName,
78        emitter_name: Option<CName>,
79        dilation: Option<f32>,
80        occlusion: Option<f32>,
81        last_known_position: Vector4,
82        busy: bool,
83        persist_until_sounds_finish: bool,
84    ) -> Result<(), Error> {
85        if self.exists_tag(&entity_id, &tag_name) {
86            warns!(
87                "emitter {entity_id} with tag name {} was already registered",
88                tag_name.as_str()
89            );
90            return Ok(());
91        }
92        let slot = EmitterSlot::new(handle, tag_name, emitter_name, persist_until_sounds_finish);
93        if let Some(mut slots) = self.entries.get_mut(&entity_id) {
94            slots.value_mut().insert(slot);
95        } else {
96            self.entries.insert(
97                entity_id,
98                EmitterSlots::new(slot, dilation, busy, last_known_position),
99            );
100        }
101        let mut next = vec![];
102        with_entries(|x| {
103            next = x.to_vec();
104        });
105        next.push((entity_id, (tag_name, occlusion)));
106        publish_entries(next);
107        lifecycle!(
108            "added emitter {entity_id} with tag name {}",
109            tag_name.as_str()
110        );
111        Ok(())
112    }
113    pub fn sync_emitters(&mut self) -> Result<(), Error> {
114        if self.entries.is_empty() {
115            return Ok(());
116        }
117        let mut next = vec![];
118        with_entries(|x| {
119            next = x.to_vec();
120        });
121        self.entries.retain(|k, v| {
122            if v.marked_for_death && !v.any_playing_handle() {
123                next.retain(|(id, _)| id != k);
124                return false;
125            }
126            let Ok((position, busy)) = Emitter::infos(*k) else {
127                next.retain(|(id, _)| id != k);
128                return false;
129            };
130            v.busy = busy;
131            v.last_known_position = position;
132            // weirdly enough if emitter is not updated, sound(s) won't update as expected.
133            // e.g. when listener moves but emitter stands still.
134            v.set_emitter_position(position);
135            if let Some((_, (_, Some(occlusion)))) = next.iter().find(|x| x.0 == *k) {
136                v.set_emitter_occlusion(*occlusion);
137            }
138            true
139        });
140        publish_entries(next);
141        Ok(())
142    }
143    pub fn unregister_emitter(&mut self, entity_id: &EntityId, tag_name: &CName) -> bool {
144        let mut last = false;
145        let mut removed = false;
146        if let Some(mut slots) = self.entries.get_mut(entity_id) {
147            removed = slots.value_mut().unregister_emitter(tag_name);
148            last = slots.is_empty();
149        }
150        if removed && last {
151            self.entries.remove(entity_id);
152        }
153        if removed {
154            let mut next = vec![];
155            with_entries(|x| {
156                next = x.to_vec();
157            });
158            next.retain(|(id, (name, _))| id != entity_id || name != tag_name);
159            publish_entries(next);
160        }
161        removed
162    }
163    pub fn on_emitter_dies(&mut self, entity_id: &EntityId) {
164        self.entries.retain(|k, v| {
165            if k == entity_id {
166                let mut retain = false;
167                v.slots.retain_mut(|x| {
168                    if x.persist_until_sounds_finish {
169                        retain = true;
170                        true
171                    } else {
172                        x.stop(IMMEDIATELY);
173                        false
174                    }
175                });
176                if retain {
177                    v.marked_for_death = true;
178                }
179                retain
180            } else {
181                true
182            }
183        });
184        let mut next = vec![];
185        let before = next.len();
186        with_entries(|x| {
187            next = x.to_vec();
188        });
189        next.retain(|(id, _)| id != entity_id);
190        let after = next.len();
191        if before != after {
192            publish_entries(next);
193        }
194    }
195    pub fn on_emitter_incapacitated(&mut self, entity_id: EntityId, tween: Tween) {
196        if let Some(mut slots) = self.entries.get_mut(&entity_id) {
197            for ref mut slot in slots.slots.iter_mut() {
198                if !slot.persist_until_sounds_finish {
199                    slot.stop(tween);
200                }
201            }
202        }

217    #[allow(clippy::too_many_arguments)]
218    pub fn play_on_emitter<T>(
219        &mut self,
220        key: &Id,
221        banks: &Banks,
222        event_name: CName,
223        entity_id: EntityId,
224        tag_name: CName,
225        ext: Option<T>,
226        control_id: Option<ControlId>,
227    ) -> Result<(f32, Option<CName>), Error>
228    where
229        Either<StaticSoundData, StreamingSoundData<FromFileError>>: With<Option<T>>,
230        T: AffectedByTimeDilation
231            + ValidateFor<Either<StaticSoundData, StreamingSoundData<FromFileError>>>,
232    {
233        let Some(mut slots) = self.get_mut(&entity_id) else {
234            return Err(SceneError::MissingEmitter { entity_id }.into());
235        };
236        let Some(slot) = slots.get_mut(&tag_name) else {
237            return Err(SceneError::MissingEmitter { entity_id }.into());
238        };
239        let data = banks.data(key);
240        if let Some(Err(e)) = ext.as_ref().map(|x| x.validate_for(&data)) {
241            return Err(Error::Validation { errors: e });
242        }
243        let dilatable = ext
244            .as_ref()
245            .map(AffectedByTimeDilation::affected_by_time_dilation)
246            .unwrap_or(true);
247        slot.play_and_store(event_name, dilatable, data.with(ext), control_id)
248            .map_err(|e| match e {
249                Either::Left(e) => Error::Engine {
250                    source: EngineError::Sound { source: e },
251                },
252                Either::Right(e) => Error::Engine {
253                    source: EngineError::FromFile { source: e },
254                },
255            })
256    }

266    pub fn is_registered_emitter(entity_id: EntityId, tag_name: Option<CName>) -> bool {
267        let mut registered = false;
268        with_entries(|x| {
269            registered = x.iter().any(|(id, (tag, _))| {
270                *id == entity_id && tag_name.map(|x| x == *tag).unwrap_or(true)
271            })
272        });

282    pub fn update_pending_occlusions(&mut self, entity_id: EntityId, factor: f32) {
283        if !self
284            .entries
285            .get(&entity_id)
286            .map(|x| x.any_occluded())
287            .unwrap_or(false)
288        {
289            return;
290        }
291        for x in self.pending_occlusions.iter_mut() {
292            if x.0 == entity_id {
293                x.1 = factor;
294                return;
295            }
296        }
297        self.pending_occlusions.push((entity_id, factor));
298    }

454        assert_ne!(hash_some_undefined, 0);
