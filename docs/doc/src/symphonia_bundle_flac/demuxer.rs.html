28/// The FLAC start of stream marker: "fLaC" in ASCII.
29const FLAC_STREAM_MARKER: [u8; 4] = *b"fLaC";

31/// Free Lossless Audio Codec (FLAC) native frame reader.
32pub struct FlacReader {
33    reader: MediaSourceStream,
34    metadata: MetadataLog,
35    tracks: Vec<Track>,
36    cues: Vec<Cue>,
37    index: Option<SeekIndex>,
38    first_frame_offset: u64,
39    parser: PacketParser,
40}
41
42impl FlacReader {
43    /// Reads all the metadata blocks, returning a fully populated `FlacReader`.
44    fn init_with_metadata(source: MediaSourceStream) -> Result<Self> {
45        let mut metadata_builder = MetadataBuilder::new();
46
47        let mut reader = source;
48        let mut tracks = Vec::new();
49        let mut cues = Vec::new();
50        let mut index = None;
51        let mut parser = Default::default();
52
53        loop {
54            let header = MetadataBlockHeader::read(&mut reader)?;
55
56            // Create a scoped bytestream to error if the metadata block read functions exceed the
57            // stated length of the block.
58            let mut block_stream = ScopedStream::new(&mut reader, u64::from(header.block_len));
59
60            match header.block_type {
61                MetadataBlockType::Application => {
62                    // TODO: Store vendor data.
63                    read_application_block(&mut block_stream, header.block_len)?;
64                }
65                // SeekTable blocks are parsed into a SeekIndex.
66                MetadataBlockType::SeekTable => {
67                    // Check if a SeekTable has already be parsed. If one has, then the file is
68                    // invalid, atleast for seeking. Either way, it's a violation of the
69                    // specification.
70                    if index.is_none() {
71                        let mut new_index = SeekIndex::new();
72                        read_seek_table_block(&mut block_stream, header.block_len, &mut new_index)?;
73                        index = Some(new_index);
74                    }
75                    else {
76                        return decode_error("flac: found more than one seek table block");
77                    }
78                }
79                // VorbisComment blocks are parsed into Tags.
80                MetadataBlockType::VorbisComment => {
81                    read_comment_block(&mut block_stream, &mut metadata_builder)?;
82                }
83                // Cuesheet blocks are parsed into Cues.
84                MetadataBlockType::Cuesheet => {
85                    read_cuesheet_block(&mut block_stream, &mut cues)?;
86                }
87                // Picture blocks are read as Visuals.
88                MetadataBlockType::Picture => {
89                    read_picture_block(&mut block_stream, &mut metadata_builder)?;
90                }
91                // StreamInfo blocks are parsed into Streams.
92                MetadataBlockType::StreamInfo => {
93                    read_stream_info_block(&mut block_stream, &mut tracks, &mut parser)?;
94                }
95                // Padding blocks are skipped.
96                MetadataBlockType::Padding => {
97                    block_stream.ignore_bytes(u64::from(header.block_len))?;
98                }
99                // Unknown block encountered. Skip these blocks as they may be part of a future
100                // version of FLAC, but  print a message.
101                MetadataBlockType::Unknown(id) => {
102                    block_stream.ignore_bytes(u64::from(header.block_len))?;
103                    info!("ignoring {} bytes of block width id={}.", header.block_len, id);
104                }
105            }
106
107            // If the stated block length is longer than the number of bytes from the block read,
108            // ignore the remaining unread data.
109            let block_unread_len = block_stream.bytes_available();
110
111            if block_unread_len > 0 {
112                info!("under read block by {} bytes.", block_unread_len);
113                block_stream.ignore_bytes(block_unread_len)?;
114            }
115
116            // Exit when the last header is read.
117            if header.is_last {
118                break;
119            }
120        }
121
122        // Commit any read metadata to the metadata log.
123        let mut metadata = MetadataLog::default();
124        metadata.push(metadata_builder.metadata());
125
126        // Synchronize the packet parser to the first audio frame.
127        let _ = parser.resync(&mut reader)?;
128
129        // The first frame offset is the byte offset from the beginning of the stream after all the
130        // metadata blocks have been read.
131        let first_frame_offset = reader.pos();
132
133        Ok(FlacReader { reader, metadata, tracks, cues, index, first_frame_offset, parser })
134    }
135}

153impl FormatReader for FlacReader {
154    fn try_new(mut source: MediaSourceStream, _options: &FormatOptions) -> Result<Self> {
155        // Read the first 4 bytes of the stream. Ideally this will be the FLAC stream marker.
156        let marker = source.read_quad_bytes()?;
157
158        if marker != FLAC_STREAM_MARKER {
159            return unsupported_error("flac: missing flac stream marker");
160        }
161
162        // Strictly speaking, the first metadata block must be a StreamInfo block. There is
163        // no technical need for this from the reader's point of view. Additionally, if the
164        // reader is fed a stream mid-way there is no StreamInfo block. Therefore, just read
165        // all metadata blocks and handle the StreamInfo block as it comes.
166        let flac = Self::init_with_metadata(source)?;
167
168        // Make sure that there is atleast one StreamInfo block.
169        if flac.tracks.is_empty() {
170            return decode_error("flac: no stream info block");
171        }
172
173        Ok(flac)
174    }
175
176    fn next_packet(&mut self) -> Result<Packet> {
177        self.parser.parse(&mut self.reader)
178    }
179
180    fn metadata(&mut self) -> Metadata<'_> {
181        self.metadata.metadata()
182    }
183
184    fn cues(&self) -> &[Cue] {
185        &self.cues
186    }
187
188    fn tracks(&self) -> &[Track] {
189        &self.tracks
190    }
191
192    fn seek(&mut self, _mode: SeekMode, to: SeekTo) -> Result<SeekedTo> {
193        if self.tracks.is_empty() {
194            return seek_error(SeekErrorKind::Unseekable);
195        }
196
197        let params = &self.tracks[0].codec_params;
198
199        // Get the timestamp of the desired audio frame.
200        let ts = match to {
201            // Frame timestamp given.
202            SeekTo::TimeStamp { ts, .. } => ts,
203            // Time value given, calculate frame timestamp from sample rate.
204            SeekTo::Time { time, .. } => {
205                // Use the sample rate to calculate the frame timestamp. If sample rate is not
206                // known, the seek cannot be completed.
207                if let Some(sample_rate) = params.sample_rate {
208                    TimeBase::new(1, sample_rate).calc_timestamp(time)
209                }
210                else {
211                    return seek_error(SeekErrorKind::Unseekable);
212                }
213            }
214        };
215
216        debug!("seeking to frame_ts={}", ts);
217
218        // If the total number of frames in the stream is known, verify the desired frame timestamp
219        // does not exceed it.
220        if let Some(n_frames) = params.n_frames {
221            if ts > n_frames {
222                return seek_error(SeekErrorKind::OutOfRange);
223            }
224        }
225
226        // If the reader supports seeking, coarsely seek to the nearest packet with a timestamp
227        // lower than the desired timestamp using a binary search.
228        if self.reader.is_seekable() {
229            // The range formed by start_byte_offset..end_byte_offset defines an area where the
230            // binary search for the packet containing the desired timestamp will be performed. The
231            // lower bound is set to the byte offset of the first frame, while the upper bound is
232            // set to the length of the stream.
233            let mut start_byte_offset = self.first_frame_offset;
234            let mut end_byte_offset =
235                self.reader.byte_len().ok_or(Error::SeekError(SeekErrorKind::Unseekable))?;
236
237            // If there is an index, use it to refine the binary search range.
238            if let Some(ref index) = self.index {
239                // Search the index for the timestamp. Adjust the search based on the result.
240                match index.search(ts) {
241                    // Search from the start of stream up-to an ending point.
242                    SeekSearchResult::Upper(upper) => {
243                        end_byte_offset = self.first_frame_offset + upper.byte_offset;
244                    }
245                    // Search from a starting point up-to the end of the stream.
246                    SeekSearchResult::Lower(lower) => {
247                        start_byte_offset = self.first_frame_offset + lower.byte_offset;
248                    }
249                    // Search between two points of the stream.
250                    SeekSearchResult::Range(lower, upper) => {
251                        start_byte_offset = self.first_frame_offset + lower.byte_offset;
252                        end_byte_offset = self.first_frame_offset + upper.byte_offset;
253                    }
254                    // Search the entire stream (default behaviour, so do nothing).
255                    SeekSearchResult::Stream => (),
256                }
257            }
258
259            // Binary search the range of bytes formed by start_by_offset..end_byte_offset for the
260            // desired frame timestamp. When the difference of the range reaches 2x the maximum
261            // frame size, exit the loop and search from the start_byte_offset linearly. The binary
262            // search becomes inefficient when the range is small.
263            while end_byte_offset - start_byte_offset > 2 * 8096 {
264                let mid_byte_offset = (start_byte_offset + end_byte_offset) / 2;
265                self.reader.seek(SeekFrom::Start(mid_byte_offset))?;
266
267                let sync = self.parser.resync(&mut self.reader)?;
268
269                if ts < sync.ts {
270                    end_byte_offset = mid_byte_offset;
271                }
272                else if ts >= sync.ts && ts < sync.ts + sync.dur {
273                    debug!("seeked to ts={} (delta={})", sync.ts, sync.ts as i64 - ts as i64);
274
275                    return Ok(SeekedTo { track_id: 0, actual_ts: sync.ts, required_ts: ts });
276                }
277                else {
278                    start_byte_offset = mid_byte_offset;
279                }
280            }
281
282            // The binary search did not find an exact frame, but the range has been narrowed. Seek
283            // to the start of the range, and continue with a linear search.
284            self.reader.seek(SeekFrom::Start(start_byte_offset))?;
285        }
286
287        // Linearly search the stream packet-by-packet for the packet that contains the desired
288        // timestamp. This search is used to find the exact packet containing the desired timestamp
289        // after the search range was narrowed by the binary search. It is also the ONLY way for a
290        // unseekable stream to be "seeked" forward.
291        let packet = loop {
292            let sync = self.parser.resync(&mut self.reader)?;
293
294            // The desired timestamp precedes the current packet's timestamp.
295            if ts < sync.ts {
296                // Attempted to seek backwards on an unseekable stream.
297                if !self.reader.is_seekable() {
298                    return seek_error(SeekErrorKind::ForwardOnly);
299                }
300                // Overshot a regular seek, or the stream is corrupted, not necessarily an error
301                // per-say.
302                else {
303                    break sync;
304                }
305            }
306            // The desired timestamp is contained within the current packet.
307            else if ts >= sync.ts && ts < sync.ts + sync.dur {
308                break sync;
309            }
310
311            // Advance the reader such that the next iteration will sync to a different frame.
312            self.reader.read_byte()?;
313        };
314
315        debug!("seeked to packet_ts={} (delta={})", packet.ts, packet.ts as i64 - ts as i64);
316
317        Ok(SeekedTo { track_id: 0, actual_ts: packet.ts, required_ts: ts })
318    }
319
320    fn into_inner(self: Box<Self>) -> MediaSourceStream {
321        self.reader
322    }
323}

325/// Reads a StreamInfo block and populates the reader with stream information.
326fn read_stream_info_block<B: ReadBytes + FiniteStream>(
327    reader: &mut B,
328    tracks: &mut Vec<Track>,
329    parser: &mut PacketParser,
330) -> Result<()> {
331    // Only one StreamInfo block, and therefore only one Track, is allowed per media source stream.
332    if tracks.is_empty() {
333        // Ensure the block length is correct for a stream information block before allocating a
334        // buffer for it.
335        if !StreamInfo::is_valid_size(reader.byte_len()) {
336            return decode_error("flac: invalid stream info block size");
337        }
338
339        // Read the stream information block as a boxed slice so that it may be attached as extra
340        // data on the codec parameters.
341        let extra_data = reader.read_boxed_slice_exact(reader.byte_len() as usize)?;
342
343        // Parse the stream info block.
344        let info = StreamInfo::read(&mut BufReader::new(&extra_data))?;
345
346        // Populate the codec parameters with the basic audio parameters of the track.
347        let mut codec_params = CodecParameters::new();
348
349        codec_params
350            .for_codec(CODEC_TYPE_FLAC)
351            .with_packet_data_integrity(true)
352            .with_extra_data(extra_data)
353            .with_sample_rate(info.sample_rate)
354            .with_time_base(TimeBase::new(1, info.sample_rate))
355            .with_bits_per_sample(info.bits_per_sample)
356            .with_channels(info.channels);
357
358        if let Some(md5) = info.md5 {
359            codec_params.with_verification_code(VerificationCheck::Md5(md5));
360        }
361
362        // Total samples per channel (the total number of frames) is optional.
363        if let Some(n_frames) = info.n_samples {
364            codec_params.with_n_frames(n_frames);
365        }
366
367        // Reset the packet parser.
368        parser.reset(info);
369
370        // Add the track.
371        tracks.push(Track::new(0, codec_params));
372    }
373    else {
374        return decode_error("flac: found more than one stream info block");
375    }
376
377    Ok(())
