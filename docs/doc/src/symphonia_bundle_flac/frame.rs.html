32/// `ChannelAssignment` describes the mapping between the samples decoded from a subframe and the
33/// channel those samples belong to. It is also through the `ChannelAssignment` that the decoder is
34/// instructed on how to decorrelate stereo channels.
35//
36/// For LeftSide or RightSide channel assignments, one channel is stored independently while the
37/// other stores a difference. The Difference is always stored as Left - Right. For the MidSide
38/// channel assignment, no channels are stored independently, rather, a Mid (average) channel and a
39/// Difference channel are stored.
40#[derive(Debug)]
41pub enum ChannelAssignment {
42    /// All channels are independantly coded and no decorrelation step is required.
43    Independant(u32),
44    /// Channel 0 is the Left channel, and channel 1 is a Difference channel. The Right channel
45    /// is restored by subtracting the Difference channel from the Left channel (R = L - D).
46    LeftSide,
47    /// Channel 0 is the Mid channel (Left/2 + Right/2), and channel 1 is the Difference channel
48    /// (Left - Right). Therefore, if M = L/2 + R/2 and D = L - R, solving for L and R the left
49    /// and right channels are: L = S/2 + M, and R = M - S/2.
50    MidSide,
51    /// Channel 0 is the Difference channel, and channel 1 is the Right channel. The Left channel
52    /// is restored by adding the Difference channel to the Right channel (L = R + D).
53    RightSide,
54}
55
56pub struct FrameHeader {
57    pub block_sequence: BlockSequence,
58    pub block_num_samples: u16,
59    pub channel_assignment: ChannelAssignment,
60    pub bits_per_sample: Option<u32>,
61    pub sample_rate: Option<u32>,
62}

64pub fn sync_frame<B: ReadBytes>(reader: &mut B) -> Result<u16> {
65    let mut sync = 0u16;
66
67    // Synchronize stream to Frame Header. FLAC specifies a byte-aligned 14 bit sync code of
68    // `0b11_1111_1111_1110`. This would be difficult to find on its own. Expand the search to
69    // a 16-bit field of `0b1111_1111_1111_10xx` and search a word at a time.
70    while (sync & 0xfffc) != 0xfff8 {
71        sync = sync.wrapping_shl(8) | u16::from(reader.read_u8()?);
72    }
73
74    Ok(sync)
75}

77pub fn read_frame_header<B: ReadBytes>(reader: &mut B, sync: u16) -> Result<FrameHeader> {
78    // The header is checksummed with a CRC8 hash. Include the sync code in this CRC.
79    let mut crc8 = Crc8Ccitt::new(0);
80    crc8.process_buf_bytes(&sync.to_be_bytes());
81
82    let mut reader_crc8 = MonitorStream::new(reader, crc8);
83
84    // Extract the blocking strategy from the expanded synchronization code.
85    let blocking_strategy = match sync & 0x1 {
86        0 => BlockingStrategy::Fixed,
87        _ => BlockingStrategy::Variable,
88    };
89
90    // Read all the standard frame description fields as one 16-bit value and extract the
91    // fields.
92    let desc = reader_crc8.read_be_u16()?;
93
94    let block_size_enc = u32::from((desc & 0xf000) >> 12);
95    let sample_rate_enc = u32::from((desc & 0x0f00) >> 8);
96    let channels_enc = u32::from((desc & 0x00f0) >> 4);
97    let bits_per_sample_enc = u32::from((desc & 0x000e) >> 1);
98
99    if (desc & 0x0001) == 1 {
100        return decode_error("flac: frame header reserved bit is not set to mandatory value");
101    }
102
103    let block_sequence = match blocking_strategy {
104        // Fixed-blocksize stream sequence blocks by a frame number.
105        BlockingStrategy::Fixed => {
106            let frame = match utf8_decode_be_u64(&mut reader_crc8)? {
107                Some(frame) => frame,
108                None => return decode_error("flac: frame sequence number is not valid"),
109            };
110
111            // The frame number should only be 31-bits. Since it is UTF8 encoded, the actual length
112            // cannot be enforced by the decoder. Return an error if the frame number exceeds the
113            // maximum 31-bit value.
114            if frame > 0x7fff_ffff {
115                return decode_error("flac: frame sequence number exceeds 31-bits");
116            }
117
118            BlockSequence::ByFrame(frame as u32)
119        }
120        // Variable-blocksize streams sequence blocks by a sample number.
121        BlockingStrategy::Variable => {
122            let sample = match utf8_decode_be_u64(&mut reader_crc8)? {
123                Some(sample) => sample,
124                None => return decode_error("flac: sample sequence number is not valid"),
125            };
126
127            // The sample number should only be 36-bits. Since it is UTF8 encoded, the actual length
128            // cannot be enforced by the decoder. Return an error if the frame number exceeds the
129            // maximum 36-bit value.
130            if sample > 0x000f_ffff_ffff {
131                return decode_error("flac: sample sequence number exceeds 36-bits");
132            }
133
134            BlockSequence::BySample(sample)
135        }
136    };
137
138    let block_num_samples = match block_size_enc {
139        0x1 => 192,
140        0x2..=0x5 => 576 * (1 << (block_size_enc - 2)),
141        0x6 => u16::from(reader_crc8.read_u8()?) + 1,
142        0x7 => {
143            let block_size = reader_crc8.read_be_u16()?;
144            if block_size == 0xffff {
145                return decode_error("flac: block size not allowed to be greater than 65535");
146            }
147            block_size + 1
148        }
149        0x8..=0xf => 256 * (1 << (block_size_enc - 8)),
150        _ => {
151            return decode_error("flac: block size set to reserved value");
152        }
153    };
154
155    let sample_rate = match sample_rate_enc {
156        0x0 => None, // Get from StreamInfo if possible.
157        0x1 => Some(88_200),
158        0x2 => Some(176_400),
159        0x3 => Some(192_000),
160        0x4 => Some(8_000),
161        0x5 => Some(16_000),
162        0x6 => Some(22_050),
163        0x7 => Some(24_000),
164        0x8 => Some(32_000),
165        0x9 => Some(44_100),
166        0xa => Some(48_000),
167        0xb => Some(96_000),
168        0xc => Some(u32::from(reader_crc8.read_u8()?)),
169        0xd => Some(u32::from(reader_crc8.read_be_u16()?)),
170        0xe => Some(u32::from(reader_crc8.read_be_u16()?) * 10),
171        _ => {
172            return decode_error("flac: sample rate set to reserved value");
173        }
174    };
175
176    if let Some(rate) = sample_rate {
177        if rate < 1 || rate > 655_350 {
178            return decode_error("flac: sample rate out of bounds");
179        }
180    }
181
182    let bits_per_sample = match bits_per_sample_enc {
183        0x0 => None, // Get from StreamInfo if possible.
184        0x1 => Some(8),
185        0x2 => Some(12),
186        0x4 => Some(16),
187        0x5 => Some(20),
188        0x6 => Some(24),
189        _ => {
190            return decode_error("flac: bits per sample set to reserved value");
191        }
192    };
193
194    let channel_assignment = match channels_enc {
195        0x0..=0x7 => ChannelAssignment::Independant(channels_enc + 1),
196        0x8 => ChannelAssignment::LeftSide,
197        0x9 => ChannelAssignment::RightSide,
198        0xa => ChannelAssignment::MidSide,
199        _ => {
200            return decode_error("flac: channel assignment set to reserved value");
201        }
202    };
203
204    // End of freame header, pop off CRC8 checksum.
205    let crc8_computed = reader_crc8.monitor().crc();
206
207    // Get expected CRC8 checksum from the header.
208    let crc8_expected = reader_crc8.into_inner().read_u8()?;
209
210    if crc8_expected != crc8_computed && cfg!(not(fuzzing)) {
211        return decode_error("flac: computed frame header CRC does not match expected CRC");
212    }
213
214    Ok(FrameHeader {
215        block_sequence,
216        block_num_samples,
217        channel_assignment,
218        bits_per_sample,
219        sample_rate,
220    })
221}

223/// A very quick check if the provided buffer is likely be a FLAC frame header.
224pub fn is_likely_frame_header(buf: &[u8]) -> bool {
225    // Minimum frame header size.
226    if buf.len() < FLAC_MIN_FRAME_HEADER_SIZE {
227        return false;
228    }
229
230    // First sync word.
231    if buf[0] != 0xff {
232        return false;
233    }
234
235    // Second sync word.
236    if (buf[1] & 0xfc) != 0xf8 {
237        return false;
238    }
239
240    // Reserved block size.
241    if (buf[2] & 0xf0) == 0x00 {
242        return false;
243    }
244
245    // Reserved sample rate.
246    if (buf[2] & 0x0f) == 0x0f {
247        return false;
248    }
249
250    // Reserved channel assignments 0xb to 0xf.
251    if ((buf[3] & 0xf0) >> 4) >= 0xb {
252        return false;
253    }
254
255    // Reserved sample size.
256    if (buf[3] & 0x0e == 0x6) || (buf[3] & 0x0e == 0x0e) {
257        return false;
258    }
259
260    // Reserved bit.
261    if buf[3] & 0x1 == 1 {
262        return false;
263    }
264
265    true
266}

268/// Decodes a big-endian unsigned integer encoded via extended UTF8. In this context, extended UTF8
269/// simply means the encoded UTF8 value may be up to 7 bytes for a maximum integer bit width of
270/// 36-bits.
271fn utf8_decode_be_u64<B: ReadBytes>(src: &mut B) -> Result<Option<u64>> {
272    // Read the first byte of the UTF8 encoded integer.
273    let mut state = u64::from(src.read_u8()?);
274
275    // UTF8 prefixes 1s followed by a 0 to indicate the total number of bytes within the multi-byte
276    // sequence. Using ranges, determine the mask that will overlap the data bits within the first
277    // byte of the sequence. For values 0-128, return the value immediately. If the value falls out
278    // of range return None as this is either not the start of a UTF8 sequence or the prefix is
279    // incorrect.
280    let mask: u8 = match state {
281        0x00..=0x7f => return Ok(Some(state)),
282        0xc0..=0xdf => 0x1f,
283        0xe0..=0xef => 0x0f,
284        0xf0..=0xf7 => 0x07,
285        0xf8..=0xfb => 0x03,
286        0xfc..=0xfd => 0x01,
287        0xfe => 0x00,
288        _ => return Ok(None),
289    };
290
291    // Obtain the data bits from the first byte by using the data mask.
292    state &= u64::from(mask);
293
294    // Read the remaining bytes within the UTF8 sequence. Since the mask 0s out the UTF8 prefix
295    // of 1s which indicate the length of the multi-byte sequence in bytes, plus an additional 0
296    // bit, the number of remaining bytes to read is the number of zeros in the mask minus 2.
297    // To avoid extra computation, simply loop from 2 to the number of zeros.
298    for _i in 2..mask.leading_zeros() {
299        // Each subsequent byte after the first in UTF8 is prefixed with 0b10xx_xxxx, therefore
300        // only 6 bits are useful. Append these six bits to the result by shifting the result left
301        // by 6 bit positions, and appending the next subsequent byte with the first two high-order
302        // bits masked out.
303        state = (state << 6) | u64::from(src.read_u8()? & 0x3f);
304
305        // TODO: Validation? Invalid if the byte is greater than 0x3f.
306    }
307
308    Ok(Some(state))
309}

326        assert_eq!(utf8_decode_be_u64(&mut stream).unwrap(), Some(8364));
