29pub struct WaveFormatChunk {
30    /// The number of channels.
31    pub n_channels: u16,
32    /// The sample rate in Hz. For non-PCM formats, this value must be interpreted as per the
33    /// format's specifications.
34    pub sample_rate: u32,
35    /// The required average data rate required in bytes/second. For non-PCM formats, this value
36    /// must be interpreted as per the format's specifications.
37    pub avg_bytes_per_sec: u32,
38    /// The byte alignment of one audio frame. For PCM formats, this is equal to
39    /// `(n_channels * format_data.bits_per_sample) / 8`. For non-PCM formats, this value must be
40    /// interpreted as per the format's specifications.
41    pub block_align: u16,
42    /// Extra data associated with the format block conditional upon the format tag.
43    pub format_data: FormatData,
44}

46impl WaveFormatChunk {
47    fn read_pcm_fmt<B: ReadBytes>(
48        reader: &mut B,
49        bits_per_sample: u16,
50        n_channels: u16,
51        len: u32,
52    ) -> Result<FormatData> {
53        // WaveFormat for a PCM format may be extended with an extra data length field followed by
54        // the extension data itself. Use the chunk length to determine if the format chunk is
55        // extended.
56        match len {
57            // Basic WavFormat struct, no extension.
58            16 => (),
59            // WaveFormatEx with extension data length field present, but no extension data.
60            18 => {
61                // Extension data length should be 0.
62                let _extension_len = reader.read_be_u16()?;
63            }
64            // WaveFormatEx with extension data length field present, and extension data.
65            40 => {
66                // Extension data length should be either 0 or 22 (if valid data is present).
67                let _extension_len = reader.read_u16()?;
68                reader.ignore_bytes(22)?;
69            }
70            _ => return decode_error("wav: malformed fmt_pcm chunk"),
71        }
72
73        // Bits per sample for PCM is both the encoded sample width, and the actual sample width.
74        // Strictly, this must either be 8 or 16 bits, but there is no reason why 24 and 32 bits
75        // can't be supported. Since these files do exist, allow for 8/16/24/32-bit samples, but
76        // error if not a multiple of 8 or greater than 32-bits.
77        //
78        // Select the appropriate codec using bits per sample. Samples are always interleaved and
79        // little-endian encoded for the PCM format.
80        let codec = match bits_per_sample {
81            8 => CODEC_TYPE_PCM_U8,
82            16 => CODEC_TYPE_PCM_S16LE,
83            24 => CODEC_TYPE_PCM_S24LE,
84            32 => CODEC_TYPE_PCM_S32LE,
85            _ => {
86                return decode_error(
87                    "wav: bits per sample for fmt_pcm must be 8, 16, 24 or 32 bits",
88                )
89            }
90        };
91
92        let channels = try_channel_count_to_mask(n_channels)?;
93        Ok(FormatData::Pcm(FormatPcm { bits_per_sample, channels, codec }))
94    }
95
96    fn read_adpcm_fmt<B: ReadBytes>(
97        reader: &mut B,
98        bits_per_sample: u16,
99        n_channels: u16,
100        len: u32,
101        codec: CodecType,
102    ) -> Result<FormatData> {
103        if bits_per_sample != 4 {
104            return decode_error("wav: bits per sample for fmt_adpcm must be 4 bits");
105        }
106
107        // WaveFormatEx with extension data length field present and with atleast frames per block data.
108        if len < 20 {
109            return decode_error("wav: malformed fmt_adpcm chunk");
110        }
111
112        let extra_size = reader.read_u16()? as u64;
113
114        match codec {
115            CODEC_TYPE_ADPCM_MS if extra_size < 32 => {
116                return decode_error("wav: malformed fmt_adpcm chunk");
117            }
118            CODEC_TYPE_ADPCM_IMA_WAV if extra_size != 2 => {
119                return decode_error("wav: malformed fmt_adpcm chunk");
120            }
121            _ => (),
122        }
123        reader.ignore_bytes(extra_size)?;
124
125        let channels = try_channel_count_to_mask(n_channels)?;
126        Ok(FormatData::Adpcm(FormatAdpcm { bits_per_sample, channels, codec }))
127    }

129    fn read_ieee_fmt<B: ReadBytes>(
130        reader: &mut B,
131        bits_per_sample: u16,
132        n_channels: u16,
133        len: u32,
134    ) -> Result<FormatData> {
135        // WaveFormat for a IEEE format should not be extended, but it may still have an extra data
136        // length parameter.
137        match len {
138            16 => (),
139            18 => {
140                let extra_size = reader.read_u16()?;
141                if extra_size != 0 {
142                    return decode_error("wav: extra data not expected for fmt_ieee chunk");
143                }
144            }
145            40 => {
146                // WAVEFORMATEXTENSIBLE is used for formats having more than two channels
147                // or higher sample resolutions than allowed by WAVEFORMATEX but for now
148                // we just ignore it
149                let _ = reader.ignore_bytes(40 - 16);
150            }
151            _ => return decode_error("wav: malformed fmt_ieee chunk"),
152        }
153
154        // Officially, only 32-bit floats are supported, but Symphonia can handle 64-bit floats.
155        //
156        // Select the appropriate codec using bits per sample. Samples are always interleaved and
157        // little-endian encoded for the IEEE Float format.
158        let codec = match bits_per_sample {
159            32 => CODEC_TYPE_PCM_F32LE,
160            64 => CODEC_TYPE_PCM_F64LE,
161            _ => return decode_error("wav: bits per sample for fmt_ieee must be 32 or 64 bits"),
162        };
163
164        let channels = try_channel_count_to_mask(n_channels)?;
165
166        Ok(FormatData::IeeeFloat(FormatIeeeFloat { channels, codec }))
167    }

169    fn read_ext_fmt<B: ReadBytes>(
170        reader: &mut B,
171        bits_per_coded_sample: u16,
172        n_channels: u16,
173        len: u32,
174    ) -> Result<FormatData> {
175        // WaveFormat for the extensible format must be extended to 40 bytes in length.
176        if len < 40 {
177            return decode_error("wav: malformed fmt_ext chunk");
178        }
179
180        let extra_size = reader.read_u16()?;
181
182        // The size of the extra data for the Extensible format is exactly 22 bytes.
183        if extra_size != 22 {
184            return decode_error("wav: extra data size not 22 bytes for fmt_ext chunk");
185        }
186
187        let bits_per_sample = reader.read_u16()?;
188
189        // Bits per coded sample for extensible formats is the width per sample as stored in the
190        // stream. This must be a multiple of 8.
191        if (bits_per_coded_sample & 0x7) != 0 {
192            return decode_error("wav: bits per coded sample for fmt_ext must be a multiple of 8");
193        }
194
195        // Bits per sample indicates the number of valid bits in the encoded sample. The sample is
196        // encoded in a bits per coded sample width value, therefore the valid number of bits must
197        // be at most bits per coded sample long.
198        if bits_per_sample > bits_per_coded_sample {
199            return decode_error(
200                "wav: bits per sample must be <= bits per coded sample for fmt_ext",
201            );
202        }
203
204        let channel_mask = fix_channel_mask(reader.read_u32()?, n_channels);
205
206        // Try to map channels.
207        let channels = match Channels::from_bits(channel_mask) {
208            Some(channels) => channels,
209            _ => return unsupported_error("wav: too many channels in mask for fmt_ext"),
210        };
211
212        let mut sub_format_guid = [0u8; 16];
213        reader.read_buf_exact(&mut sub_format_guid)?;
214
215        // These GUIDs identifiy the format of the data chunks. These definitions can be found in
216        // ksmedia.h of the Microsoft Windows Platform SDK.
217        #[rustfmt::skip]
218        const KSDATAFORMAT_SUBTYPE_PCM: [u8; 16] = [
219            0x01, 0x00, 0x00, 0x00, 0x00, 0x00, 0x10, 0x00,
220            0x80, 0x00, 0x00, 0xaa, 0x00, 0x38, 0x9b, 0x71,
221        ];
222        // #[rustfmt::skip]
223        // const KSDATAFORMAT_SUBTYPE_ADPCM: [u8; 16] = [
224        //     0x02, 0x00, 0x00, 0x00, 0x00, 0x00, 0x10, 0x00,
225        //     0x80, 0x00, 0x00, 0xaa, 0x00, 0x38, 0x9b, 0x71,
226        // ];
227        #[rustfmt::skip]
228        const KSDATAFORMAT_SUBTYPE_IEEE_FLOAT: [u8; 16] = [
229            0x03, 0x00, 0x00, 0x00, 0x00, 0x00, 0x10, 0x00,
230            0x80, 0x00, 0x00, 0xaa, 0x00, 0x38, 0x9b, 0x71,
231        ];
232        #[rustfmt::skip]
233        const KSDATAFORMAT_SUBTYPE_ALAW: [u8; 16] = [
234            0x06, 0x00, 0x00, 0x00, 0x00, 0x00, 0x10, 0x00,
235            0x80, 0x00, 0x00, 0xaa, 0x00, 0x38, 0x9b, 0x71,
236        ];
237        #[rustfmt::skip]
238        const KSDATAFORMAT_SUBTYPE_MULAW: [u8; 16] = [
239            0x04, 0x00, 0x00, 0x00, 0x00, 0x00, 0x10, 0x00,
240            0x80, 0x00, 0x00, 0xaa, 0x00, 0x38, 0x9b, 0x71,
241        ];
242
243        // Verify support based on the format GUID.
244        let codec = match sub_format_guid {
245            KSDATAFORMAT_SUBTYPE_PCM => {
246                // Only support up-to 32-bit integer samples.
247                if bits_per_coded_sample > 32 {
248                    return decode_error(
249                        "bits per sample for fmt_ext PCM sub-type must be <= 32 bits",
250                    );
251                }
252
253                // Use bits per coded sample to select the codec to use. If bits per sample is less
254                // than the bits per coded sample, the codec will expand the sample during decode.
255                match bits_per_coded_sample {
256                    8 => CODEC_TYPE_PCM_U8,
257                    16 => CODEC_TYPE_PCM_S16LE,
258                    24 => CODEC_TYPE_PCM_S24LE,
259                    32 => CODEC_TYPE_PCM_S32LE,
260                    _ => unreachable!(),
261                }
262            }
263            KSDATAFORMAT_SUBTYPE_IEEE_FLOAT => {
264                // IEEE floating formats do not support truncated sample widths.
265                if bits_per_sample != bits_per_coded_sample {
266                    return decode_error(
267                        "wav: bits per sample for fmt_ext IEEE sub-type must equal bits per coded sample"
268                    );
269                }
270
271                // Select the appropriate codec based on the bits per coded sample.
272                match bits_per_coded_sample {
273                    32 => CODEC_TYPE_PCM_F32LE,
274                    64 => CODEC_TYPE_PCM_F64LE,
275                    _ => {
276                        return decode_error(
277                            "wav: bits per sample for fmt_ext IEEE sub-type must be 32 or 64 bits",
278                        )
279                    }
280                }
281            }
282            KSDATAFORMAT_SUBTYPE_ALAW => CODEC_TYPE_PCM_ALAW,
283            KSDATAFORMAT_SUBTYPE_MULAW => CODEC_TYPE_PCM_MULAW,
284            _ => return unsupported_error("wav: unsupported fmt_ext sub-type"),
285        };
286
287        Ok(FormatData::Extensible(FormatExtensible {
288            bits_per_sample,
289            bits_per_coded_sample,
290            channels,
291            sub_format_guid,
292            codec,
293        }))
294    }

296    fn read_alaw_pcm_fmt<B: ReadBytes>(
297        reader: &mut B,
298        n_channels: u16,
299        len: u32,
300    ) -> Result<FormatData> {
301        if len != 18 {
302            return decode_error("wav: malformed fmt_alaw chunk");
303        }
304
305        let extra_size = reader.read_u16()?;
306
307        if extra_size > 0 {
308            reader.ignore_bytes(u64::from(extra_size))?;
309        }
310
311        let channels = try_channel_count_to_mask(n_channels)?;
312        Ok(FormatData::ALaw(FormatALaw { codec: CODEC_TYPE_PCM_ALAW, channels }))
313    }

334    pub(crate) fn packet_info(&self) -> Result<PacketInfo> {
335        match self.format_data {
336            FormatData::Adpcm(FormatAdpcm { codec, bits_per_sample, .. })
337            //| WaveFormatData::Extensible(WaveFormatExtensible { codec, bits_per_sample, .. })
338                if codec == CODEC_TYPE_ADPCM_MS =>
339            {
340                let frames_per_block = ((((self.block_align - (7 * self.n_channels)) * 8)
341                    / (bits_per_sample * self.n_channels))
342                    + 2) as u64;
343                PacketInfo::with_blocks(self.block_align, frames_per_block)
344            }
345            FormatData::Adpcm(FormatAdpcm { codec, bits_per_sample, .. })
346                if codec == CODEC_TYPE_ADPCM_IMA_WAV =>
347            {
348                let frames_per_block = (((self.block_align - (4 * self.n_channels)) * 8)
349                    / (bits_per_sample * self.n_channels)
350                    + 1) as u64;
351                PacketInfo::with_blocks(self.block_align, frames_per_block)
352            }
353            _ => Ok(PacketInfo::without_blocks(self.block_align)),
354        }
355    }

358impl ParseChunk for WaveFormatChunk {
359    fn parse<B: ReadBytes>(reader: &mut B, _tag: [u8; 4], len: u32) -> Result<WaveFormatChunk> {
360        // WaveFormat has a minimal length of 16 bytes. This may be extended with format specific
361        // data later.
362        if len < 16 {
363            return decode_error("wav: malformed fmt chunk");
364        }
365
366        let format = reader.read_u16()?;
367        let n_channels = reader.read_u16()?;
368        let sample_rate = reader.read_u32()?;
369        let avg_bytes_per_sec = reader.read_u32()?;
370        let block_align = reader.read_u16()?;
371        let bits_per_sample = reader.read_u16()?;
372
373        // The definition of these format identifiers can be found in mmreg.h of the Microsoft
374        // Windows Platform SDK.
375        const WAVE_FORMAT_PCM: u16 = 0x0001;
376        const WAVE_FORMAT_ADPCM: u16 = 0x0002;
377        const WAVE_FORMAT_IEEE_FLOAT: u16 = 0x0003;
378        const WAVE_FORMAT_ALAW: u16 = 0x0006;
379        const WAVE_FORMAT_MULAW: u16 = 0x0007;
380        const WAVE_FORMAT_ADPCM_IMA: u16 = 0x0011;
381        const WAVE_FORMAT_EXTENSIBLE: u16 = 0xfffe;
382
383        let format_data = match format {
384            // The PCM Wave Format
385            WAVE_FORMAT_PCM => Self::read_pcm_fmt(reader, bits_per_sample, n_channels, len),
386            // The Microsoft ADPCM Format
387            WAVE_FORMAT_ADPCM => {
388                Self::read_adpcm_fmt(reader, bits_per_sample, n_channels, len, CODEC_TYPE_ADPCM_MS)
389            }
390            // The IEEE Float Wave Format
391            WAVE_FORMAT_IEEE_FLOAT => Self::read_ieee_fmt(reader, bits_per_sample, n_channels, len),
392            // The Extensible Wave Format
393            WAVE_FORMAT_EXTENSIBLE => Self::read_ext_fmt(reader, bits_per_sample, n_channels, len),
394            // The Alaw Wave Format.
395            WAVE_FORMAT_ALAW => Self::read_alaw_pcm_fmt(reader, n_channels, len),
396            // The MuLaw Wave Format.
397            WAVE_FORMAT_MULAW => Self::read_mulaw_pcm_fmt(reader, n_channels, len),
398            // The IMA ADPCM Format
399            WAVE_FORMAT_ADPCM_IMA => Self::read_adpcm_fmt(
400                reader,
401                bits_per_sample,
402                n_channels,
403                len,
404                CODEC_TYPE_ADPCM_IMA_WAV,
405            ),
406            // Unsupported format.
407            _ => return unsupported_error("wav: unsupported wave format"),
408        }?;
409
410        Ok(WaveFormatChunk { n_channels, sample_rate, avg_bytes_per_sec, block_align, format_data })
411    }
412}

414impl fmt::Display for WaveFormatChunk {
415    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
416        writeln!(f, "WaveFormatChunk {{")?;
417        writeln!(f, "\tn_channels: {},", self.n_channels)?;
418        writeln!(f, "\tsample_rate: {} Hz,", self.sample_rate)?;
419        writeln!(f, "\tavg_bytes_per_sec: {},", self.avg_bytes_per_sec)?;
420        writeln!(f, "\tblock_align: {},", self.block_align)?;
421
422        match self.format_data {
423            FormatData::Pcm(ref pcm) => {
424                writeln!(f, "\tformat_data: Pcm {{")?;
425                writeln!(f, "\t\tbits_per_sample: {},", pcm.bits_per_sample)?;
426                writeln!(f, "\t\tchannels: {},", pcm.channels)?;
427                writeln!(f, "\t\tcodec: {},", pcm.codec)?;
428            }
429            FormatData::Adpcm(ref adpcm) => {
430                writeln!(f, "\tformat_data: Adpcm {{")?;
431                writeln!(f, "\t\tbits_per_sample: {},", adpcm.bits_per_sample)?;
432                writeln!(f, "\t\tchannels: {},", adpcm.channels)?;
433                writeln!(f, "\t\tcodec: {},", adpcm.codec)?;
434            }
435            FormatData::IeeeFloat(ref ieee) => {
436                writeln!(f, "\tformat_data: IeeeFloat {{")?;
437                writeln!(f, "\t\tchannels: {},", ieee.channels)?;
438                writeln!(f, "\t\tcodec: {},", ieee.codec)?;
439            }
440            FormatData::Extensible(ref ext) => {
441                writeln!(f, "\tformat_data: Extensible {{")?;
442                writeln!(f, "\t\tbits_per_sample: {},", ext.bits_per_sample)?;
443                writeln!(f, "\t\tbits_per_coded_sample: {},", ext.bits_per_coded_sample)?;
444                writeln!(f, "\t\tchannels: {},", ext.channels)?;
445                writeln!(f, "\t\tsub_format_guid: {:?},", &ext.sub_format_guid)?;
446                writeln!(f, "\t\tcodec: {},", ext.codec)?;
447            }
448            FormatData::ALaw(ref alaw) => {
449                writeln!(f, "\tformat_data: ALaw {{")?;
450                writeln!(f, "\t\tchannels: {},", alaw.channels)?;
451                writeln!(f, "\t\tcodec: {},", alaw.codec)?;
452            }
453            FormatData::MuLaw(ref mulaw) => {
454                writeln!(f, "\tformat_data: MuLaw {{")?;
455                writeln!(f, "\t\tchannels: {},", mulaw.channels)?;
456                writeln!(f, "\t\tcodec: {},", mulaw.codec)?;
457            }
458        };
459
460        writeln!(f, "\t}}")?;
461        writeln!(f, "}}")
462    }
463}

489pub struct ListChunk {
490    pub form: [u8; 4],
491    pub len: u32,
492}
493
494impl ListChunk {
495    pub fn skip<B: ReadBytes>(&self, reader: &mut B) -> Result<()> {
496        ChunksReader::<NullChunks>::new(self.len, ByteOrder::LittleEndian).finish(reader)
497    }
498}
499
500impl ParseChunk for ListChunk {
501    fn parse<B: ReadBytes>(reader: &mut B, _tag: [u8; 4], len: u32) -> Result<Self> {
502        // A List chunk must contain atleast the list/form identifier. However, an empty list
503        // (len == 4) is permissible.
504        if len < 4 {
505            return decode_error("wav: malformed list chunk");
506        }
507
508        Ok(ListChunk { form: reader.read_quad_bytes()?, len: len - 4 })
509    }
510}
511
512impl fmt::Display for ListChunk {
513    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
514        writeln!(f, "ListChunk {{")?;
515        writeln!(f, "\tform: {},", String::from_utf8_lossy(&self.form))?;
516        writeln!(f, "\tlen: {},", self.len)?;
517        writeln!(f, "}}")
518    }
519}

545pub enum RiffWaveChunks {
546    Format(ChunkParser<WaveFormatChunk>),
547    List(ChunkParser<ListChunk>),
548    Fact(ChunkParser<FactChunk>),
549    Data(ChunkParser<DataChunk>),
550}

552macro_rules! parser {
553    ($class:expr, $result:ty, $tag:expr, $len:expr) => {
554        Some($class(ChunkParser::<$result>::new($tag, $len)))
555    };
556}
557
558impl ParseChunkTag for RiffWaveChunks {
559    fn parse_tag(tag: [u8; 4], len: u32) -> Option<Self> {
560        match &tag {
561            b"fmt " => parser!(RiffWaveChunks::Format, WaveFormatChunk, tag, len),
562            b"LIST" => parser!(RiffWaveChunks::List, ListChunk, tag, len),
563            b"fact" => parser!(RiffWaveChunks::Fact, FactChunk, tag, len),
564            b"data" => parser!(RiffWaveChunks::Data, DataChunk, tag, len),
565            _ => None,
566        }
567    }
568}

574impl ParseChunkTag for RiffInfoListChunks {
575    fn parse_tag(tag: [u8; 4], len: u32) -> Option<Self> {
576        // Right now it is assumed all list chunks are INFO chunks, but that's not really
577        // guaranteed.
578        //
579        // TODO: Actually validate that the chunk is an info chunk.
580        parser!(RiffInfoListChunks::Info, InfoChunk, tag, len)
581    }
582}
583
584pub fn append_fact_params(codec_params: &mut CodecParameters, fact: &FactChunk) {
585    codec_params.with_n_frames(u64::from(fact.n_frames));
586}
587
588pub fn read_info_chunk(source: &mut MediaSourceStream, len: u32) -> Result<MetadataRevision> {
589    let mut info_list = ChunksReader::<RiffInfoListChunks>::new(len, ByteOrder::LittleEndian);
590
591    let mut metadata_builder = MetadataBuilder::new();
592
593    loop {
594        let chunk = info_list.next(source)?;
595
596        if let Some(RiffInfoListChunks::Info(info)) = chunk {
597            let parsed_info = info.parse(source)?;
598            metadata_builder.add_tag(parsed_info.tag);
599        }
600        else {
601            break;
602        }
603    }
604
605    info_list.finish(source)?;
606
607    Ok(metadata_builder.metadata())
608}
