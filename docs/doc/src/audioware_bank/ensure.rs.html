26/// Ensure no duplicate mod folder name across depots: `r6\audioware` and `mods`.

32) -> Result<(), Error> {
33    ensure!(
34        !redmod_exists || !mods.iter().any(|x| x.same_folder_name(r#mod.as_ref())),
35        DuplicateAcrossDepotsSnafu {

37        }
38    );
39    Ok(())

96/// Ensure [CName] does not already exist in [game pool](CNamePool),
97/// unless it already exists for some [locale](Locale).
98#[inline]
99pub fn ensure_localized_key_unique(
100    ids: &HashSet<Id>,
101    cname: &str,
102    _: Locale,
103) -> Result<bool, Error> {
104    // Conflict trait will make sure there's no identical duplicate, so no need to check twice.
105    if ids
106        .iter()
107        .any(|x| AsRef::<CName>::as_ref(x).as_str() == cname && x.locale().is_some())
108    {
109        return Ok(true);
110    }
111    ensure_key_unique(cname)?;
112    Ok(false)
113}

160/// Ensure [Manifest] does not contain duplicate keys among
161/// [Sfx],
162/// [Ono],
163/// etc.
164pub fn ensure_manifest_no_duplicates(manifest: &Manifest) -> Result<(), Error> {
165    let mut hashset = HashSet::with_capacity(100);
166    if let Some(sfx) = manifest.sfx.as_ref() {
167        for key in sfx.keys() {
168            ensure!(
169                hashset.insert(key.as_str()),
170                ConflictingKeySnafu { cname: key.clone() }
171            );
172        }
173    }
174    if let Some(onos) = manifest.onos.as_ref() {
175        for key in onos.keys() {
176            ensure!(
177                hashset.insert(key.as_str()),
178                ConflictingKeySnafu { cname: key.clone() }
179            );
180        }
181    }
182    if let Some(voices) = manifest.voices.as_ref() {
183        for key in voices.keys() {
184            ensure!(
185                hashset.insert(key.as_str()),
186                ConflictingKeySnafu { cname: key.clone() }
187            );
188        }
189    }
190    if let Some(music) = manifest.music.as_ref() {
191        for key in music.keys() {
192            ensure!(
193                hashset.insert(key.as_str()),
194                ConflictingKeySnafu { cname: key.clone() }
195            );
196        }
197    }
198    Ok(())
199}

201/// Ensure audio file [Path] is located inside [Mod] depot.
202pub fn ensure_located_in_depot(
203    file: &impl AsRef<std::path::Path>,
204    folder: &Mod,
205) -> Result<(), Error> {
206    let arg = std::fs::canonicalize(folder)?;
207    let path = arg.join(file);
208    if !path.starts_with(arg) {
209        return Err(Error::Validation {
210            source: validation::Error::AudioOutsideDepot {
211                path: path.display().to_string(),
212            },
213        });
214    }
215    Ok(())
216}

244pub fn ensure_valid_audio_data(
245    path: &impl AsRef<std::path::Path>,
246    m: &Mod,
247    usage: Usage,
248) -> Result<Either<StaticSoundData, StreamingSoundData<FromFileError>>, Error> {
249    use snafu::ResultExt;
250    let filepath = m.as_ref().join(path.as_ref());
251    ensure_located_in_depot(path, m)?;
252    let data = match usage {
253        Usage::OnDemand | Usage::InMemory => StaticSoundData::from_file(filepath)
254            .context(InvalidAudioSnafu {
255                path: path.as_ref().display().to_string(),
256            })
257            .map(Either::Left)?,
258        Usage::Streaming => StreamingSoundData::from_file(filepath)
259            .context(InvalidAudioSnafu {
260                path: path.as_ref().display().to_string(),
261            })
262            .map(Either::Right)?,
263    };
264    Ok(data)
265}

267/// Ensure given settings are valid for audio.
268pub fn ensure_valid_contextual_audio_settings(
269    audio: &Either<StaticSoundData, StreamingSoundData<FromFileError>>,
270    settings: Option<&Settings>,
271    path: &Path,
272) -> Result<(), Error> {
273    let mut errors = settings.validate().err().unwrap_or_default();
274    errors.extend(settings.validate_for(audio).err().unwrap_or_default());
275    ensure!(
276        errors.is_empty(),
277        InvalidAudioSettingsSnafu {
278            which: path.display().to_string(),
279            why: errors
280        }
281    );
282    Ok(())
283}

285#[doc(hidden)]
286pub fn ensure_valid_jingle_captions(
287    audio: &Either<StaticSoundData, StreamingSoundData<FromFileError>>,
288    captions: &[Caption],
289) -> Result<(), Error> {
290    if !captions.is_empty() {
291        let duration = match audio {
292            Either::Left(x) => x.duration(),
293            Either::Right(x) => x.duration(),
294        }
295        .as_secs_f32();
296        for (idx, caption) in captions.iter().enumerate() {
297            if caption.starts >= duration {
298                return Err(Error::from(validation::Error::InvalidAudioCaption {
299                    which: "starts".to_string(),
300                    why: format!("greater than audio duration (captions[{idx}])"),
301                }));
302            }
303        }
304        let mut previous_starts = captions[0].starts;
305        for (idx, caption) in captions.iter().enumerate().skip(1) {
306            if previous_starts >= caption.starts {
307                return Err(Error::from(validation::Error::InvalidAudioCaption {
308                    which: "starts".to_string(),
309                    why: format!(
310                        "unordered sequence (captions[{idx}] and captions[{}])",
311                        idx - 1
312                    ),
313                }));
314            }
315            previous_starts = caption.starts;
316        }
317    }
318    Ok(())
319}

335        CannotStoreDataSnafu {
336            key,
337            path: path.as_ref().display().to_string()
338        }
339    );
340    Ok(())

405/// Ensure guarantees are upheld.
406#[allow(clippy::too_many_arguments)]
407fn ensure<'a, K: PartialEq + Eq + Hash + Clone + Into<Key> + Conflictual>(
408    k: &'a str,
409    key: K,
410    path: PathBuf,
411    m: &Mod,
412    usage: Usage,
413    settings: Option<Settings>,
414    set: &'a mut HashSet<Id>,
415    map: &'a mut HashMap<K, StaticSoundData>,
416    smap: &'a mut HashMap<K, Settings>,
417    source: Source,
418) -> Result<(), Error>
419where
420    HashSet<Id>: Conflict<K>,
421{
422    let data = ensure_valid_audio_and_settings(&path, m, usage, settings.as_ref())?
423        .map_either_with(
424            (usage, settings.as_ref().and_then(|x| x.region.clone())),
425            |ctx, data| {
426                if let (Usage::InMemory, Some(region)) = ctx {
427                    data.slice(region)
428                } else {
429                    data
430                }
431            },
432            |_, data| data,
433        );
434    ensure_key_no_conflict(&key, k, set)?;
435    let id: Id = match usage {
436        Usage::InMemory => Id::InMemory(key.clone().into(), source),
437        Usage::OnDemand => Id::OnDemand(
438            crate::Usage::Static(key.clone().into(), m.as_ref().join(path.clone())),
439            source,
440        ),
441        Usage::Streaming => Id::OnDemand(
442            crate::Usage::Streaming(key.clone().into(), m.as_ref().join(path.clone())),
443            source,
444        ),
445    };
446    if usage == Usage::InMemory {
447        ensure_store_data(key, data.left().unwrap(), settings, &path, map)?;
448    } else if let Some(settings) = settings {
449        ensure_store_settings(&key, settings, smap)?;
450    }
451    ensure_store_id(id, set)?;
452    Ok(())
453}

455/// Ensure guarantees are upheld.
456#[allow(clippy::too_many_arguments)]
457fn ensure_scene<'a, K: PartialEq + Eq + Hash + Clone + Into<SceneKey> + Conflictual>(
458    key: K,
459    path: PathBuf,
460    m: &Mod,
461    usage: Usage,
462    settings: Option<Settings>,
463    set: &'a mut HashSet<SceneId>,
464    map: &'a mut HashMap<K, StaticSoundData>,
465    smap: &'a mut HashMap<K, Settings>,
466) -> Result<(), Error>
467where
468    HashSet<SceneId>: Conflict<K>,
469    SceneKey: From<K>,
470{
471    let data = ensure_valid_audio_and_settings(&path, m, usage, settings.as_ref())?
472        .map_either_with(
473            (usage, settings.as_ref().and_then(|x| x.region.clone())),
474            |ctx, data| {
475                if let (Usage::InMemory, Some(region)) = ctx {
476                    data.slice(region)
477                } else {
478                    data
479                }
480            },
481            |_, data| data,
482        );
483    let id: SceneId = match usage {
484        Usage::InMemory => SceneId::InMemory(key.clone().into()),
485        Usage::OnDemand => SceneId::OnDemand(crate::Usage::Static(
486            key.clone().into(),

488        )),
489        Usage::Streaming => SceneId::OnDemand(crate::Usage::Streaming(
490            key.clone().into(),

494    if usage == Usage::InMemory {

496    } else if let Some(settings) = settings {
497        ensure_store_settings(&key, settings, smap)?;
498    }
499    ensure_store_scene_id(id, set)?;
500    Ok(())

520    ensure(
521        k,

524        m,
525        usage.unwrap_or(Usage::InMemory),
526        settings,
527        set,
528        map,
529        smap,
530        Source::Sfx,
531    )?;
532
533    if !existed {
534        CNamePool::add_cstr(&c_string);
535    }
536    Ok(())

575/// Ensure [Voice] guarantees are upheld.
576#[allow(clippy::too_many_arguments)]
577pub fn ensure_voice<'a>(
578    k: &'a str,
579    v: Voice,
580    m: &Mod,
581    set: &'a mut HashSet<Id>,
582    simple: &'a mut HashMap<LocaleKey, StaticSoundData>,
583    complex: &'a mut HashMap<BothKey, StaticSoundData>,
584    simple_subs: &'a mut HashMap<LocaleKey, DialogLine>,
585    complex_subs: &'a mut HashMap<BothKey, DialogLine>,
586    simple_settings: &'a mut HashMap<LocaleKey, Settings>,
587    complex_settings: &'a mut HashMap<BothKey, Settings>,
588) -> Result<(), Error> {
589    let mut existed = false;
590    let v: AnyVoice = v.into();
591    let c_string = std::ffi::CString::new(k)?;
592    let cname = CName::new(k);
593    let mut simple_key: LocaleKey;
594    let mut complex_key: BothKey;
595    match v {
596        Either::Left((aud, usage, subs)) => {
597            for (locale, Audio { file, settings }) in aud {
598                existed = existed || ensure_localized_key_unique_or_inserted(set, k, locale)?;
599                simple_key = LocaleKey(cname, locale);
600                if let Some(subs) = subs.as_ref().and_then(|x| x.get(&locale)) {
601                    ensure_store_subtitle::<LocaleKey>(
602                        simple_key.clone(),
603                        subs.clone(),
604                        simple_subs,
605                    )?;
606                }
607                ensure(
608                    k,
609                    simple_key,
610                    file,
611                    m,
612                    usage,
613                    settings.clone(),
614                    set,
615                    simple,
616                    simple_settings,
617                    Source::Voices,
618                )?;
619            }
620        }
621        Either::Right((aud, usage, subs)) => {
622            for (locale, genders) in aud {
623                existed = existed || ensure_localized_key_unique_or_inserted(set, k, locale)?;
624                for (gender, Audio { file, settings }) in genders {
625                    complex_key = BothKey(cname, locale, gender);
626                    if let Some(subs) = subs.as_ref().and_then(|x| x.get(&locale)) {
627                        ensure_store_subtitle::<BothKey>(
628                            complex_key.clone(),
629                            if gender == PlayerGender::Female {
630                                subs.female.clone()
631                            } else {
632                                subs.male.clone()
633                            },
634                            complex_subs,
635                        )?;
636                    }
637                    ensure(
638                        k,
639                        complex_key,
640                        file,
641                        m,
642                        usage,
643                        settings.clone(),
644                        set,
645                        complex,
646                        complex_settings,
647                        Source::Voices,
648                    )?;
649                }
650            }
651        }
652    }
653
654    if !existed {
655        CNamePool::add_cstr(&c_string);
656    }
657    Ok(())
658}

696#[doc(hidden)]
697pub fn ensure_jingles<'a>(
698    k: &'a str,
699    v: Jingle,
700    m: &Mod,
701    set: &'a mut HashSet<Id>,
702    smap: &'a mut HashMap<UniqueKey, Settings>,
703) -> Result<(), Error> {
704    let existed = ensure_key_unique_or_inserted(k)?;
705    let Audio { file, settings } = (&v).into();
706    ensure_valid_audio_with_settings_and_captions(
707        &file,
708        m,
709        Usage::Streaming,
710        settings.as_ref(),
711        v.captions(),
712    )?;
713    let c_string = std::ffi::CString::new(k)?;
714    let cname = CName::new(k);
715    let key = UniqueKey(cname);
716    ensure_key_no_conflict(&key, k, set)?;
717    let id: Id = Id::OnDemand(
718        crate::Usage::Streaming(crate::Key::Unique(key.clone()), m.as_ref().join(file)),
719        Source::Jingle,
720    );
721    if let Some(settings) = settings {
722        ensure_store_settings::<UniqueKey>(&key, settings, smap)?;
723    }
724    ensure_store_id(id, set)?;
725
726    if !existed {
727        CNamePool::add_cstr(&c_string);
728    }
729    Ok(())
730}

732#[doc(hidden)]
733#[allow(clippy::too_many_arguments)]
734pub fn ensure_scene_dialogs<'a>(
735    k: i64,
736    v: SceneDialogs,
737    m: &Mod,
738    set: &'a mut HashSet<SceneId>,
739    single: &'a mut HashMap<SceneLocaleKey, StaticSoundData>,
740    dual: &'a mut HashMap<SceneBothKey, StaticSoundData>,
741    single_settings: &'a mut HashMap<SceneLocaleKey, Settings>,
742    dual_settings: &'a mut HashMap<SceneBothKey, Settings>,
743) -> Result<(), Error> {
744    let mut errors = Vec::with_capacity(10);
745    let mut locale_key: SceneLocaleKey;
746    let mut both_key: SceneBothKey;
747    let v: AnySceneDialog = v.into();
748    let cruid = Cruid::from(k);
749    match v {
750        Either::Left((aud, usage)) => {
751            for (locale, Audio { file, settings }) in aud {
752                locale_key = SceneLocaleKey(cruid, locale);
753                ensure_scene_key_no_conflict(&locale_key, &(k, locale), set)?;
754                if let Err(e) = ensure_scene(
755                    locale_key,
756                    file,
757                    m,
758                    usage,
759                    settings,
760                    set,
761                    single,
762                    single_settings,
763                ) {
764                    errors.push(e);
765                    continue;
766                }
767            }
768        }
769        Either::Right((aud, usage)) => {
770            let mut checked_conflict;
771            'outer: for (locale, genders) in aud {
772                checked_conflict = false;
773                for (gender, Audio { file, settings }) in genders {
774                    both_key = SceneBothKey(Cruid::from(k), locale, gender);
775                    // only check once per pair, otherwise gender will conflict with each other
776                    if !checked_conflict {
777                        if let Err(e) = ensure_scene_key_no_conflict(&both_key, &(k, locale), set) {
778                            errors.push(e);
779                            continue 'outer;
780                        }
781                        checked_conflict = true;
782                    }
783                    if let Err(e) =
784                        ensure_scene(both_key, file, m, usage, settings, set, dual, dual_settings)
785                    {
786                        errors.push(e);
787                        // other gender might already have been added
788                        dual.remove(&SceneBothKey(Cruid::from(k), locale, gender.opposite()));
789                        dual_settings.remove(&SceneBothKey(
790                            Cruid::from(k),
791                            locale,
792                            gender.opposite(),
793                        ));
794                        continue 'outer;
795                    }
796                }
797            }
798        }
799    };
800    Ok(())
801}
