3use red4ext_rs::types::{CName, Cruid};
4use snafu::Snafu;
5
6#[derive(Debug, Snafu)]
7pub enum Error {
8    #[snafu(display("Registry error: {source}"), visibility(pub(crate)))]
9    Registry {
10        source: self::registry::Error<CName>,
11    },
12    #[snafu(display("Registry error: {source}"), visibility(pub(crate)))]
13    SceneRegistry {
14        source: self::registry::Error<Cruid>,
15    },
16    #[snafu(display("Validation error: {source}"), visibility(pub(crate)))]
17    Validation { source: self::validation::Error },
18    #[snafu(display("Manifest error: {source}"), visibility(pub(crate)))]
19    Manifest {
20        source: audioware_manifest::error::Error,
21    },
22    #[snafu(display("Error(s): {}", errors.iter().map(|e| format!("{e}")).collect::<Vec<_>>().join("\n  -> ")), visibility(pub(crate)))]
23    Multiple { errors: Vec<Error> },
24}

26pub mod registry {
27    use audioware_manifest::{SpokenLocale, WrittenLocale};
28    use red4ext_rs::types::{CName, Cruid};
29    use snafu::Snafu;
30
31    pub trait ErrorDisplay {
32        fn error_display(&self) -> impl std::fmt::Display;
33    }
34
35    impl ErrorDisplay for CName {
36        fn error_display(&self) -> impl std::fmt::Display {
37            self.as_str()
38        }
39    }
40
41    impl ErrorDisplay for Cruid {
42        fn error_display(&self) -> impl std::fmt::Display {
43            // WolvenKit and Codeware represents CRUID as u64
44            // but RED4ext.SDK as i64
45            i64::from(*self) as u64
46        }
47    }
48
49    #[derive(Debug, Snafu)]
50    pub enum Error<K: ErrorDisplay> {
51        #[snafu(display("missing spoken locale: {} for {}", locale, key.error_display()), visibility(pub(crate)))]
52        MissingSpokenLocale { key: K, locale: SpokenLocale },
53        #[snafu(display("missing written locale: {} for {}", locale, key.error_display()), visibility(pub(crate)))]
54        MissingWrittenLocale { key: K, locale: WrittenLocale },
55        #[snafu(display("requires gender: {}", key.error_display()), visibility(pub(crate)))]
56        RequireGender { key: K },
57        #[snafu(display("not found: {}", key.error_display()), visibility(pub(crate)))]
58        NotFound { key: K },
59    }
60}

62pub mod validation {
63
64    use audioware_manifest::{DialogLine, Locale};
65    use snafu::Snafu;
66
67    use crate::{Id, Key, SceneId, SceneKey};
68
69    #[derive(Debug, Snafu)]
70    pub enum Error {
71        #[snafu(
72            display(
73                "duplicate folder across 'r6\\audioware' and 'mods' folders, skipping folder in 'r6\\audioware' ({folder})"
74            ),
75            visibility(pub(crate))
76        )]
77        DuplicateAcrossDepots { folder: String },
78        #[snafu(display("CName already exists: {cname}"), visibility(pub(crate)))]
79        NonUniqueKey { cname: String },
80        #[snafu(
81            display("CName conflicts with existing id: {cname}"),
82            visibility(pub(crate))
83        )]
84        ConflictingKey { cname: String },
85        #[snafu(
86            display("RUID conflicts with existing id: {} for {locale}", *cruid as u64),
87            visibility(pub(crate))
88        )]
89        ConflictingSceneKey { cruid: i64, locale: Locale },
90        #[snafu(display("audio outside depot: {path}"), visibility(pub(crate)))]
91        AudioOutsideDepot { path: String },
92        #[snafu(
93            display("cannot load audio: {path} ({source})"),
94            visibility(pub(crate))
95        )]
96        InvalidAudio {
97            path: String,
98            source: kira::sound::FromFileError,
99        },
100        #[snafu(
101            display("invalid cname: {source}"),
102            visibility(pub),
103            context(suffix(false))
104        )]
105        /// An error occured while converting audio ID to CName.
106        InvalidAudioID { source: std::ffi::NulError },
107        #[snafu(display("invalid audio setting(s) {which}: {}", why.iter().map(|x| format!("{}: {}", x.which, x.why)).collect::<Vec<_>>().join("\n")), visibility(pub(crate)))]
108        InvalidAudioSettings {
109            which: String,
110            why: Vec<audioware_manifest::error::ValidationError>,
111        },
112        #[snafu(
113            display("invalid audio caption: {which} ({why})"),
114            visibility(pub(crate))
115        )]
116        InvalidAudioCaption { which: String, why: String },
117        #[snafu(display("cannot store data: {key} ({path})"), visibility(pub(crate)))]
118        CannotStoreData { key: Key, path: String },
119        #[snafu(
120            display("cannot store scene data: {key} ({path})"),
121            visibility(pub(crate))
122        )]
123        CannotStoreSceneData { key: SceneKey, path: String },
124        #[snafu(
125            display("cannot store subtitle: {key} ({value})"),
126            visibility(pub(crate))
127        )]
128        CannotStoreSubtitle { key: Key, value: DialogLine },
129        #[snafu(display("cannot store audio settings"), visibility(pub(crate)))]
130        CannotStoreSettings,
131        #[snafu(display("cannot store id: {id}"), visibility(pub(crate)))]
132        CannotStoreAgnosticId { id: Id },
133        #[snafu(display("cannot store scene id: {id}"), visibility(pub(crate)))]
134        CannotStoreSceneId { id: SceneId },
135        #[snafu(display("IO: {source}"), visibility(pub(crate)))]
136        IO { source: std::io::Error },
137    }
138}

140impl From<self::registry::Error<CName>> for self::Error {
141    fn from(source: self::registry::Error<CName>) -> Self {
142        Self::Registry { source }
143    }
144}
