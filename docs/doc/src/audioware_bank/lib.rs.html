90    /// Return audio duration (as seconds) if any, otherwise `-1.0`.
91    pub fn duration(
92        &self,
93        cname: &CName,
94        locale: Locale,
95        gender: PlayerGender,
96        total: bool,
97    ) -> f32 {
98        let locale = SpokenLocale::from(locale);
99        if let Ok(id) = self.ids.try_get(cname, &locale, Some(&gender)) {
100            match (total, id, self.data(id)) {
101                // if no need for total and in-memory, sound data already embed settings
102                (false, Id::InMemory(..), data) => data
103                    .left()
104                    .expect("streaming cannot be stored in-memory")
105                    .slice_duration(),
106                // if no need for total and on-demand, check settings
107                (false, Id::OnDemand(..), data) => match (data, self.settings(id)) {
108                    (Either::Left(x), settings) => x.with(settings).slice_duration(),
109                    (Either::Right(x), settings) => x.with(settings).slice_duration(),
110                },
111                // if need total
112                (true, _, data) => match data {
113                    Either::Left(x) => x.total_duration(),
114                    Either::Right(x) => x.total_duration(),
115                },
116            }
117            .as_secs_f32()
118        } else {
119            -1.0
120        }
121    }

133    fn mods() -> (Vec<Mod>, Vec<Error>) {
134        let mut errors = Vec::with_capacity(10);
135        let mut mods = Vec::with_capacity(30);
136        let mut redmod_exists = false;
137        if let Ok(redmod) = REDmod::try_new() {
138            mods = redmod.mods();
139            redmod_exists = true;
140        }
141        if let Ok(r6audioware) = R6Audioware::try_new() {
142            for m in r6audioware.mods() {
143                if let Err(e) =
144                    ensure_no_duplicate_accross_depots(redmod_exists, &m, mods.as_slice())
145                {
146                    errors.push(e);
147                    continue;
148                }
149                mods.push(m);
150            }
151        }
152        (mods, errors)

154    /// Initialize banks.
155    pub fn new() -> (Self, Initialization) {
156        let since = Instant::now();
157
158        let mut file: Vec<u8>;
159        let mut manifest: Manifest;
160        let (mods, mut errors) = Self::mods();
161        let mut scene_errors = Vec::with_capacity(30);
162
163        let mut ids: HashSet<Id> = HashSet::new();
164        let mut scene_ids: HashSet<SceneId> = HashSet::new();
165        let mut uniques: HashMap<UniqueKey, StaticSoundData> = HashMap::new();
166        let mut genders: HashMap<GenderKey, StaticSoundData> = HashMap::new();
167        let mut single_voices: HashMap<LocaleKey, StaticSoundData> = HashMap::new();
168        let mut dual_voices: HashMap<BothKey, StaticSoundData> = HashMap::new();
169        let mut single_subs: HashMap<LocaleKey, DialogLine> = HashMap::new();
170        let mut dual_subs: HashMap<BothKey, DialogLine> = HashMap::new();
171        let mut unique_settings: HashMap<UniqueKey, Settings> = HashMap::new();
172        let mut gender_settings: HashMap<GenderKey, Settings> = HashMap::new();
173        let mut single_settings: HashMap<LocaleKey, Settings> = HashMap::new();
174        let mut dual_settings: HashMap<BothKey, Settings> = HashMap::new();
175        let mut single_scene_dialogs: HashMap<SceneLocaleKey, StaticSoundData> = HashMap::new();
176        let mut dual_scene_dialogs: HashMap<SceneBothKey, StaticSoundData> = HashMap::new();
177        let mut single_scene_dialogs_settings: HashMap<SceneLocaleKey, Settings> = HashMap::new();
178        let mut dual_scene_dialogs_settings: HashMap<SceneBothKey, Settings> = HashMap::new();
179
180        for m in mods {
181            let paths = m.manifests_paths();
182            for ref path in paths {
183                file = match std::fs::read(path).context(CannotReadManifest {
184                    manifest: path.display().to_string(),
185                }) {
186                    Ok(x) => x,
187                    Err(e) => {
188                        errors.push(e.into());
189                        continue;
190                    }
191                };
192                match serde_yaml::from_slice::<Manifest>(file.as_slice()).context(
193                    CannotParseManifest {
194                        manifest: path.display().to_string(),
195                    },
196                ) {
197                    Ok(x) => {
198                        manifest = x;
199                    }
200                    Err(e) => {
201                        errors.push(e.into());
202                        continue;
203                    }
204                };
205                if let Err(e) = ensure_manifest_no_duplicates(&manifest) {
206                    errors.push(e);
207                    continue;
208                }
209                if let Some(sfx) = manifest.sfx {
210                    for (key, value) in sfx {
211                        match ensure_sfx(
212                            key.as_str(),
213                            value,
214                            &m,
215                            &mut ids,
216                            &mut uniques,
217                            &mut unique_settings,
218                        ) {
219                            Ok(x) => x,
220                            Err(e) => {
221                                errors.push(e);
222                                continue;
223                            }
224                        }
225                    }
226                }
227                if let Some(onos) = manifest.onos {
228                    for (key, value) in onos {
229                        match ensure_ono(
230                            key.as_str(),
231                            value,
232                            &m,
233                            &mut ids,
234                            &mut genders,
235                            &mut gender_settings,
236                        ) {
237                            Ok(x) => x,
238                            Err(e) => {
239                                errors.push(e);
240                                continue;
241                            }
242                        };
243                    }
244                }
245                if let Some(voices) = manifest.voices {
246                    for (key, value) in voices {
247                        match ensure_voice(
248                            key.as_str(),
249                            value,
250                            &m,
251                            &mut ids,
252                            &mut single_voices,
253                            &mut dual_voices,
254                            &mut single_subs,
255                            &mut dual_subs,
256                            &mut single_settings,
257                            &mut dual_settings,
258                        ) {
259                            Ok(x) => x,
260                            Err(e) => {
261                                errors.push(e);
262                                continue;
263                            }
264                        };
265                    }
266                }
267                if let Some(music) = manifest.music {
268                    for (key, value) in music {
269                        match ensure_music(
270                            key.as_str(),
271                            value,
272                            &m,
273                            &mut ids,
274                            &mut uniques,
275                            &mut unique_settings,
276                        ) {
277                            Ok(x) => x,
278                            Err(e) => {
279                                errors.push(e);
280                                continue;
281                            }
282                        };
283                    }
284                }
285                if let Some(jingles) = manifest.jingles {
286                    for (key, value) in jingles {
287                        match ensure_jingles(
288                            key.as_str(),
289                            value,
290                            &m,
291                            &mut ids,
292                            &mut unique_settings,
293                        ) {
294                            Ok(x) => x,
295                            Err(e) => {
296                                errors.push(e);
297                                continue;
298                            }
299                        };
300                    }
301                }
302                if let Some(scene_dialogs) = manifest.scene_dialogs {
303                    for (key, value) in scene_dialogs {
304                        match ensure_scene_dialogs(
305                            key as i64,
306                            value,
307                            &m,
308                            &mut scene_ids,
309                            &mut single_scene_dialogs,
310                            &mut dual_scene_dialogs,
311                            &mut single_scene_dialogs_settings,
312                            &mut dual_scene_dialogs_settings,
313                        ) {
314                            Ok(x) => x,
315                            Err(e) => {
316                                scene_errors.push(e);
317                                continue;
318                            }
319                        };
320                    }
321                }
322            }
323        }
324
325        let lengths = ids.iter().fold((0, 0, 0), |acc, x| {
326            let (mut odsta, mut odstr, mut imsta) = acc;
327            match x {
328                Id::OnDemand(Usage::Static(..), ..) => odsta += 1,
329                Id::OnDemand(Usage::Streaming(..), ..) => odstr += 1,
330                Id::InMemory(..) => imsta += 1,
331            }
332            (odsta, odstr, imsta)
333        });
334
335        let scene_lengths = scene_ids.iter().fold((0, 0, 0), |acc, x| {
336            let (mut odsta, mut odstr, mut imsta) = acc;
337            match x {
338                SceneId::OnDemand(Usage::Static(..), ..) => odsta += 1,
339                SceneId::OnDemand(Usage::Streaming(..), ..) => odstr += 1,
340                SceneId::InMemory(..) => imsta += 1,
341            }
342            (odsta, odstr, imsta)
343        });
344
345        #[cfg(feature = "hot-reload")]
346        let errors = errors
347            .into_iter()
348            .map(std::sync::Arc::new)
349            .collect::<Vec<_>>();
350
351        #[cfg(feature = "hot-reload")]
352        let scene_errors = scene_errors
353            .into_iter()
354            .map(std::sync::Arc::new)
355            .collect::<Vec<_>>();
356
357        let report = Initialization {
358            duration: Instant::now() - since,
359            lengths: format!(
360                r##"ids:
361- on-demand static audio    -> {}
362- on-demand streaming audio -> {}
363- in-memory static audio    -> {}"##,
364                lengths.0, lengths.1, lengths.2
365            ),
366            scene_lengths: format!(
367                r##"scene ids:
368- on-demand static audio    -> {}
369- on-demand streaming audio -> {}
370- in-memory static audio    -> {}"##,
371                scene_lengths.0, scene_lengths.1, scene_lengths.2
372            ),
373            len_ids: ids.len(),
374            len_scene_ids: scene_ids.len(),
375            errors,
376            scene_errors,
377        };
378
379        (
380            Self {
381                ids,
382                scene_ids,
383                uniques,
384                genders,
385                single_voices,
386                dual_voices,
387                single_subs,
388                dual_subs,
389                unique_settings,
390                gender_settings,
391                single_settings,
392                dual_settings,
393                single_scene_dialogs,
394                dual_scene_dialogs,
395                single_scene_dialogs_settings,
396                dual_scene_dialogs_settings,
397            },
398            report,
399        )
400    }
401    #[cfg(feature = "hot-reload")]
402    pub fn hot_reload(&mut self) -> Initialization {
403        PREVIOUS_IDS
404            .lock()
405            .expect("already loaded before")
406            .clone_from(&self.ids.drain().collect());
407        PREVIOUS_SCENE_IDS
408            .lock()
409            .expect("already loaded before")
410            .clone_from(&self.scene_ids.drain().collect());
411        let (banks, initialization) = Self::new();
412        self.ids = banks.ids;
413        self.scene_ids = banks.scene_ids;
414        self.uniques = banks.uniques;
415        self.genders = banks.genders;
416        self.single_voices = banks.single_voices;
417        self.dual_voices = banks.dual_voices;
418        self.single_subs = banks.single_subs;
419        self.dual_subs = banks.dual_subs;
420        self.unique_settings = banks.unique_settings;
421        self.gender_settings = banks.gender_settings;
422        self.single_settings = banks.single_settings;
423        self.dual_settings = banks.dual_settings;
424        self.single_scene_dialogs = banks.single_scene_dialogs;
425        self.dual_scene_dialogs = banks.dual_scene_dialogs;
426        self.single_scene_dialogs_settings = banks.single_scene_dialogs_settings;
427        self.dual_scene_dialogs_settings = banks.dual_scene_dialogs_settings;
428        initialization
429    }

432/// Outcome of [Banks] initialization.
433#[cfg_attr(feature = "hot-reload", derive(Clone))]
434pub struct Initialization {
435    duration: Duration,
436    lengths: String,
437    scene_lengths: String,
438    len_ids: usize,
439    len_scene_ids: usize,
440    #[cfg(not(feature = "hot-reload"))]
441    pub errors: Vec<Error>,
442    #[cfg(feature = "hot-reload")]
443    pub errors: Vec<std::sync::Arc<Error>>,
444    #[cfg(not(feature = "hot-reload"))]
445    pub scene_errors: Vec<Error>,
446    #[cfg(feature = "hot-reload")]
447    pub scene_errors: Vec<std::sync::Arc<Error>>,
448}
449
450pub enum InitializationOutcome {
451    CompleteFailure = 0,
452    PartialSuccess = 1,
453    Success = 2,
454}
455
456impl Initialization {
457    pub fn outcome_sections(&self) -> InitializationOutcome {
458        if self.errors.is_empty() {
459            InitializationOutcome::Success
460        } else if self.len_ids == 0 {
461            InitializationOutcome::CompleteFailure
462        } else {
463            InitializationOutcome::PartialSuccess
464        }
465    }
466    pub fn outcome_scene(&self) -> InitializationOutcome {
467        if self.scene_errors.is_empty() {
468            InitializationOutcome::Success
469        } else if self.len_scene_ids == 0 {
470            InitializationOutcome::CompleteFailure
471        } else {
472            InitializationOutcome::PartialSuccess
473        }
474    }
475    pub fn outcome(&self) -> InitializationOutcome {
476        match (self.outcome_sections(), self.outcome_scene()) {
477            (InitializationOutcome::Success, InitializationOutcome::Success) => {
478                InitializationOutcome::Success
479            }
480            (InitializationOutcome::CompleteFailure, InitializationOutcome::CompleteFailure) => {
481                InitializationOutcome::CompleteFailure
482            }
483            _ => InitializationOutcome::PartialSuccess,
484        }
485    }
486}

488impl std::fmt::Display for Initialization {
489    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
490        let Initialization {
491            duration,
492            lengths,
493            len_ids,
494            errors,
495            scene_errors,
496            scene_lengths,
497            len_scene_ids,
498        } = self;
499        write!(
500            f,
501            r##"initialization took {duration:?}
502{lengths}
503for a total of: {len_ids} id(s)
504{}
505-------------------------------
506{scene_lengths}
507for a total of: {len_scene_ids} scene id(s)
508{}
509"##,
510            if errors.is_empty() {
511                "no error reported!".to_string()
512            } else {
513                format!(
514                    "error(s):\n{}",
515                    errors
516                        .iter()
517                        .map(|e| format!("- {e}"))
518                        .collect::<Vec<_>>()
519                        .join("\n")
520                )
521            },
522            if scene_errors.is_empty() {
523                "no scene error reported!".to_string()
524            } else {
525                format!(
526                    "scene error(s):\n{}",
527                    scene_errors
528                        .iter()
529                        .map(|e| format!("- {e}"))
530                        .collect::<Vec<_>>()
531                        .join("\n")
532                )
533            }
534        )
535    }

538pub trait TryGet {
539    type Id;
540    type Raw;
541    fn try_get<K: ErrorDisplay + Clone>(
542        &self,
543        name: &K,
544        spoken: &SpokenLocale,
545        gender: Option<&PlayerGender>,
546    ) -> Result<&Self::Id, Error>
547    where
548        Self::Raw: PartialEq<K>,
549        error::Error: From<crate::error::registry::Error<K>>;
550}
551
552impl TryGet for HashSet<Id> {
553    type Id = Id;
554    type Raw = CName;
555
556    fn try_get<K: ErrorDisplay + Clone>(
557        &self,
558        name: &K,
559        spoken: &SpokenLocale,
560        gender: Option<&PlayerGender>,
561    ) -> Result<&Self::Id, Error>
562    where
563        Self::Raw: PartialEq<K>,
564        error::Error: From<crate::error::registry::Error<K>>,
565    {
566        let mut maybe_missing_locale = false;
567        let mut key: &Key;
568        for id in self.iter() {
569            key = id.as_ref();
570            if let Some(key) = key.as_unique()
571                && key.as_ref() == name
572            {
573                return Ok(id);
574            }
575            if let Some(GenderKey(k, g)) = key.as_gender()
576                && k == name
577            {
578                if gender.is_none() {
579                    return Err(RegistryError::RequireGender { key: name.clone() }.into());
580                }
581                if Some(g) == gender {
582                    return Ok(id);
583                }
584            }
585            if let Some(LocaleKey(k, l)) = key.as_locale()
586                && k == name
587            {
588                maybe_missing_locale = true;
589                if l == spoken {
590                    return Ok(id);
591                }
592            }
593            if let Some(BothKey(k, l, g)) = key.as_both()
594                && k == name
595            {
596                maybe_missing_locale = true;
597                if l == spoken {
598                    if gender.is_none() {
599                        return Err(RegistryError::RequireGender { key: name.clone() }.into());
600                    }
601                    if gender == Some(g) {
602                        return Ok(id);
603                    }
604                }
605            }
606        }
607        if maybe_missing_locale {
608            return Err(RegistryError::MissingSpokenLocale {
609                key: name.clone(),
610                locale: *spoken,
611            }
612            .into());
613        }
614        Err(RegistryError::NotFound { key: name.clone() }.into())
615    }
616}

618impl TryGet for HashSet<SceneId> {
619    type Id = SceneId;
620    type Raw = Cruid;
621
622    fn try_get<K: ErrorDisplay + Clone>(
623        &self,
624        name: &K,
625        spoken: &SpokenLocale,
626        gender: Option<&PlayerGender>,
627    ) -> Result<&Self::Id, Error>
628    where
629        Self::Raw: PartialEq<K>,
630        error::Error: From<crate::error::registry::Error<K>>,
631    {
632        let mut maybe_missing_locale = false;
633        let mut key: &SceneKey;
634        for id in self.iter() {
635            key = id.as_ref();
636            match key {
637                SceneKey::Locale(SceneLocaleKey(key, locale)) if key == name => {
638                    maybe_missing_locale = true;
639                    if locale == spoken {
640                        return Ok(id);
641                    }
642                }
643                SceneKey::Both(SceneBothKey(key, locale, g)) if key == name => {
644                    maybe_missing_locale = true;
645                    if locale == spoken {
646                        if gender.is_none() {
647                            return Err(RegistryError::RequireGender { key: name.clone() }.into());
648                        }
649                        if gender == Some(g) {
650                            return Ok(id);
651                        }
652                    }
653                }
654                _ => continue,
655            }
656        }
657        if maybe_missing_locale {
658            return Err(RegistryError::MissingSpokenLocale {
659                key: name.clone(),
660                locale: *spoken,
661            }
662            .into());
663        }
664        Err(RegistryError::NotFound { key: name.clone() }.into())
665    }
