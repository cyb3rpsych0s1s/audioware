24/// A piece of audio loaded into memory all at once.
25///
26/// These can be cheaply cloned, as the audio data is shared
27/// among all clones.
28#[derive(Clone, PartialEq)]
29pub struct StaticSoundData {
30	/// The sample rate of the audio (in Hz).
31	pub sample_rate: u32,
32	/// The raw samples that make up the audio.
33	pub frames: Arc<[Frame]>,
34	/// Settings for the sound.
35	pub settings: StaticSoundSettings,
36	/**
37	The portion of the sound this [`StaticSoundData`] represents.
38
39	Note that the [`StaticSoundData`] holds the entire piece of audio
40	it was originally given regardless of the value of `slice`, but
41	[`StaticSoundData::num_frames`], [`StaticSoundData::duration`],
42	and [`StaticSoundData::frame_at_index`] will all behave as if
43	this [`StaticSoundData`] only contained the specified portion of
44	audio.
45	*/
46	pub slice: Option<(usize, usize)>,
47}

331	/**
332	Sets the portion of the audio this [`StaticSoundData`] represents.
333
334	This returns a cheap clone of the [`StaticSoundData`] with the modified slice.
335
336	Note that the [`StaticSoundData`] holds the entire piece of audio it was originally
337	given regardless of the value of `slice`, but [`StaticSoundData::num_frames`],
338	[`StaticSoundData::duration`], and [`StaticSoundData::frame_at_index`] will all behave
339	as if this [`StaticSoundData`] only contained the specified portion of audio.
340
341	# Example
342
343	```
344	use kira::{
345		sound::static_sound::{StaticSoundData, StaticSoundSettings},
346		Frame,
347	};
348	let sound = StaticSoundData {
349		sample_rate: 1,
350		frames: (0..10).map(|i| Frame::from_mono(i as f32)).collect(),
351		settings: StaticSoundSettings::default(),
352		slice: None,
353	};
354	let sliced = sound.slice(3.0..6.0);
355	assert_eq!(sliced.num_frames(), 3);
356	assert_eq!(sliced.frame_at_index(0), Some(Frame::from_mono(3.0)));
357	assert_eq!(sliced.frame_at_index(1), Some(Frame::from_mono(4.0)));
358	```
359	*/
360	#[must_use = "This method returns a modified StaticSoundData and does not mutate the original value"]
361	pub fn slice(&self, region: impl IntoOptionalRegion) -> Self {
362		let mut new = self.clone();
363		new.slice = region.into_optional_region().map(|Region { start, end }| {
364			let start = start.into_samples(self.sample_rate);
365			let end = match end {
366				EndPosition::EndOfAudio => self.frames.len(),
367				EndPosition::Custom(end) => end.into_samples(self.sample_rate),
368			};
369			(start, end)
370		});
371		new
372	}

374	pub(super) fn split(self) -> (StaticSound, StaticSoundHandle) {
375		let (command_writers, command_readers) = command_writers_and_readers();
376		let sound = StaticSound::new(self, command_readers);
377		let shared = sound.shared();
378		(
379			sound,
380			StaticSoundHandle {
381				command_writers,
382				shared,
383			},
384		)
385	}

425pub(crate) fn num_frames(frames: &[Frame], slice: Option<(usize, usize)>) -> usize {
426	if let Some((start, end)) = slice {
427		end - start
428	} else {
429		frames.len()
430	}
431}
432
433pub(crate) fn frame_at_index(
434	index: usize,
435	frames: &[Frame],
436	slice: Option<(usize, usize)>,
437) -> Option<Frame> {
438	if index >= num_frames(frames, slice) {
439		return None;
440	}
441	let start = slice.map(|(start, _)| start).unwrap_or_default();
442	Some(frames[index + start])
