22/// A streaming sound that is not playing yet.
23pub struct StreamingSoundData<Error: Send + 'static> {
24	pub(crate) decoder: Box<dyn Decoder<Error = Error>>,
25	/// Settings for the streaming sound.
26	pub settings: StreamingSoundSettings,
27	/**
28	The portion of the sound this [`StreamingSoundData`] represents.
29
30	Note that the [`StreamingSoundData`] holds the entire piece of audio
31	it was originally given regardless of the value of `slice`, but
32	[`StreamingSoundData::num_frames`] and [`StreamingSoundData::duration`]
33	will behave as if this [`StreamingSoundData`] only contained the specified
34	portion of audio.
35	*/
36	pub slice: Option<(usize, usize)>,
37}

273	/// Returns the number of frames in the [`StreamingSoundData`].
274	///
275	/// If [`StreamingSoundData::slice`] is `Some`, this will be the number
276	/// of frames in the slice.
277	#[must_use]
278	pub fn num_frames(&self) -> usize {
279		if let Some((start, end)) = self.slice {
280			end - start
281		} else {
282			self.decoder.num_frames()
283		}
284	}
285
286	/// Returns the duration of the audio.
287	///
288	/// If [`StreamingSoundData::slice`] is `Some`, this will be the duration
289	/// of the slice.
290	#[must_use]
291	pub fn duration(&self) -> Duration {
292		Duration::from_secs_f64(self.num_frames() as f64 / self.decoder.sample_rate() as f64)
293	}
294
295	/// Returns the total duration of the audio, regardless of its slice.
296	#[must_use]
297	pub fn unsliced_duration(&self) -> Duration {
298		Duration::from_secs_f64(
299			self.decoder.num_frames() as f64 / self.decoder.sample_rate() as f64,
300		)
301	}

303	/**
304	Sets the portion of the audio this [`StreamingSoundData`] represents.
305	*/
306	#[must_use = "This method consumes self and returns a modified StreamingSoundData, so the return value should be used"]
307	pub fn slice(mut self, region: impl IntoOptionalRegion) -> Self {
308		self.slice = region.into_optional_region().map(|Region { start, end }| {
309			let start = start.into_samples(self.decoder.sample_rate());
310			let end = match end {
311				EndPosition::EndOfAudio => self.decoder.num_frames(),
312				EndPosition::Custom(end) => end.into_samples(self.decoder.sample_rate()),
313			};
314			(start, end)
315		});
316		self
317	}

359impl<Error: Send + 'static> StreamingSoundData<Error> {
360	pub(crate) fn split(
361		self,
362	) -> Result<
363		(
364			StreamingSound,
365			StreamingSoundHandle<Error>,
366			DecodeScheduler<Error>,
367		),
368		Error,
369	> {
370		let (command_writers, command_readers, decode_scheduler_command_readers) =
371			command_writers_and_readers();
372		let (error_producer, error_consumer) = RingBuffer::new(ERROR_BUFFER_CAPACITY);
373		let sample_rate = self.decoder.sample_rate();
374		let shared = Arc::new(Shared::new());
375		let (scheduler, frame_consumer) = DecodeScheduler::new(
376			self.decoder,
377			self.slice,
378			self.settings,
379			shared.clone(),
380			decode_scheduler_command_readers,
381			error_producer,
382		)?;
383		let sound = StreamingSound::new(
384			sample_rate,
385			self.settings,
386			shared.clone(),
387			frame_consumer,
388			command_readers,
389			&scheduler,
390		);
391		let handle = StreamingSoundHandle {
392			shared,
393			command_writers,
394			error_consumer: Mutex::new(error_consumer),
395		};
396		Ok((sound, handle, scheduler))
397	}
398}

400impl<Error: Send + 'static> SoundData for StreamingSoundData<Error> {
401	type Error = Error;
402
403	type Handle = StreamingSoundHandle<Error>;
404
405	#[allow(clippy::type_complexity)]
406	fn into_sound(self) -> Result<(Box<dyn crate::sound::Sound>, Self::Handle), Self::Error> {
407		let (sound, handle, scheduler) = self.split()?;
408		scheduler.start();
409		Ok((Box::new(sound), handle))
410	}
