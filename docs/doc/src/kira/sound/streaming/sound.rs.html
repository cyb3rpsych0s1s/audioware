27pub(crate) struct Shared {
28	state: AtomicU8,
29	position: AtomicU64,
30	reached_end: AtomicBool,
31	encountered_error: AtomicBool,
32}
33
34impl Shared {
35	#[must_use]
36	pub fn new() -> Self {
37		Self {
38			position: AtomicU64::new(0.0f64.to_bits()),
39			state: AtomicU8::new(PlaybackState::Playing as u8),
40			reached_end: AtomicBool::new(false),
41			encountered_error: AtomicBool::new(false),
42		}
43	}
44
45	#[must_use]
46	pub fn state(&self) -> PlaybackState {
47		match self.state.load(Ordering::SeqCst) {
48			0 => PlaybackState::Playing,
49			1 => PlaybackState::Pausing,
50			2 => PlaybackState::Paused,
51			3 => PlaybackState::WaitingToResume,
52			4 => PlaybackState::Resuming,
53			5 => PlaybackState::Stopping,
54			6 => PlaybackState::Stopped,
55			_ => panic!("Invalid playback state"),
56		}
57	}
58
59	pub fn set_state(&self, state: PlaybackState) {
60		self.state.store(state as u8, Ordering::SeqCst);
61	}
62
63	#[must_use]
64	pub fn position(&self) -> f64 {
65		f64::from_bits(self.position.load(Ordering::SeqCst))
66	}
67
68	#[must_use]
69	pub fn reached_end(&self) -> bool {
70		self.reached_end.load(Ordering::SeqCst)
71	}
72
73	#[must_use]
74	pub fn encountered_error(&self) -> bool {
75		self.encountered_error.load(Ordering::SeqCst)
76	}
77}

79pub(crate) struct StreamingSound {
80	command_readers: CommandReaders,
81	sample_rate: u32,
82	frame_consumer: Consumer<TimestampedFrame>,
83	start_time: StartTime,
84	playback_state_manager: PlaybackStateManager,
85	current_frame: usize,
86	fractional_position: f64,
87	volume: Parameter<Decibels>,
88	playback_rate: Parameter<PlaybackRate>,
89	panning: Parameter<Panning>,
90	shared: Arc<Shared>,
91}
92
93impl StreamingSound {
94	#[must_use]
95	pub(super) fn new<Error: Send + 'static>(
96		sample_rate: u32,
97		settings: StreamingSoundSettings,
98		shared: Arc<Shared>,
99		frame_consumer: Consumer<TimestampedFrame>,
100		command_readers: CommandReaders,
101		scheduler: &DecodeScheduler<Error>,
102	) -> Self {
103		let current_frame = scheduler.current_frame();
104		let start_position = current_frame as f64 / sample_rate as f64;
105		shared
106			.position
107			.store(start_position.to_bits(), Ordering::SeqCst);
108		Self {
109			command_readers,
110			sample_rate,
111			frame_consumer,
112			start_time: settings.start_time,
113			playback_state_manager: PlaybackStateManager::new(settings.fade_in_tween),
114			current_frame,
115			fractional_position: 0.0,
116			volume: Parameter::new(settings.volume, Decibels::IDENTITY),
117			playback_rate: Parameter::new(settings.playback_rate, PlaybackRate(1.0)),
118			panning: Parameter::new(settings.panning, Panning::CENTER),
119			shared,
120		}
121	}
122
123	fn update_shared_playback_state(&mut self) {
124		self.shared
125			.set_state(self.playback_state_manager.playback_state());
126	}
127
128	fn update_current_frame(&mut self) {
129		let chunk = self
130			.frame_consumer
131			.read_chunk(self.frame_consumer.slots().min(4))
132			.unwrap();
133		let (a, b) = chunk.as_slices();
134		let mut iter = a.iter().chain(b.iter());
135		if let Some(TimestampedFrame { index, .. }) = iter.nth(1) {
136			self.current_frame = *index;
137		}
138	}
139
140	#[must_use]
141	fn next_frames(&mut self) -> [Frame; 4] {
142		let mut frames = [Frame::ZERO; 4];
143		let chunk = self
144			.frame_consumer
145			.read_chunk(self.frame_consumer.slots().min(4))
146			.unwrap();
147		let (a, b) = chunk.as_slices();
148		let mut iter = a.iter().chain(b.iter());
149		for frame in &mut frames {
150			*frame = iter
151				.next()
152				.copied()
153				.map(|TimestampedFrame { frame, .. }| frame)
154				.unwrap_or(Frame::ZERO);
155		}
156		frames
157	}
158
159	#[must_use]
160	fn position(&self) -> f64 {
161		(self.current_frame as f64 + self.fractional_position) / self.sample_rate as f64
162	}
163
164	fn pause(&mut self, fade_out_tween: Tween) {
165		self.playback_state_manager.pause(fade_out_tween);
166		self.update_shared_playback_state();
167	}
168
169	fn resume(&mut self, start_time: StartTime, fade_in_tween: Tween) {
170		self.playback_state_manager
171			.resume(start_time, fade_in_tween);
172		self.update_shared_playback_state();
173	}
174
175	fn stop(&mut self, fade_out_tween: Tween) {
176		self.playback_state_manager.stop(fade_out_tween);
177		self.update_shared_playback_state();
178	}
179
180	fn read_commands(&mut self) {
181		read_commands_into_parameters!(self, volume, playback_rate, panning);
182		if let Some(tween) = self.command_readers.pause.read() {
183			self.pause(tween);
184		}
185		if let Some((start_time, tween)) = self.command_readers.resume.read() {
186			self.resume(start_time, tween);
187		}
188		if let Some(tween) = self.command_readers.stop.read() {
189			self.stop(tween);
190		}
191	}
192}

194impl Sound for StreamingSound {
195	fn on_start_processing(&mut self) {
196		self.update_current_frame();
197		self.shared
198			.position
199			.store(self.position().to_bits(), Ordering::SeqCst);
200		self.read_commands();
201	}
202
203	fn process(&mut self, out: &mut [Frame], dt: f64, info: &Info) {
204		if self.shared.encountered_error() {
205			self.playback_state_manager.mark_as_stopped();
206			self.update_shared_playback_state();
207			out.fill(Frame::ZERO);
208			return;
209		}
210
211		// update parameters
212		self.volume.update(dt * out.len() as f64, info);
213		self.playback_rate.update(dt * out.len() as f64, info);
214		self.panning.update(dt * out.len() as f64, info);
215		let changed_playback_state = self
216			.playback_state_manager
217			.update(dt * out.len() as f64, info);
218		if changed_playback_state {
219			self.update_shared_playback_state();
220		}
221
222		let will_never_start = self.start_time.update(dt * out.len() as f64, info);
223		if will_never_start {
224			self.playback_state_manager.mark_as_stopped();
225			self.update_shared_playback_state();
226		}
227		if self.start_time != StartTime::Immediate {
228			out.fill(Frame::ZERO);
229			return;
230		}
231
232		if !self.playback_state_manager.playback_state().is_advancing() {
233			out.fill(Frame::ZERO);
234			return;
235		}
236		// pause playback while waiting for audio data. the first frame
237		// in the ringbuffer is the previous frame, so we need to make
238		// sure there's at least 2 before we continue playing.
239		if self.frame_consumer.slots() < 2 && !self.shared.reached_end() {
240			out.fill(Frame::ZERO);
241			return;
242		}
243
244		let num_frames = out.len();
245		for (i, frame) in out.iter_mut().enumerate() {
246			let time_in_chunk = (i + 1) as f64 / num_frames as f64;
247			let volume = self.volume.interpolated_value(time_in_chunk).as_amplitude();
248			let fade_volume = self
249				.playback_state_manager
250				.interpolated_fade_volume(time_in_chunk)
251				.as_amplitude();
252			let panning = self.panning.interpolated_value(time_in_chunk);
253			let playback_rate = self.playback_rate.interpolated_value(time_in_chunk);
254			let next_frames = self.next_frames();
255			let interpolated_out = interpolate_frame(
256				next_frames[0],
257				next_frames[1],
258				next_frames[2],
259				next_frames[3],
260				self.fractional_position as f32,
261			);
262			self.fractional_position += self.sample_rate as f64 * playback_rate.0.max(0.0) * dt;
263			while self.fractional_position >= 1.0 {
264				self.fractional_position -= 1.0;
265				self.frame_consumer.pop().ok();
266			}
267			if self.shared.reached_end() && self.frame_consumer.is_empty() {
268				self.playback_state_manager.mark_as_stopped();
269				self.update_shared_playback_state();
270			}
271			*frame = (interpolated_out * fade_volume * volume).panned(panning);
272		}
273	}
274
275	fn finished(&self) -> bool {
276		self.playback_state_manager.playback_state() == PlaybackState::Stopped
277	}
278}
