1/*!
2Decodes data gradually from an audio file.
3
4To play a streaming sound, pass a [`StreamingSoundData`] to
5[`AudioManager::play`](crate::AudioManager::play).
6
7```no_run
8use kira::{
9	AudioManager, AudioManagerSettings, DefaultBackend,
10	sound::streaming::{StreamingSoundData, StreamingSoundSettings},
11};
12
13let mut manager = AudioManager::<DefaultBackend>::new(AudioManagerSettings::default())?;
14let sound_data = StreamingSoundData::from_file("sound.ogg")?;
15manager.play(sound_data)?;
16# Result::<(), Box<dyn std::error::Error>>::Ok(())
17```
18
19Streaming sounds use less memory than static sounds, but they use more
20CPU, and they can have delays when starting or seeking.
21*/
