4/// A point in time in a piece of audio.
5#[derive(Debug, Clone, Copy, PartialEq)]
6#[cfg_attr(feature = "serde", derive(serde::Serialize, serde::Deserialize))]
7pub enum PlaybackPosition {
8	/// The time in seconds.
9	Seconds(f64),
10	/// The time in samples (individual audio data points).
11	Samples(usize),
12}
13
14impl PlaybackPosition {
15	#[must_use]
16	pub(crate) fn into_samples(self, sample_rate: u32) -> usize {
17		match self {
18			PlaybackPosition::Seconds(seconds) => (seconds * sample_rate as f64).round() as usize,
19			PlaybackPosition::Samples(samples) => samples,
20		}
21	}
22}
23
24impl From<f64> for PlaybackPosition {
25	fn from(v: f64) -> Self {
26		Self::Seconds(v)
27	}
28}
29
30impl Default for PlaybackPosition {
31	fn default() -> Self {
32		Self::Seconds(0.0)
33	}
34}
