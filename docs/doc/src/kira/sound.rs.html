1/*!
2Sources of audio.
3
4Any type that implements [`SoundData`] can be played using
5[`AudioManager::play`](crate::AudioManager::play). Kira comes with two
6[`SoundData`] implementations:
7
8- [`StaticSoundData`](static_sound::StaticSoundData), which loads an entire chunk of audio
9  into memory. This is more appropriate for short sounds, sounds you want to play multiple times,
10  or sounds where consistent start times are important.
11- [`StreamingSoundData`](streaming::StreamingSoundData), which streams audio from a file or cursor
12  (only available on desktop platforms). This is more appropriate for long sounds that you only
13  play once at a time, like background music. Streaming sounds use less memory than static sounds.
14
15These two sound types should cover most use cases, but if you need something else, you can
16create your own types that implement the [`SoundData`] and [`Sound`] traits.
17*/

37/// A source of audio that is loaded, but not yet playing.
38pub trait SoundData {
39	/// Errors that can occur when starting the sound.
40	type Error;
41
42	/// The type that can be used to control the sound once
43	/// it has started.
44	type Handle;
45
46	/// Converts the loaded sound into a live, playing sound
47	/// and a handle to control it.
48	///
49	/// The [`Sound`] implementation will be sent to the audio renderer
50	/// for playback, and the handle will be returned to the user by
51	/// [`AudioManager::play`](crate::AudioManager::play).
52	#[allow(clippy::type_complexity)]
53	fn into_sound(self) -> Result<(Box<dyn Sound>, Self::Handle), Self::Error>;
54}

56/// An actively playing sound.
57///
58/// For performance reasons, the methods of this trait should not allocate
59/// or deallocate memory.
60#[allow(unused_variables)]
61pub trait Sound: Send {
62	/// Called whenever a new batch of audio samples is requested by the backend.
63	///
64	/// This is a good place to put code that needs to run fairly frequently,
65	/// but not for every single audio sample.
66	fn on_start_processing(&mut self) {}
67
68	/// Produces the next [`Frame`]s of audio. This should overwrite
69	/// the entire `out` slice with new audio.
70	///
71	/// `dt` is the time between each frame (in seconds).
72	fn process(&mut self, out: &mut [Frame], dt: f64, info: &Info);
73
74	/// Processes a single [`Frame`]. Mostly useful for testing.
75	///
76	/// `dt` is the time elapsed since the previous frame (in seconds).
77	fn process_one(&mut self, dt: f64, info: &Info) -> Frame {
78		let mut out = [Frame::ZERO];
79		self.process(&mut out, dt, info);
80		out[0]
81	}
82
83	/// Returns `true` if the sound is finished and can be unloaded.
84	///
85	/// For finite sounds, this will typically be when playback has reached the
86	/// end of the sound. For infinite sounds, this will typically be when the
87	/// handle for the sound is dropped.
88	#[must_use]
89	fn finished(&self) -> bool;
90}

92/// The playback state of a sound.
93#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
94#[cfg_attr(feature = "serde", derive(serde::Serialize, serde::Deserialize))]
95pub enum PlaybackState {
96	/// The sound is playing normally.
97	Playing,
98	/// The sound is fading out, and when the fade-out
99	/// is finished, playback will pause.
100	Pausing,
101	/// Playback is paused.
102	Paused,
103	/// The sound is paused, but is schedule to resume in the future.
104	WaitingToResume,
105	/// The sound is fading back in after being previously paused.
106	Resuming,
107	/// The sound is fading out, and when the fade-out
108	/// is finished, playback will stop.
109	Stopping,
110	/// The sound has stopped and can no longer be resumed.
111	Stopped,
112}
113
114impl PlaybackState {
115	/// Whether the sound is advancing and outputting audio given
116	/// its current playback state.
117	pub fn is_advancing(self) -> bool {
118		match self {
119			PlaybackState::Playing => true,
120			PlaybackState::Pausing => true,
121			PlaybackState::Paused => false,
122			PlaybackState::WaitingToResume => false,
123			PlaybackState::Resuming => true,
124			PlaybackState::Stopping => true,
125			PlaybackState::Stopped => false,
126		}
127	}
128}

130/// A portion of audio.
131#[derive(Debug, Clone, Copy, PartialEq, Default)]
132#[cfg_attr(feature = "serde", derive(serde::Serialize, serde::Deserialize))]
133pub struct Region {
134	/// The starting time of the region (in seconds).
135	pub start: PlaybackPosition,
136	/// The (exclusive) ending time of the region.
137	pub end: EndPosition,
138}
139
140impl<T: Into<PlaybackPosition>> From<RangeFrom<T>> for Region {
141	fn from(range: RangeFrom<T>) -> Self {
142		Self {
143			start: range.start.into(),
144			end: EndPosition::EndOfAudio,
145		}
146	}
147}
148
149impl<T: Into<PlaybackPosition>> From<Range<T>> for Region {
150	fn from(range: Range<T>) -> Self {
151		Self {
152			start: range.start.into(),
153			end: EndPosition::Custom(range.end.into()),
154		}
155	}
156}
157
158impl<T: Into<PlaybackPosition>> From<RangeTo<T>> for Region {
159	fn from(range: RangeTo<T>) -> Self {
160		Self {
161			start: PlaybackPosition::Samples(0),
162			end: EndPosition::Custom(range.end.into()),
163		}
164	}
165}
166
167impl From<RangeFull> for Region {
168	fn from(_: RangeFull) -> Self {
169		Self {
170			start: PlaybackPosition::Samples(0),
171			end: EndPosition::EndOfAudio,
172		}
173	}
174}

176/// A trait for types that can be converted into an `Option<Region>`.
177pub trait IntoOptionalRegion {
178	/// Converts the type into an `Option<Region>`.
179	#[must_use]
180	fn into_optional_region(self) -> Option<Region>;
181}
182
183impl<T: Into<Region>> IntoOptionalRegion for T {
184	fn into_optional_region(self) -> Option<Region> {
185		Some(self.into())
186	}
187}
188
189impl IntoOptionalRegion for Option<Region> {
190	fn into_optional_region(self) -> Option<Region> {
191		self
192	}
193}
