23const MIN_Q: f64 = 0.01;
24
25struct EqFilter {
26	command_readers: CommandReaders,
27	kind: EqFilterKind,
28	frequency: Parameter,
29	gain: Parameter<Decibels>,
30	q: Parameter,
31	ic1eq: Frame,
32	ic2eq: Frame,
33}
34
35impl EqFilter {
36	#[must_use]
37	fn new(builder: EqFilterBuilder, command_readers: CommandReaders) -> Self {
38		Self {
39			command_readers,
40			kind: builder.kind,
41			frequency: Parameter::new(builder.frequency, 500.0),
42			gain: Parameter::new(builder.gain, Decibels::IDENTITY),
43			q: Parameter::new(builder.q, 1.0),
44			ic1eq: Frame::ZERO,
45			ic2eq: Frame::ZERO,
46		}
47	}
48}
49
50impl Effect for EqFilter {
51	fn on_start_processing(&mut self) {
52		if let Some(kind) = self.command_readers.set_kind.read() {
53			self.kind = kind;
54		}
55		read_commands_into_parameters!(self, frequency, gain, q);
56	}
57
58	fn process(&mut self, input: &mut [Frame], dt: f64, info: &Info) {
59		self.frequency.update(dt * input.len() as f64, info);
60		self.gain.update(dt * input.len() as f64, info);
61		self.q.update(dt * input.len() as f64, info);
62
63		let num_frames = input.len();
64		for (i, frame) in input.iter_mut().enumerate() {
65			let time_in_chunk = (i + 1) as f64 / num_frames as f64;
66			let frequency = self.frequency.interpolated_value(time_in_chunk);
67			let q = self.q.interpolated_value(time_in_chunk);
68			let gain = self.gain.interpolated_value(time_in_chunk);
69
70			let Coefficients {
71				a1,
72				a2,
73				a3,
74				m0,
75				m1,
76				m2,
77			} = Coefficients::calculate(self.kind, frequency, q, gain, dt);
78			let v3 = *frame - self.ic2eq;
79			let v1 = self.ic1eq * (a1 as f32) + v3 * (a2 as f32);
80			let v2 = self.ic2eq + self.ic1eq * (a2 as f32) + v3 * (a3 as f32);
81			self.ic1eq = v1 * 2.0 - self.ic1eq;
82			self.ic2eq = v2 * 2.0 - self.ic2eq;
83			*frame = *frame * (m0 as f32) + v1 * (m1 as f32) + v2 * (m2 as f32)
84		}
85	}
86}

88/// The shape of the frequency adjustment curve.
89#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
90#[cfg_attr(feature = "serde", derive(serde::Serialize, serde::Deserialize))]
91pub enum EqFilterKind {
92	/// Frequencies around the user-defined frequency are adjusted.
93	Bell,
94	/// Frequencies around and lower than the user-defined frequency are adjusted.
95	LowShelf,
96	/// Frequencies around and higher than the user-defined frequency are adjusted.
97	HighShelf,
98}
99
100struct Coefficients {
101	a1: f64,
102	a2: f64,
103	a3: f64,
104	m0: f64,
105	m1: f64,
106	m2: f64,
107}
108
109impl Coefficients {
110	#[must_use]
111	fn calculate(kind: EqFilterKind, frequency: f64, q: f64, gain: Decibels, dt: f64) -> Self {
112		// In my testing, the filter goes unstable when the frequency exceeds half the sample rate,
113		// so I'm clamping this value to 0.5
114		let relative_frequency = (frequency * dt).clamp(0.0001, 0.5);
115		let q = q.max(MIN_Q);
116		match kind {
117			EqFilterKind::Bell => {
118				let a = 10.0f64.powf(gain.0 as f64 / 40.0);
119				let g = (PI * relative_frequency).tan();
120				let k = 1.0 / (q * a);
121				let a1 = 1.0 / (1.0 + g * (g + k));
122				let a2 = g * a1;
123				let a3 = g * a2;
124				let m0 = 1.0;
125				let m1 = k * (a * a - 1.0);
126				let m2 = 0.0;
127				Self {
128					a1,
129					a2,
130					a3,
131					m0,
132					m1,
133					m2,
134				}
135			}
136			EqFilterKind::LowShelf => {
137				let a = 10.0f64.powf(gain.0 as f64 / 40.0);
138				let g = (PI * relative_frequency).tan() / a.sqrt();
139				let k = 1.0 / q;
140				let a1 = 1.0 / (1.0 + g * (g + k));
141				let a2 = g * a1;
142				let a3 = g * a2;
143				let m0 = 1.0;
144				let m1 = k * (a - 1.0);
145				let m2 = a * a - 1.0;
146				Self {
147					a1,
148					a2,
149					a3,
150					m0,
151					m1,
152					m2,
153				}
154			}
155			EqFilterKind::HighShelf => {
156				let a = 10.0f64.powf(gain.0 as f64 / 40.0);
157				let g = (PI * relative_frequency).tan() * a.sqrt();
158				let k = 1.0 / q;
159				let a1 = 1.0 / (1.0 + g * (g + k));
160				let a2 = g * a1;
161				let a3 = g * a2;
162				let m0 = a * a;
163				let m1 = k * (1.0 - a) * a;
164				let m2 = 1.0 - a * a;
165				Self {
166					a1,
167					a2,
168					a3,
169					m0,
170					m1,
171					m2,
172				}
173			}
174		}
175	}
176}
