10/// Configures a compressor.
11pub struct CompressorBuilder {
12	/// The volume above which volume will start to be decreased (in decibels).
13	pub threshold: Value<f64>,
14	/// How much the signal will be compressed.
15	///
16	/// A ratio of `2.0` (or 2 to 1) means an increase of 3dB will
17	/// become an increase of 1.5dB. Ratios between `0.0` and `1.0`
18	/// will actually expand the audio.
19	pub ratio: Value<f64>,
20	/// How much time it takes for the volume attenuation to ramp up once
21	/// the input volume exceeds the threshold.
22	pub attack_duration: Value<Duration>,
23	/// How much time it takes for the volume attenuation to relax once
24	/// the input volume dips below the threshold.
25	pub release_duration: Value<Duration>,
26	/// The amount to change the volume after processing (in dB).
27	///
28	/// This can be used to compensate for the decrease in volume resulting
29	/// from compression. This is only applied to the wet signal, nto the
30	/// dry signal.
31	pub makeup_gain: Value<Decibels>,
32	/// How much dry (unprocessed) signal should be blended
33	/// with the wet (processed) signal.
34	pub mix: Value<Mix>,
35}
36
37impl CompressorBuilder {
38	pub(crate) const DEFAULT_THRESHOLD: f64 = 0.0;
39	pub(crate) const DEFAULT_RATIO: f64 = 1.0;
40	pub(crate) const DEFAULT_ATTACK_DURATION: Duration = Duration::from_millis(10);
41	pub(crate) const DEFAULT_RELEASE_DURATION: Duration = Duration::from_millis(100);
42	pub(crate) const DEFAULT_MAKEUP_GAIN: Decibels = Decibels(0.0);
43	pub(crate) const DEFAULT_MIX: Mix = Mix::WET;
44
45	/// Creates a new [`CompressorBuilder`] with the default settings.

54			mix: Value::Fixed(Self::DEFAULT_MIX),
55		}
56	}
