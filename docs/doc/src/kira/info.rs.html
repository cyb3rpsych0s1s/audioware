17/// Provides info about resources on the audio thread.
18///
19/// You'll only need this if you're implementing one of Kira's traits,
20/// like [`Sound`](crate::sound::Sound) or [`Effect`](crate::effect::Effect).
21pub struct Info<'a> {
22	kind: InfoKind<'a>,
23	spatial_track_info: Option<SpatialTrackInfo>,
24}
25
26impl<'a> Info<'a> {
27	pub(crate) fn new(
28		clocks: &'a Arena<Clock>,
29		modulators: &'a Arena<Box<dyn Modulator>>,
30		listeners: &'a Arena<Listener>,
31		spatial_track_info: Option<SpatialTrackInfo>,
32	) -> Self {
33		Self {
34			kind: InfoKind::Real {
35				clocks,
36				modulators,
37				listeners,
38			},
39			spatial_track_info,
40		}
41	}
42
43	/// Gets information about the clock with the given ID if it
44	/// exists, returns `None` otherwise.
45	#[must_use]
46	pub fn clock_info(&self, id: ClockId) -> Option<ClockInfo> {
47		match &self.kind {
48			InfoKind::Real { clocks, .. } => clocks.get(id.0).map(|clock| ClockInfo {
49				ticking: clock.ticking(),
50				time: ClockTime {
51					clock: id,
52					ticks: match clock.state() {
53						ClockState::NotStarted => 0,
54						ClockState::Started { ticks, .. } => ticks,
55					},
56					fraction: match clock.state() {
57						ClockState::NotStarted => 0.0,
58						ClockState::Started {
59							fractional_position,
60							..
61						} => fractional_position,
62					},
63				},
64			}),
65			InfoKind::Mock { clock_info, .. } => clock_info.get(id.0).copied(),
66		}
67	}
68
69	/// Returns whether something with the given start time should
70	/// start now, later, or never given the current state of the clocks.
71	#[must_use]
72	pub fn when_to_start(&self, time: ClockTime) -> WhenToStart {
73		if let Some(clock_info) = self.clock_info(time.clock) {
74			if clock_info.ticking && clock_info.time >= time {
75				WhenToStart::Now
76			} else {
77				WhenToStart::Later
78			}
79		} else {
80			WhenToStart::Never
81		}
82	}
83
84	/// Gets the value of the modulator with the given ID if it
85	/// exists, returns `None` otherwise.
86	#[must_use]
87	pub fn modulator_value(&self, id: ModulatorId) -> Option<f64> {
88		match &self.kind {
89			InfoKind::Real { modulators, .. } => {
90				modulators.get(id.0).map(|modulator| modulator.value())
91			}
92			InfoKind::Mock {
93				modulator_values, ..
94			} => modulator_values.get(id.0).copied(),
95		}
96	}
97
98	/// Gets information about the listener linked to the current spatial track
99	/// if there is one.
100	#[must_use]
101	pub fn listener_info(&self) -> Option<ListenerInfo> {
102		self.spatial_track_info.and_then(|spatial_track_info| {
103			let listener_id = spatial_track_info.listener_id;
104			match &self.kind {
105				InfoKind::Real { listeners, .. } => {
106					listeners.get(listener_id.0).map(|listener| ListenerInfo {
107						position: listener.position.value().into(),
108						orientation: listener.orientation.value().into(),
109						previous_position: listener.position.previous_value().into(),
110						previous_orientation: listener.orientation.previous_value().into(),
111					})
112				}
113				InfoKind::Mock { listener_info, .. } => listener_info.get(listener_id.0).copied(),
114			}
115		})
116	}
117
118	/// If this is called from an effect on a spatial track, returns the distance
119	/// of the spatial track's from the spatial track. Otherwise, returns `None`.
120	pub fn listener_distance(&self) -> Option<f32> {
121		self.spatial_track_info.zip(self.listener_info()).map(
122			|(spatial_track_info, listener_info)| {
123				Vec3::from(listener_info.position).distance(spatial_track_info.position)
124			},
125		)
126	}
127}

129/// Information about the current state of a [clock](super::clock).
130#[derive(Debug, Clone, Copy, PartialEq)]
131pub struct ClockInfo {
132	/// Whether the clock is currently running.
133	pub ticking: bool,
134	/// The current time of the clock.
135	pub time: ClockTime,
136}
137
138/// When something should start given the current state
139/// of the clocks.
140///
141/// The "something" in question can be anything that
142/// would start at a given [`StartTime`](crate::StartTime).
143#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
144pub enum WhenToStart {
145	/// The thing should start now.
146	Now,
147	/// The thing should start later because the appropriate
148	/// clock isn't ticking or hasn't reached the target tick
149	/// yet.
150	Later,
151	/// The thing will never start because the clock it depends
152	/// on no longer exists.
153	Never,
154}

156/// Information about a listener.
157#[derive(Debug, Clone, Copy, PartialEq)]
158pub struct ListenerInfo {
159	/// The position of the listener.
160	pub position: mint::Vector3<f32>,
161	/// The rotation of the listener.
162	pub orientation: mint::Quaternion<f32>,
163	/// The position of the listener prior to the last update.
164	pub previous_position: mint::Vector3<f32>,
165	/// The rotation of the listener prior to the last update.
166	pub previous_orientation: mint::Quaternion<f32>,
167}
168
169impl ListenerInfo {
170	/// Returns the interpolated position between the previous and current
171	/// position of the listener.
172	pub fn interpolated_position(self, amount: f32) -> mint::Vector3<f32> {
173		let position: Vec3 = self.position.into();
174		let previous_position: Vec3 = self.previous_position.into();
175		previous_position.lerp(position, amount).into()
176	}
177
178	/// Returns the interpolated orientation between the previous and current
179	/// orientation of the listener.
180	pub fn interpolated_orientation(self, amount: f32) -> mint::Quaternion<f32> {
181		let orientation: Quat = self.orientation.into();
182		let previous_orientation: Quat = self.previous_orientation.into();
183		previous_orientation.lerp(orientation, amount).into()
184	}
185}

187/// Generates a fake `Info` with arbitrary data. Useful for writing unit tests.
188pub struct MockInfoBuilder {
189	clock_info: Arena<ClockInfo>,
190	modulator_values: Arena<f64>,
191	listener_info: Arena<ListenerInfo>,
192	spatial_track_info: Option<SpatialTrackInfo>,
193}
194
195impl MockInfoBuilder {
196	/// Creates a new `MockInfoBuilder`.
197	pub fn new() -> Self {
198		Self {
199			clock_info: Arena::new(100),
200			modulator_values: Arena::new(100),
201			listener_info: Arena::new(100),
202			spatial_track_info: None,
203		}
204	}
205
206	/// Adds a fake clock with the given ticking state and time. Returns a fake
207	/// `ClockId`.
208	pub fn add_clock(&mut self, ticking: bool, ticks: u64, fraction: f64) -> ClockId {
209		let id = ClockId(
210			self.clock_info
211				.controller()
212				.try_reserve()
213				.expect("clock info arena is full"),
214		);
215		self.clock_info
216			.insert_with_key(
217				id.0,
218				ClockInfo {
219					ticking,
220					time: ClockTime {
221						clock: id,
222						ticks,
223						fraction,
224					},
225				},
226			)
227			.unwrap();
228		id
229	}
230
231	/// Adds a fake modulator outputting the given value. Returns a fake `ModulatorId`.
232	pub fn add_modulator(&mut self, value: f64) -> ModulatorId {
233		let id = ModulatorId(
234			self.modulator_values
235				.controller()
236				.try_reserve()
237				.expect("modulator info arena is full"),
238		);
239		self.modulator_values.insert_with_key(id.0, value).unwrap();
240		id
241	}
242
243	/// Adds a fake listener at the given position and orientation. Returns a fake `ListenerId`.
244	pub fn add_listener(&mut self, listener_info: ListenerInfo) -> ListenerId {
245		let id = ListenerId(
246			self.listener_info
247				.controller()
248				.try_reserve()
249				.expect("listener info arena is full"),
250		);
251		self.listener_info
252			.insert_with_key(id.0, listener_info)
253			.unwrap();
254		id
255	}
256
257	/// Consumes the `MockInfoProvider` and returns a fake `Info`.
258	pub fn build(self) -> Info<'static> {
259		Info {
260			kind: InfoKind::Mock {
261				clock_info: self.clock_info,
262				modulator_values: self.modulator_values,
263				listener_info: self.listener_info,
264			},
265			spatial_track_info: self.spatial_track_info,
266		}
267	}
268}

273	}
