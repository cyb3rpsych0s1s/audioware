20/// Controls a mixer track.
21///
22/// When a [`TrackHandle`] is dropped, the corresponding mixer
23/// track will be removed.
24#[derive(Debug)]
25pub struct TrackHandle {
26	pub(crate) renderer_shared: Arc<RendererShared>,
27	pub(crate) shared: Arc<TrackShared>,
28	pub(crate) command_writers: CommandWriters,
29	pub(crate) sound_controller: ResourceController<Box<dyn Sound>>,
30	pub(crate) sub_track_controller: ResourceController<Track>,
31	pub(crate) send_volume_command_writers:
32		HashMap<SendTrackId, CommandWriter<ValueChangeCommand<Decibels>>>,
33	pub(crate) internal_buffer_size: usize,
34}
35
36impl TrackHandle {
37	/// Returns the current playback state of the track.
38	#[must_use]
39	pub fn state(&self) -> TrackPlaybackState {
40		self.shared.state()
41	}
42
43	/// Plays a sound.
44	pub fn play<D: SoundData>(
45		&mut self,
46		sound_data: D,
47	) -> Result<D::Handle, PlaySoundError<D::Error>> {
48		let (sound, handle) = sound_data
49			.into_sound()
50			.map_err(PlaySoundError::IntoSoundError)?;
51		self.sound_controller
52			.insert(sound)
53			.map_err(|_| PlaySoundError::SoundLimitReached)?;
54		Ok(handle)
55	}
56
57	/// Adds a child track to this track.
58	pub fn add_sub_track(
59		&mut self,
60		builder: TrackBuilder,
61	) -> Result<TrackHandle, ResourceLimitReached> {
62		let (mut track, handle) =
63			builder.build(self.renderer_shared.clone(), self.internal_buffer_size);
64		track.init_effects(self.renderer_shared.sample_rate.load(Ordering::SeqCst));
65		self.sub_track_controller.insert(track)?;
66		Ok(handle)
67	}
68
69	/// Adds a spatial child track to this track.
70	pub fn add_spatial_sub_track(
71		&mut self,
72		listener: impl Into<ListenerId>,
73		position: impl Into<Value<mint::Vector3<f32>>>,
74		builder: SpatialTrackBuilder,
75	) -> Result<SpatialTrackHandle, ResourceLimitReached> {
76		let (mut track, handle) = builder.build(
77			self.renderer_shared.clone(),
78			self.internal_buffer_size,
79			listener.into(),
80			position.into().to_(),
81		);
82		track.init_effects(self.renderer_shared.sample_rate.load(Ordering::SeqCst));
83		self.sub_track_controller.insert(track)?;
84		Ok(handle)
85	}
86
87	/// Sets the (post-effects) volume of the mixer track.
88	pub fn set_volume(&mut self, volume: impl Into<Value<Decibels>>, tween: Tween) {
89		self.command_writers.set_volume.write(ValueChangeCommand {
90			target: volume.into(),
91			tween,
92		})
93	}
94
95	/// Sets the volume of this track's route to a send track.
96	///
97	/// This can only be used to change the volume of existing routes,
98	/// not to add new routes.
99	pub fn set_send(
100		&mut self,
101		to: impl Into<SendTrackId>,
102		volume: impl Into<Value<Decibels>>,
103		tween: Tween,
104	) -> Result<(), NonexistentRoute> {
105		let to = to.into();
106		self.send_volume_command_writers
107			.get_mut(&to)
108			.ok_or(NonexistentRoute)?
109			.write(ValueChangeCommand {
110				target: volume.into(),
111				tween,
112			});
113		Ok(())
114	}
115
116	/// Fades out the track to silence with the given tween and then
117	/// pauses playback, pausing all sounds and emitters playing on this
118	/// track.
119	pub fn pause(&mut self, tween: Tween) {
120		self.command_writers.pause.write(tween)
121	}
122
123	/// Resumes playback and fades in the sound from silence
124	/// with the given tween, resuming all sounds and emitters
125	/// playing on this track.
126	pub fn resume(&mut self, tween: Tween) {
127		self.resume_at(StartTime::Immediate, tween)
128	}
129
130	/// Resumes playback at the given start time and fades in
131	/// the sound from silence with the given tween.
132	pub fn resume_at(&mut self, start_time: StartTime, tween: Tween) {
133		self.command_writers.resume.write((start_time, tween))
134	}
135
136	/// Returns the maximum number of sounds that can play simultaneously on this track.
137	#[must_use]
138	pub fn sound_capacity(&self) -> usize {
139		self.sound_controller.capacity()
140	}
141
142	/// Returns the number of sounds currently playing on this track.
143	#[must_use]
144	pub fn num_sounds(&self) -> usize {
145		self.sound_controller.len()
146	}
147
148	/// Returns the maximum number of child tracks this track can have.
149	#[must_use]
150	pub fn sub_track_capacity(&self) -> usize {
151		self.sub_track_controller.capacity()
152	}
153
154	/// Returns the number of child tracks this track has.
155	#[must_use]
156	pub fn num_sub_tracks(&self) -> usize {
157		self.sub_track_controller.len()
158	}
159}

161impl Drop for TrackHandle {
162	fn drop(&mut self) {

164	}
