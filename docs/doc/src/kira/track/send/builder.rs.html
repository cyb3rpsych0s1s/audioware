10/// Configures a mixer track.
11pub struct SendTrackBuilder {
12	/// The volume of the send track.
13	pub(crate) volume: Value<Decibels>,
14	/// The effects that should be applied to the input audio
15	/// for this track.
16	pub(crate) effects: Vec<Box<dyn Effect>>,
17}

19impl SendTrackBuilder {
20	/// Creates a new [`SendTrackBuilder`] with the default settings.
21	#[must_use]
22	pub fn new() -> Self {
23		Self {
24			volume: Value::Fixed(Decibels::IDENTITY),
25			effects: vec![],
26		}
27	}

29	/**
30	Sets the volume of the send track.
31
32	# Examples
33
34	Set the volume to a fixed decibel value:
35
36	```
37	# use kira::track::SendTrackBuilder;
38	let builder = SendTrackBuilder::new().volume(-6.0);
39	```
40
41	Link the volume to a modulator:
42
43	```no_run
44	use kira::{
45		AudioManager, AudioManagerSettings, DefaultBackend,
46		modulator::tweener::TweenerBuilder,
47		track::SendTrackBuilder,
48		Easing, Value, Mapping,
49		Decibels,
50	};
51
52	let mut manager = AudioManager::<DefaultBackend>::new(AudioManagerSettings::default())?;
53	let tweener = manager.add_modulator(TweenerBuilder {
54		initial_value: 0.5,
55	})?;
56	let builder = SendTrackBuilder::new().volume(Value::FromModulator {
57		id: tweener.id(),
58		mapping: Mapping {
59			input_range: (0.0, 1.0),
60			output_range: (Decibels::SILENCE, Decibels::IDENTITY),
61			easing: Easing::Linear,
62		},
63	});
64	# Result::<(), Box<dyn std::error::Error>>::Ok(())
65	```
66	*/
67	#[must_use = "This method consumes self and returns a modified SendTrackBuilder, so the return value should be used"]
68	pub fn volume(self, volume: impl Into<Value<Decibels>>) -> Self {
69		Self {
70			volume: volume.into(),
71			..self
72		}
73	}

75	/**
76	Adds an effect to the send track.
77
78	# Examples
79
80	```
81	use kira::{track::SendTrackBuilder, effect::delay::DelayBuilder};
82
83	let mut builder = SendTrackBuilder::new();
84	let delay_handle = builder.add_effect(DelayBuilder::new());
85	```
86	*/
87	pub fn add_effect<B: EffectBuilder>(&mut self, builder: B) -> B::Handle {
88		let (effect, handle) = builder.build();
89		self.effects.push(effect);
90		handle
91	}
92
93	/**
94	Adds an effect to the send track and returns the [`SendTrackBuilder`].
95
96	If you need to modify the effect later, use [`add_effect`](Self::add_effect),
97	which returns the effect handle.
98
99	# Examples
100
101	```
102	use kira::{
103		track::SendTrackBuilder,
104		effect::{filter::FilterBuilder, reverb::ReverbBuilder},
105	};
106
107	let mut builder = SendTrackBuilder::new()
108		.with_effect(FilterBuilder::new())
109		.with_effect(ReverbBuilder::new());
110	```
111	*/
112	#[must_use = "This method consumes self and returns a modified SendTrackBuilder, so the return value should be used"]
113	pub fn with_effect<B: EffectBuilder>(mut self, builder: B) -> Self {
114		self.add_effect(builder);
115		self
116	}
117
118	/** Adds an already built effect into this track.
119
120	`Box<dyn Effect>` values are created when calling `build` on an effect builder, which gives you
121	an effect handle, as well as this boxed effect, which is the actual audio effect.
122
123	This is a lower-level method than [`Self::add_effect`], and you should probably use it rather
124	than this method, unless you have a reason to.
125
126	# Examples
127
128	```
129	use kira::track::SendTrackBuilder;
130	use kira::effect::{EffectBuilder, delay::DelayBuilder};
131
132	let mut builder = SendTrackBuilder::new();
133	let delay_builder = DelayBuilder::new();
134	let (effect, delay_handle) = delay_builder.build();
135	let delay_handle = builder.add_built_effect(effect);
136	```
137	*/
138	pub fn add_built_effect(&mut self, effect: Box<dyn Effect>) {
139		self.effects.push(effect);
140	}
141
142	/** Add an already-built effect and return the [`SendTrackBuilder`].
143
144	 `Box<dyn Effect>` values are created when calling `build` on an effect builder, which gives you
145	 an effect handle, as well as this boxed effect, which is the actual audio effect.
146
147	 This is a lower-level method than [`Self::with_effect`], and you should probably use it rather
148	 than this method, unless you have a reason to.
149
150	# Examples
151
152	```
153	use kira::{
154		track::SendTrackBuilder,
155		effect::{filter::FilterBuilder, reverb::ReverbBuilder, EffectBuilder},
156	};
157
158	let (filter_effect, filter_handle) = FilterBuilder::new().build();
159	let (reverb_effect, reverb_handle) = ReverbBuilder::new().build();
160	let mut builder = SendTrackBuilder::new()
161		.with_built_effect(filter_effect)
162		.with_built_effect(reverb_effect);
163	```
164	 */
165	#[must_use = "This method consumes self and returns a modified SendTrackBuilder, so the return value should be used"]
166	pub fn with_built_effect(mut self, effect: Box<dyn Effect>) -> Self {
167		self.add_built_effect(effect);
168		self
169	}
170
171	#[must_use]
172	pub(crate) fn build(
173		self,
174		id: SendTrackId,
175		internal_buffer_size: usize,
176	) -> (SendTrack, SendTrackHandle) {
177		let (set_volume_command_writer, set_volume_command_reader) = command_writer_and_reader();
178		let shared = Arc::new(TrackShared::new());
179		let track = SendTrack {
180			shared: shared.clone(),
181			volume: Parameter::new(self.volume, Decibels::IDENTITY),
182			set_volume_command_reader,
183			effects: self.effects,
184			input: vec![Frame::ZERO; internal_buffer_size],
185			internal_buffer_size,
186		};
187		let handle = SendTrackHandle {
188			id,
189			shared,
190			set_volume_command_writer,
191		};
192		(track, handle)
193	}
194}
