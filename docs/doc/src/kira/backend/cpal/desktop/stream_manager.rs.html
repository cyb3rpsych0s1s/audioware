35pub(super) struct StreamManagerController {
36	should_drop: Arc<AtomicBool>,
37	num_stream_errors_discarded: Arc<AtomicU64>,
38	handled_stream_error_consumer: Mutex<Consumer<StreamError>>,
39}

41impl StreamManagerController {
42	pub fn stop(&self) {
43		self.should_drop.store(true, Ordering::SeqCst);
44	}
45
46	#[must_use]
47	pub fn num_stream_errors_discarded(&self) -> u64 {
48		self.num_stream_errors_discarded.load(Ordering::Acquire)
49	}
50
51	pub fn pop_handled_error(&mut self) -> Option<StreamError> {
52		self.handled_stream_error_consumer
53			.get_mut()
54			.unwrap()
55			.pop()
56			.ok()
57	}

60/// Starts a cpal stream and restarts it if needed
61/// in the case of device changes or disconnections.
62pub(super) struct StreamManager {
63	state: State,
64	device_name: String,
65	sample_rate: u32,
66	custom_device: bool,
67	buffer_size: BufferSize,
68}
69
70impl StreamManager {
71	pub fn start(
72		renderer: RendererWithCpuUsage,
73		device: Device,
74		mut config: StreamConfig,
75		custom_device: bool,
76		buffer_size: BufferSize,
77	) -> Result<StreamManagerController, Error> {
78		let should_drop = Arc::new(AtomicBool::new(false));
79		let should_drop_clone = should_drop.clone();
80
81		let num_stream_errors_discarded = Arc::new(AtomicU64::new(0));
82		let num_stream_errors_discarded_clone = num_stream_errors_discarded.clone();
83
84		let (mut handled_stream_error_producer, handled_stream_error_consumer) =
85			RingBuffer::new(64);
86
87		let (mut initial_result_producer, mut initial_result_consumer) = RingBuffer::new(1);
88
89		std::thread::spawn(move || {
90			let mut stream_manager = StreamManager {
91				state: State::Idle { renderer },
92				device_name: device_name(&device),
93				sample_rate: config.sample_rate.0,
94				custom_device,
95				buffer_size,
96			};
97			let mut unhandled_stream_error_consumer = match stream_manager.start_stream(
98				&device,
99				&mut config,
100				num_stream_errors_discarded.clone(),
101			) {
102				Ok(unhandled_stream_error_consumer) => {
103					initial_result_producer.push(Ok(())).unwrap();
104					unhandled_stream_error_consumer
105				}
106				Err(err) => {
107					initial_result_producer.push(Err(err)).unwrap();
108					return;
109				}
110			};
111			loop {
112				std::thread::sleep(CHECK_STREAM_INTERVAL);
113				if should_drop.load(Ordering::SeqCst) {
114					break;
115				}
116				stream_manager.check_stream(
117					&mut unhandled_stream_error_consumer,
118					&mut handled_stream_error_producer,
119					&num_stream_errors_discarded,
120				);
121			}
122		});
123
124		loop {
125			if let Ok(result) = initial_result_consumer.pop() {
126				result?;
127				break;
128			}
129			std::thread::sleep(Duration::from_micros(100));
130		}
131
132		Ok(StreamManagerController {
133			should_drop: should_drop_clone,
134			num_stream_errors_discarded: num_stream_errors_discarded_clone,
135			handled_stream_error_consumer: Mutex::new(handled_stream_error_consumer),
136		})
137	}
138
139	/// Restarts the stream if the audio device gets disconnected.
140	fn check_stream(
141		&mut self,
142		unhandled_stream_error_consumer: &mut Consumer<StreamError>,
143		handled_stream_error_producer: &mut Producer<StreamError>,
144		num_stream_errors_discarded: &Arc<AtomicU64>,
145	) {
146		if let State::Running { .. } = &self.state {
147			while let Ok(error) = unhandled_stream_error_consumer.pop() {
148				match error {
149					// check for device disconnection
150					StreamError::DeviceNotAvailable => {
151						self.stop_stream();
152						if let Ok((device, mut config)) = default_device_and_config() {
153							// TODO: gracefully handle errors that occur in this function
154							*unhandled_stream_error_consumer = self
155								.start_stream(
156									&device,
157									&mut config,
158									num_stream_errors_discarded.clone(),
159								)
160								.unwrap();
161						}
162					}
163					StreamError::BackendSpecific { err: _ } => {}
164				}
165				match handled_stream_error_producer.push(error) {
166					Ok(()) => {}
167					Err(PushError::Full(_stream_error)) => {
168						num_stream_errors_discarded.fetch_add(1, Ordering::AcqRel);
169					}
170				}
171			}
172			// check for device changes if a custom device hasn't been specified
173			// Disabled on macos due to audio artifacts that seem to occur when the device is
174			// queried while playing.
175			// see: https://github.com/tesselode/kira/issues/38
176			#[cfg(not(target_os = "macos"))]
177			if !self.custom_device {
178				if let Ok((device, mut config)) = default_device_and_config() {
179					let device_name = device_name(&device);
180					let sample_rate = config.sample_rate.0;
181					if device_name != self.device_name || sample_rate != self.sample_rate {
182						self.stop_stream();
183						*unhandled_stream_error_consumer = self
184							.start_stream(&device, &mut config, num_stream_errors_discarded.clone())
185							.unwrap();
186					}
187				}
188			}
189		}
190	}
191
192	fn start_stream(
193		&mut self,
194		device: &Device,
195		config: &mut StreamConfig,
196		num_stream_errors_discarded: Arc<AtomicU64>,
197	) -> Result<Consumer<StreamError>, Error> {
198		let mut renderer =
199			if let State::Idle { renderer } = std::mem::replace(&mut self.state, State::Empty) {
200				renderer
201			} else {
202				panic!("trying to start a stream when the stream manager is not idle");
203			};
204		config.buffer_size = self.buffer_size; // this won't change anything if the buffer size is BufferSize::Default
205		let device_name = device_name(device);
206		let sample_rate = config.sample_rate.0;
207		if sample_rate != self.sample_rate {
208			renderer.on_change_sample_rate(sample_rate);
209		}
210		self.device_name = device_name;
211		self.sample_rate = sample_rate;
212		let (mut renderer_wrapper, renderer_consumer) = SendOnDrop::new(renderer);
213		let (mut unhandled_stream_error_producer, unhandled_stream_error_consumer) =
214			RingBuffer::new(64);
215		let channels = config.channels;
216		let stream = device.build_output_stream(
217			config,
218			move |data: &mut [f32], _| {
219				#[cfg(feature = "assert_no_alloc")]
220				assert_no_alloc::assert_no_alloc(|| {
221					process_renderer(&mut renderer_wrapper, data, channels, sample_rate);
222				});
223				#[cfg(not(feature = "assert_no_alloc"))]
224				process_renderer(&mut renderer_wrapper, data, channels, sample_rate);
225			},
226			move |error| match unhandled_stream_error_producer.push(error) {
227				Ok(()) => {}
228				Err(PushError::Full(_stream_error)) => {
229					num_stream_errors_discarded.fetch_add(1, Ordering::AcqRel);
230				}
231			},
232			None,
233		)?;
234		stream.play()?;
235		self.state = State::Running {
236			stream,
237			renderer_consumer,
238		};
239		Ok(unhandled_stream_error_consumer)
240	}
241
242	fn stop_stream(&mut self) {
243		if let State::Running {
244			mut renderer_consumer,
245			stream,
246			..
247		} = std::mem::replace(&mut self.state, State::Empty)
248		{
249			drop(stream);
250			let renderer = renderer_consumer
251				.pop()
252				.expect("Could not retrieve the renderer after dropping a stream");
253			self.state = State::Idle { renderer };
254		} else {
255			panic!("Trying to stop the stream when it's not running")
256		}
257	}
258}

260fn default_device_and_config() -> Result<(Device, StreamConfig), Error> {
261	let host = cpal::default_host();
262	let device = host
263		.default_output_device()
264		.ok_or(Error::NoDefaultOutputDevice)?;
265	let config = device.default_output_config()?.config();
266	Ok((device, config))
267}
268
269fn device_name(device: &Device) -> String {
270	device
271		.name()
272		.unwrap_or_else(|_| "device name unavailable".to_string())
273}

275fn process_renderer(
276	renderer: &mut SendOnDrop<RendererWithCpuUsage>,
277	data: &mut [f32],
278	channels: u16,
279	sample_rate: u32,
280) {
281	renderer.on_start_processing();
282	renderer.process(data, channels, sample_rate);
