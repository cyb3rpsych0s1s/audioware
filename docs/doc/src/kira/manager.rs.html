1//! The main entrypoint for controlling audio from gameplay code.
2//!
3//! In order to play audio, you'll need to create an [`AudioManager`].
4//! The [`AudioManager`] keeps track of playing sounds and manages other
5//! resources like clocks, mixer tracks, and spatial scenes. Once the
6//! [`AudioManager`] is dropped, its audio output will be stopped.

30/// Controls audio from gameplay code.
31pub struct AudioManager<B: Backend = DefaultBackend> {
32	backend: B,
33	resource_controllers: ResourceControllers,
34	renderer_shared: Arc<RendererShared>,
35	internal_buffer_size: usize,
36}
37
38impl<B: Backend> AudioManager<B> {
39	/**
40	Creates a new [`AudioManager`].
41
42	# Examples
43
44	Create an [`AudioManager`] using the [`DefaultBackend`] and the
45	default settings:
46
47	```no_run
48	use kira::{AudioManager, AudioManagerSettings, DefaultBackend};
49
50	let audio_manager = AudioManager::<DefaultBackend>::new(AudioManagerSettings::default())?;
51	# Result::<(), Box<dyn std::error::Error>>::Ok(())
52	```
53
54	Create an [`AudioManager`] with a reverb effect on the main mixer track:
55
56	```no_run
57	use kira::{
58		AudioManager, AudioManagerSettings, DefaultBackend,
59		track::MainTrackBuilder,
60		effect::reverb::ReverbBuilder,
61	};
62
63	let audio_manager = AudioManager::<DefaultBackend>::new(AudioManagerSettings {
64		main_track_builder: MainTrackBuilder::new().with_effect(ReverbBuilder::new()),
65		..Default::default()
66	})?;
67	# Result::<(), Box<dyn std::error::Error>>::Ok(())
68	```
69	*/
70	pub fn new(settings: AudioManagerSettings<B>) -> Result<Self, B::Error> {
71		let (mut backend, sample_rate) =
72			B::setup(settings.backend_settings, settings.internal_buffer_size)?;
73		let renderer_shared = Arc::new(RendererShared::new(sample_rate));
74		let (resources, resource_controllers) = create_resources(
75			settings.capacities,
76			settings.main_track_builder,
77			sample_rate,
78			settings.internal_buffer_size,
79		);
80		let renderer = Renderer::new(
81			renderer_shared.clone(),
82			settings.internal_buffer_size,
83			resources,
84		);
85		backend.start(renderer)?;
86		Ok(Self {
87			backend,
88			resource_controllers,
89			renderer_shared,
90			internal_buffer_size: settings.internal_buffer_size,
91		})
92	}

94	/**
95	Plays a sound.
96
97	# Examples
98
99	```no_run
100	# use kira::{AudioManager, AudioManagerSettings, DefaultBackend};
101	use kira::sound::static_sound::{StaticSoundData, StaticSoundSettings};
102
103	# let mut manager = AudioManager::<DefaultBackend>::new(AudioManagerSettings::default())?;
104	let sound_data = StaticSoundData::from_file("sound.ogg")?;
105	manager.play(sound_data)?;
106	# Result::<(), Box<dyn std::error::Error>>::Ok(())
107	```
108	*/
109	pub fn play<D: SoundData>(
110		&mut self,
111		sound_data: D,
112	) -> Result<D::Handle, PlaySoundError<D::Error>> {
113		self.main_track().play(sound_data)
114	}

116	/// Creates a mixer sub-track.
117	pub fn add_sub_track(
118		&mut self,
119		builder: TrackBuilder,
120	) -> Result<TrackHandle, ResourceLimitReached> {
121		let (mut track, handle) =
122			builder.build(self.renderer_shared.clone(), self.internal_buffer_size);
123		track.init_effects(self.renderer_shared.sample_rate.load(Ordering::SeqCst));
124		self.resource_controllers
125			.sub_track_controller
126			.insert(track)?;
127		Ok(handle)
128	}
129
130	/// Adds a spatial mixer sub-track.
131	pub fn add_spatial_sub_track(
132		&mut self,
133		listener: impl Into<ListenerId>,
134		position: impl Into<Value<mint::Vector3<f32>>>,
135		builder: SpatialTrackBuilder,
136	) -> Result<SpatialTrackHandle, ResourceLimitReached> {
137		let (mut track, handle) = builder.build(
138			self.renderer_shared.clone(),
139			self.internal_buffer_size,
140			listener.into(),
141			position.into().to_(),
142		);
143		track.init_effects(self.renderer_shared.sample_rate.load(Ordering::SeqCst));
144		self.resource_controllers
145			.sub_track_controller
146			.insert(track)?;
147		Ok(handle)
148	}

150	/// Creates a mixer send track.
151	pub fn add_send_track(
152		&mut self,
153		builder: SendTrackBuilder,
154	) -> Result<SendTrackHandle, ResourceLimitReached> {
155		let key = self
156			.resource_controllers
157			.send_track_controller
158			.try_reserve()?;
159		let id = SendTrackId(key);
160		let (mut track, handle) = builder.build(id, self.internal_buffer_size);
161		track.init_effects(self.renderer_shared.sample_rate.load(Ordering::SeqCst));
162		self.resource_controllers
163			.send_track_controller
164			.insert_with_key(key, track);
165		Ok(handle)
166	}

168	/**
169	Creates a clock.
170
171	# Examples
172
173	```no_run
174	# use kira::{AudioManager, AudioManagerSettings, DefaultBackend};
175	use kira::clock::ClockSpeed;
176
177	# let mut manager = AudioManager::<DefaultBackend>::new(AudioManagerSettings::default())?;
178	let clock = manager.add_clock(ClockSpeed::TicksPerMinute(120.0))?;
179	# Result::<(), Box<dyn std::error::Error>>::Ok(())
180	```
181	*/
182	pub fn add_clock(
183		&mut self,
184		speed: impl Into<Value<ClockSpeed>>,
185	) -> Result<ClockHandle, ResourceLimitReached> {
186		let key = self.resource_controllers.clock_controller.try_reserve()?;
187		let id = ClockId(key);
188		let (clock, handle) = Clock::new(speed.into(), id);
189		self.resource_controllers
190			.clock_controller
191			.insert_with_key(key, clock);
192		Ok(handle)
193	}

225	/**
226	Creates a listener.
227
228	# Examples
229
230	```no_run
231	# use kira::{AudioManager, AudioManagerSettings, DefaultBackend};
232	# let mut manager = AudioManager::<DefaultBackend>::new(AudioManagerSettings::default())?;
233	// This example uses `glam`, but you can use any math library that has interoperability
234	// with `mint`.
235	let listener = manager.add_listener(glam::vec3(0.0, 0.0, 0.0), glam::Quat::IDENTITY)?;
236	# Result::<(), Box<dyn std::error::Error>>::Ok(())
237	```
238	*/
239	pub fn add_listener(
240		&mut self,
241		position: impl Into<Value<mint::Vector3<f32>>>,
242		orientation: impl Into<Value<mint::Quaternion<f32>>>,
243	) -> Result<ListenerHandle, ResourceLimitReached> {
244		let key = self
245			.resource_controllers
246			.listener_controller
247			.try_reserve()?;
248		let id = ListenerId(key);
249		let (listener, handle) = Listener::new(id, position.into().to_(), orientation.into().to_());
250		self.resource_controllers
251			.listener_controller
252			.insert_with_key(key, listener);
253		Ok(handle)
254	}

256	/**
257	Returns a handle to the main mixer track.
258
259	# Examples
260
261	Use the main track handle to adjust the volume of all audio:
262
263	```no_run
264	# use kira::{AudioManager, AudioManagerSettings, DefaultBackend};
265	use kira::Tween;
266
267	# let mut manager = AudioManager::<DefaultBackend>::new(AudioManagerSettings::default())?;
268	manager.main_track().set_volume(-6.0, Tween::default());
269	# Result::<(), Box<dyn std::error::Error>>::Ok(())
270	```
271	*/
272	#[must_use]
273	pub fn main_track(&mut self) -> &mut MainTrackHandle {
274		&mut self.resource_controllers.main_track_handle
275	}
