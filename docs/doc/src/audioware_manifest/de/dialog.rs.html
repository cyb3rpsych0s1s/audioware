10#[derive(Debug, Deserialize)]
11#[serde(untagged)]
12// variants declaration order matters: see https://github.com/cyb3rpsych0s1s/audioware/pull/39
13pub enum SceneDialogs {
14    SingleInline {
15        #[serde(flatten)]
16        dialogs: HashMap<Locale, PathBuf>,
17        usage: Option<Usage>,
18        settings: Option<Settings>,
19    },
20    DualInline {
21        #[serde(flatten)]
22        dialogs: HashMap<Locale, GenderBased<PathBuf>>,
23        usage: Option<Usage>,
24        settings: Option<Settings>,
25    },
26}
27
28pub type AnySceneDialog =
29    Either<(HashMap<Locale, Audio>, Usage), (HashMap<Locale, GenderBased<Audio>>, Usage)>;
30
31impl From<SceneDialogs> for AnySceneDialog {
32    fn from(value: SceneDialogs) -> Self {
33        let default_usage = Usage::OnDemand;
34        match value {
35            SceneDialogs::SingleInline {
36                dialogs,
37                usage,
38                settings,
39            } => {
40                let dialogs = paths_into_audios(dialogs, settings);
41                Either::Left((dialogs, usage.unwrap_or(default_usage)))
42            }
43            SceneDialogs::DualInline {
44                dialogs,
45                usage,
46                settings,
47            } => {
48                let dialogs: HashMap<Locale, GenderBased<Audio>> = dialogs
49                    .into_iter()
50                    .map(|(k, v)| {
51                        (
52                            k,
53                            GenderBased::<Audio> {
54                                female: Audio {
55                                    file: v.female,
56                                    settings: settings.clone(),
57                                },
58                                male: Audio {
59                                    file: v.male,
60                                    settings: settings.clone(),
61                                },
62                            },
63                        )
64                    })
65                    .collect();
66                Either::Right((dialogs, usage.unwrap_or(default_usage)))
67            }
68        }
69    }
70}
