13#[serde(untagged)]
14// variants declaration order matters: see https://github.com/cyb3rpsych0s1s/audioware/pull/39
15pub enum Voice {
16    SingleMulti {
17        #[serde(flatten)]
18        dialogs: HashMap<Locale, Dialog>,
19        usage: Option<Usage>,
20        line: Option<ScnDialogLineType>,
21        settings: Option<Settings>,
22    },
23    SingleInline {
24        #[serde(flatten)]
25        dialogs: HashMap<Locale, PathBuf>,
26        usage: Option<Usage>,
27        settings: Option<Settings>,
28    },
29    DualMulti {
30        #[serde(flatten)]
31        dialogs: HashMap<Locale, Dialogs>,
32        usage: Option<Usage>,
33        line: Option<ScnDialogLineType>,
34        settings: Option<Settings>,
35    },
36    DualInline {
37        #[serde(flatten)]
38        dialogs: HashMap<Locale, GenderBased<PathBuf>>,
39        usage: Option<Usage>,
40        settings: Option<Settings>,
41    },
42}

44#[derive(Debug, Deserialize)]
45pub struct Dialog {
46    #[serde(flatten)]
47    pub basic: Audio,
48    pub subtitle: String,
49}
50
51impl From<&Dialog> for Audio {
52    fn from(value: &Dialog) -> Self {
53        value.basic.clone()
54    }
55}
56
57impl From<(&Dialog, Option<&Settings>)> for Audio {
58    fn from(value: (&Dialog, Option<&Settings>)) -> Self {
59        let mut audio: Audio = value.into();
60        if let Some(settings) = value.1 {
61            audio.merge_settings(settings.clone());
62        }
63        audio
64    }
65}

67#[derive(Debug, Deserialize)]
68#[serde(untagged)]
69#[allow(clippy::large_enum_variant)]
70pub enum Dialogs {
71    Different {
72        #[serde(flatten)]
73        dialogs: GenderBased<Dialog>,
74    },
75    Shared {
76        #[serde(flatten)]
77        paths: GenderBased<PathBuf>,
78        subtitle: String,
79    },
80}

82/// ultimately a voice is just either file path for each locale
83/// with optional gender,
84/// optional corresponding dialog lines,
85/// audio usage
86/// and optional settings
87pub type AnyVoice = Either<
88    (
89        HashMap<Locale, Audio>,
90        Usage,
91        Option<HashMap<Locale, DialogLine>>,
92    ),
93    (
94        HashMap<Locale, GenderBased<Audio>>,
95        Usage,
96        Option<HashMap<Locale, GenderBased<DialogLine>>>,
97    ),
98>;
99
100impl From<Voice> for AnyVoice {
101    fn from(value: Voice) -> Self {
102        let default_usage = Usage::OnDemand;
103        let default_line = ScnDialogLineType::Regular;
104        match value {
105            Voice::SingleInline {
106                dialogs,
107                usage,
108                settings,
109            } => {
110                let dialogs = paths_into_audios(dialogs, settings);
111                Either::Left((dialogs, usage.unwrap_or(default_usage), None))
112            }
113            Voice::SingleMulti {
114                dialogs,
115                usage,
116                line,
117                settings,
118            } => {
119                let mut aud: HashMap<Locale, Audio> = HashMap::with_capacity(dialogs.len());
120                let mut sub: HashMap<Locale, DialogLine> = HashMap::with_capacity(dialogs.len());
121                for (k, ref v) in dialogs.into_iter() {
122                    let audio: Audio = (v, settings.as_ref()).into();
123                    aud.insert(k, audio);
124                    sub.insert(
125                        k,
126                        DialogLine {
127                            msg: v.subtitle.clone(),
128                            line: line.unwrap_or(default_line),
129                        },
130                    );
131                }
132                Either::Left((aud, usage.unwrap_or(default_usage), Some(sub)))
133            }
134            Voice::DualInline {
135                dialogs,
136                usage,
137                settings,
138            } => {
139                let dialogs: HashMap<Locale, GenderBased<Audio>> = dialogs
140                    .into_iter()
141                    .map(|(k, v)| {
142                        (
143                            k,
144                            GenderBased::<Audio> {
145                                female: Audio {
146                                    file: v.female,
147                                    settings: settings.clone(),
148                                },
149                                male: Audio {
150                                    file: v.male,
151                                    settings: settings.clone(),
152                                },
153                            },
154                        )
155                    })
156                    .collect();
157                Either::Right((dialogs, usage.unwrap_or(default_usage), None))
158            }
159            Voice::DualMulti {
160                dialogs,
161                usage,
162                line,
163                settings,
164            } => {
165                let mut aud: HashMap<Locale, GenderBased<Audio>> =
166                    HashMap::with_capacity(dialogs.len());
167                let mut sub: HashMap<Locale, GenderBased<DialogLine>> =
168                    HashMap::with_capacity(dialogs.len());
169                for (k, v) in dialogs.into_iter() {
170                    match v {
171                        super::Dialogs::Different { dialogs } => {
172                            let mut female = dialogs.female.basic.clone();
173                            let mut male = dialogs.male.basic.clone();
174                            if let Some(ref settings) = settings {
175                                female.merge_settings(settings.clone());
176                                male.merge_settings(settings.clone());
177                            }
178                            let aud_dialogs = GenderBased { female, male };
179                            let aud_subs = GenderBased {
180                                female: DialogLine {
181                                    msg: dialogs.female.subtitle.clone(),
182                                    line: line.unwrap_or(ScnDialogLineType::Regular),
183                                },
184                                male: DialogLine {
185                                    msg: dialogs.male.subtitle.clone(),
186                                    line: line.unwrap_or(ScnDialogLineType::Regular),
187                                },
188                            };
189                            aud.insert(k, aud_dialogs);
190                            sub.insert(k, aud_subs);
191                        }
192                        super::Dialogs::Shared { paths, subtitle } => {
193                            let (fem, male) = (paths.female, paths.male);
194                            aud.insert(
195                                k,
196                                GenderBased {
197                                    female: Audio {
198                                        file: fem.clone(),
199                                        settings: settings.clone(),
200                                    },
201                                    male: Audio {
202                                        file: male.clone(),
203                                        settings: settings.clone(),
204                                    },
205                                },
206                            );
207                            let same = DialogLine {
208                                msg: subtitle.clone(),
209                                line: line.unwrap_or(default_line),
210                            };
211                            sub.insert(
212                                k,
213                                GenderBased {
214                                    female: same.clone(),
215                                    male: same,
216                                },
217                            );
218                        }
219                    }
220                }
221                Either::Right((aud, usage.unwrap_or(default_usage), Some(sub)))
222            }
223        }
224    }
225}
