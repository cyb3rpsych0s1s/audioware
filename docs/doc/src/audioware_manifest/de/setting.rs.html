24/// Deserialization type
25/// for [kira::sound::static_sound::StaticSoundSettings]
26/// and [kira::sound::streaming::StreamingSoundSettings].
27#[derive(Debug, Default, Deserialize, Clone)]
28pub struct Settings {
29    #[serde(with = "humantime_serde", default)]
30    pub start_time: Option<Duration>,
31    #[serde(with = "humantime_serde", default)]
32    pub start_position: Option<Duration>,
33    pub volume: Option<Amplitude>,
34    pub panning: Option<f32>,
35    #[serde(rename = "loop")]
36    pub r#loop: Option<bool>,
37    pub region: Option<self::Region>,
38    #[serde(deserialize_with = "factor_or_semitones", default)]
39    pub playback_rate: Option<PlaybackRate>,
40    pub fade_in_tween: Option<Interpolation>,
41    pub affected_by_time_dilation: Option<bool>,
42}

44macro_rules! impl_with {
45    ($self:expr, $settings:expr) => {{
46        if let Some(x) = $settings.start_time.map(StartTime::Delayed) {
47            $self = $self.start_time(x);
48        }
49        if let Some(x) = $settings
50            .start_position
51            .as_ref()
52            .map(Duration::as_secs_f64)
53            .map(PlaybackPosition::Seconds)
54        {
55            $self = $self.start_position(x);
56        }
57        if let Some(x) = $settings.volume {
58            $self = $self.volume(x.as_decibels());
59        }
60        if let Some(x) = $settings.panning {
61            $self = $self.panning(x);
62        }
63        if $settings.region.is_some() || $settings.r#loop.unwrap_or(false) {
64            if let Some(x) = $settings.region {
65                if $settings.r#loop.unwrap_or(false) {
66                    $self = $self.loop_region(x);
67                } else {
68                    $self = $self.slice(x);
69                }
70            } else {
71                $self = $self.loop_region(..);
72            }
73        }
74        if let Some(x) = $settings.playback_rate {
75            $self = $self.playback_rate(x);
76        }
77        if let Some(x) = $settings.fade_in_tween.map(Into::<Tween>::into) {
78            $self = $self.fade_in_tween(x);
79        }
80        $self
81    }};
82}

147impl IntoOptionalRegion for self::Region {
148    fn into_optional_region(self) -> Option<kira::sound::Region> {
149        if self.starts.is_none() && self.ends.is_none() {
150            return None;
151        }
152        Some(kira::sound::Region {
153            start: PlaybackPosition::Seconds(
154                self.starts
155                    .as_ref()
156                    .map(Duration::as_secs_f64)
157                    .unwrap_or(0.),
158            ),
159            end: self
160                .ends
161                .as_ref()
162                .map(Duration::as_secs_f64)
163                .map(PlaybackPosition::Seconds)
164                .map(EndPosition::Custom)
165                .unwrap_or(EndPosition::EndOfAudio),
166        })
167    }

170fn factor_or_semitones<'de, D>(deserializer: D) -> Result<Option<PlaybackRate>, D::Error>
171where
172    D: serde::Deserializer<'de>,
173{
174    let s: Option<&str> = Deserialize::deserialize(deserializer)?;
175    if let Some(s) = s {
176        let s = s.trim();
177        if s.starts_with('x') || s.starts_with('X') {
178            return Ok(Some(PlaybackRate(
179                s[1..].trim().parse().map_err(serde::de::Error::custom)?,
180            )));
181        }
182        if s.ends_with('♯') {
183            return Ok(Some(
184                Semitones(
185                    s[..(s.len() - '♯'.len_utf8())]
186                        .trim()
187                        .parse()
188                        .map_err(serde::de::Error::custom)?,
189                )
190                .into(),
191            ));
192        }
193        if s.ends_with('♭') {
194            return Ok(Some(
195                Semitones(
196                    -s[..(s.len() - '♭'.len_utf8())]
197                        .trim()
198                        .parse()
199                        .map_err(serde::de::Error::custom)?,
200                )
201                .into(),
202            ));
203        }
204        return Err(serde::de::Error::custom(format!(
205            "invalid factor or semitone: {s}"
206        )));
207    }
208    Ok(None)
209}

238macro_rules! impl_from_settings {
239    ($into:path) => {
240        #[allow(clippy::needless_update)]
241        impl From<self::Settings> for $into {
242            fn from(value: self::Settings) -> Self {
243                Self {
244                    start_time: value
245                        .start_time
246                        .map(::kira::StartTime::Delayed)
247                        .unwrap_or_default(),
248                    start_position: value
249                        .start_position
250                        .map(|x| ::kira::sound::PlaybackPosition::Seconds(x.as_secs_f64()))
251                        .unwrap_or_default(),
252                    volume: value
253                        .volume
254                        .map(|x| ::kira::Value::Fixed(x.as_decibels()))
255                        .unwrap_or_default(),
256                    panning: value
257                        .panning
258                        .map(|x| ::kira::Panning(x))
259                        .map(Into::into)
260                        .unwrap_or_default(),
261                    fade_in_tween: value.fade_in_tween.map(Into::into),
262                    loop_region: if value.r#loop.unwrap_or_default() {
263                        value
264                            .region
265                            .map(IntoOptionalRegion::into_optional_region)
266                            .unwrap_or(Some((..).into()))
267                    } else {
268                        None
269                    },
270                    playback_rate: value.playback_rate.map(Into::into).unwrap_or_default(),
271                    ..Default::default()
272                }
273            }
274        }
275    };

281impl Validate for Settings {
282    fn validate(&self) -> Result<(), Vec<ValidationError>> {
283        let mut errors: Vec<_> = vec![];
284        if let Some(panning) = self.panning
285            && !(0.0..=1.0).contains(&panning)
286        {
287            errors.push(ValidationError {
288                which: "panning",
289                why: "must be a value between 0.0 and 1.0 (inclusive)",
290            });
291        }
292        if let Some(volume) = self.volume
293            && volume.as_decibels() > Decibels(85.0)
294        {
295            errors.push(ValidationError {
296                which: "volume",
297                why: "audio should not be louder than 85.0 dB",
298            });
299        }
300        if errors.is_empty() {
301            return Ok(());
302        }
303        Err(errors)
304    }
305}

313impl ValidateFor<Either<StaticSoundData, StreamingSoundData<FromFileError>>> for Settings {
314    fn validate_for(
315        &self,
316        audio: &Either<StaticSoundData, StreamingSoundData<FromFileError>>,
317    ) -> Result<(), Vec<ValidationError>> {
318        let mut errors = vec![];
319        if let Some(ref region) = self.region {
320            let total_duration = match audio {
321                Either::Left(x) => x.unsliced_duration(),
322                Either::Right(x) => x.unsliced_duration(),
323            }
324            .as_secs_f64();
325            let start: f64 = match (region.starts(), audio) {
326                (Some(PlaybackPosition::Seconds(seconds)), _) => seconds,
327                (Some(PlaybackPosition::Samples(samples)), Either::Left(data)) => {
328                    samples as f64 / data.sample_rate as f64
329                }
330                // no sample rate method yet
331                (Some(PlaybackPosition::Samples(_)), Either::Right(_)) => {
332                    errors.push(ValidationError {
333                        which: "region.starts",
334                        why: "samples unit is not supported with streaming sound yet",
335                    });
336                    return Err(errors);
337                }
338                // none implicitly means beginning of the audio
339                (None, _) => 0.0,
340            };
341            let end: f64 = match (region.ends(), audio) {
342                (Some(EndPosition::Custom(PlaybackPosition::Seconds(x))), _) => x,
343                (
344                    Some(EndPosition::Custom(PlaybackPosition::Samples(samples))),
345                    Either::Left(data),
346                ) => samples as f64 / data.sample_rate as f64,
347                // no sample rate method yet
348                (Some(EndPosition::Custom(PlaybackPosition::Samples(_))), Either::Right(_)) => {
349                    errors.push(ValidationError { which: "region.ends", why: "samples unit is not supported with streaming sound yet" });
350                    return Err(errors);
351                }
352                (Some(EndPosition::EndOfAudio), Either::Left(_)) |
353                (Some(EndPosition::EndOfAudio), Either::Right(_)) |
354                // none implicitly means end of the audio
355                (None, _) => total_duration,
356            };
357            if start < 0.
358                || end <= 0.
359                || start >= total_duration
360                || end > total_duration
361                || start >= end
362            {
363                errors.push(ValidationError {
364                    which: "region",
365                    why: "must be within audio duration and starts before it ends",
366                });
367            }
368            if let Some(start_position) = self.start_position.map(|x| x.as_secs_f64())
369                && start_position >= end
370            {
371                errors.push(ValidationError {
372                    which: "start_position",
373                    why: "greater than audio duration",
374                });
375            }
376        } else if let Some(start_position) = self.start_position.map(|x| x.as_secs_f64()) {
377            let duration = match audio {
378                Either::Left(x) => x.duration(),
379                Either::Right(x) => x.duration(),
380            }
381            .as_secs_f64();
382            if start_position >= duration {
383                errors.push(ValidationError {
384                    which: "start_position",
385                    why: "greater than audio duration",
386                });
387            }
388        }
389        if errors.is_empty() {
390            return Ok(());
391        }
392        Err(errors)
393    }
394}
