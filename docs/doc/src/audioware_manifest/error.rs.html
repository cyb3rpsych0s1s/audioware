3use snafu::Snafu;
4
5#[derive(Debug, Snafu)]
6#[snafu(visibility(pub(crate)))]
7pub enum Error {
8    /// Binary is not located where expected.
9    #[snafu(display("unable to read binary location"))]
10    BinaryLocation { source: std::io::Error },
11    /// Missing folder.
12    #[snafu(display("unable to locate parent folder (expected '{folder}')"))]
13    NoFolder { folder: &'static str },
14    /// An error occured while reading a [Depot](crate::Depot) folder.
15    #[snafu(display("cannot read folder: {depot}"))]
16    CannotReadDepot { depot: String },
17    #[snafu(
18        display("cannot read file: {manifest}"),
19        visibility(pub),
20        context(suffix(false))
21    )]
22    /// An error occured while reading a [Manifest](crate::Manifest) file.
23    CannotReadManifest {
24        manifest: String,
25        source: std::io::Error,
26    },
27    #[snafu(
28        display("cannot parse file: {manifest}\n{source}"),
29        visibility(pub),
30        context(suffix(false))
31    )]
32    /// An error occured while parsing a [Manifest](crate::Manifest) file.
33    CannotParseManifest {
34        manifest: String,
35        source: serde_yaml::Error,
36    },
37}

39#[derive(Debug, Snafu, PartialEq)]
40pub enum ConversionError {
41    /// Cyberpunk 2077 does not support this [Locale](crate::Locale).
42    #[snafu(display("invalid locale: {value}"))]
43    InvalidLocale { value: String },
44    /// Locale type is not supported by Locale subset.
45    #[snafu(display("unsupported locale for {}: {value}", r#type))]
46    UnsupportedLocale { r#type: String, value: String },
47    /// Cyberpunk 2077 does not support this [PlayerGender](crate::PlayerGender).
48    #[snafu(display("invalid gender: {value}"))]
49    InvalidGender { value: String },
50    /// Audio stream buffer size is invalid.
51    #[snafu(display("invalid buffer size: {value}"))]
52    InvalidBufferSize { value: String },
53    /// Audio stream buffer size is missing.
54    #[snafu(display("missing buffer size"))]
55    MissingBufferSize,
56}
