88macro_rules! io_try_or_ret {
89    ($expr:expr) => {
90        match $expr {
91            Ok(val) => val,
92            // An end-of-bitstream error is classified under ErrorKind::Other. This condition
93            // should not be treated as an error, rather, it should return from the function
94            // immediately without error.
95            Err(ref e) if e.kind() == std::io::ErrorKind::Other => return Ok(()),
96            Err(e) => return Err(e.into()),
97        }
98    };
99}
100
101macro_rules! try_or_ret {
102    ($expr:expr) => {
103        match $expr {
104            Ok(val) => val,
105            // An end-of-bitstream error is classified under ErrorKind::Other. This condition
106            // should not be treated as an error, rather, it should return from the function
107            // immediately without error.
108            Err(Error::IoError(ref e)) if e.kind() == std::io::ErrorKind::Other => return Ok(()),
109            Err(e) => return Err(e),
110        }
111    };
112}

160    fn read_setup(
161        bs: &mut BitReaderRtl<'_>,
162        bs0_exp: u8,
163        bs1_exp: u8,
164        max_codebook: u8,
165    ) -> Result<Floor0Setup> {
166        let floor0_order = bs.read_bits_leq32(8)? as u8;
167        let floor0_rate = bs.read_bits_leq32(16)? as u16;
168        let floor0_bark_map_size = bs.read_bits_leq32(16)? as u16;
169        let floor0_amplitude_bits = bs.read_bits_leq32(6)? as u8;
170        let floor0_amplitude_offset = bs.read_bits_leq32(8)? as u8;
171        let floor0_number_of_books = bs.read_bits_leq32(4)? as u8 + 1;
172        let mut floor0_book_list = [0; 16];
173
174        let end = usize::from(floor0_number_of_books);
175
176        for book in &mut floor0_book_list[..end] {
177            *book = bs.read_bits_leq32(8)? as u8;
178
179            if *book >= max_codebook {
180                return decode_error("vorbis: floor0, invalid codebook number");
181            }
182        }
183
184        // Pre-compute the Bark-scale maps.
185        let floor0_map_short = bark_map(1 << (bs0_exp - 1), floor0_rate, floor0_bark_map_size);
186        let floor0_map_long = bark_map(1 << (bs1_exp - 1), floor0_rate, floor0_bark_map_size);
187
188        let floor_type0 = Floor0Setup {
189            floor0_order,
190            floor0_bark_map_size,
191            floor0_amplitude_bits,
192            floor0_amplitude_offset,
193            floor0_number_of_books,
194            floor0_book_list,
195            floor0_map_short_bs_exp: bs0_exp,
196            floor0_map_short,
197            floor0_map_long,
198        };
199
200        Ok(floor_type0)
201    }

204impl Floor for Floor0 {
205    fn read_channel(
206        &mut self,
207        bs: &mut BitReaderRtl<'_>,
208        codebooks: &[VorbisCodebook],
209    ) -> Result<()> {
210        // Assume the floor is unused until it is decoded successfully.
211        self.is_unused = true;
212
213        self.amplitude =
214            io_try_or_ret!(bs.read_bits_leq64(u32::from(self.setup.floor0_amplitude_bits)));
215
216        if self.amplitude != 0 {
217            // Read the index into the floor's codebook list that contains the actual codebook
218            // index.
219            let floor_book_idx_bits = ilog(u32::from(self.setup.floor0_number_of_books));
220            let floor_book_idx = io_try_or_ret!(bs.read_bits_leq32(floor_book_idx_bits)) as usize;
221
222            // Get the actual codebook index from the floor's codebook list.
223            let codebook_idx = self.setup.floor0_book_list[floor_book_idx] as usize;
224
225            if codebook_idx >= codebooks.len() {
226                return decode_error("vorbis: floor0, invalid codebook");
227            }
228
229            // Get the codebook for this floor.
230            let codebook = &codebooks[codebook_idx];
231
232            let order = usize::from(self.setup.floor0_order);
233            let mut i = 0;
234            let mut last = 0.0;
235
236            while i < order {
237                let i0 = i;
238
239                // Read and obtain the VQ vector from the codebook.
240                let vq = try_or_ret!(codebook.read_vq(bs));
241
242                // The VQ vector may be much larger (up-to 65535 scalars) than the remaining number
243                // of coefficients (up-to 255 scalars). Cap the amount of coefficients to be
244                // processed.
245                i += min(order - i0, vq.len());
246
247                // Add the value of last coefficient from the previous iteration to each scalar
248                // value read from the VQ vector and append those valeus to the coefficient vector.
249                for (c, &vq) in self.coeffs[i0..i].iter_mut().zip(vq) {
250                    *c = last + vq;
251                }
252
253                // Store the value of the last coefficient in the coefficient vector for the next
254                // iteration.
255                last = self.coeffs[i - 1];
256            }
257
258            // Pre-compute the 2 times the cosine of all coefficients.
259            for coeff in self.coeffs[..order].iter_mut() {
260                *coeff = 2.0 * coeff.cos();
261            }
262        }
263
264        // The floor is used if the amplitude is not 0.
265        self.is_unused = self.amplitude == 0;
266
267        Ok(())
268    }
269
270    fn is_unused(&self) -> bool {
271        self.is_unused
272    }
273
274    fn synthesis(&mut self, bs_exp: u8, floor: &mut [f32]) -> Result<()> {
275        debug_assert!(!self.is_unused);
276
277        // Block size.
278        let n = (1 << bs_exp) >> 1;
279
280        // Select the correct Bark-scale map based on the block-size exponent.
281        let map = if bs_exp == self.setup.floor0_map_short_bs_exp {
282            &self.setup.floor0_map_short
283        }
284        else {
285            &self.setup.floor0_map_long
286        };
287
288        let omega_step = std::f32::consts::PI / f32::from(self.setup.floor0_bark_map_size);
289
290        let mut i = 0;
291
292        loop {
293            let iter_cond = map[i];
294
295            let omega = omega_step * iter_cond as f32;
296            let cos_omega = omega.cos();
297            let two_cos_omega = 2.0 * cos_omega;
298
299            let mut p = 1.0;
300            let mut q = 1.0;
301
302            let mut iter = self.coeffs[..usize::from(self.setup.floor0_order)].chunks_exact(2);
303
304            // Calculate p using coefficients with odd indicies, and q using coefficients with even
305            // indicies.
306            for coeffs in &mut iter {
307                p *= coeffs[1] - two_cos_omega;
308                q *= coeffs[0] - two_cos_omega;
309            }
310
311            // If order is an odd number, then there should be exactly one extra coefficient.
312            let last_coeff = iter.remainder();
313
314            if !last_coeff.is_empty() {
315                q *= last_coeff[0] - two_cos_omega;
316
317                p = p * p * (1.0 - (cos_omega * cos_omega));
318                q = q * q * 0.25;
319            }
320            else {
321                p = p * p * ((1.0 - cos_omega) / 2.0);
322                q = q * q * ((1.0 + cos_omega) / 2.0);
323            }
324
325            if p + q == 0.0 {
326                return decode_error("vorbis: invalid floor0 coefficients");
327            }
328
329            let linear_floor_value = linear_floor0_value(
330                p,
331                q,
332                self.amplitude,
333                self.setup.floor0_amplitude_bits,
334                self.setup.floor0_amplitude_offset,
335            );
336
337            // Fill in the floor values where the map value is the same.
338            for (floor, &map) in floor[i..n].iter_mut().zip(&map[i..n]) {
339                if map != iter_cond {
340                    break;
341                }
342
343                *floor = linear_floor_value;
344                i += 1;
345            }
346
347            if i >= n {
348                break;
349            }
350        }
351
352        Ok(())
353    }
354}

356/// Vorbis I specification, section 6.2.3.
357#[inline(always)]
358fn bark(x: f64) -> f64 {
359    (13.1 * (0.00074 * x).atan()) + (2.24 * (0.0000000185 * x * x).atan()) + (0.0001 * x)
360}
361
362fn bark_map(n: u32, floor0_rate: u16, floor0_bark_map_size: u16) -> Vec<i32> {
363    let mut map = Vec::with_capacity(n as usize);
364
365    let foobar_min = i32::from(floor0_bark_map_size) - 1;
366    let rate = f64::from(floor0_rate);
367    let rate_by_2n = rate / (2.0 * f64::from(n));
368
369    let c = f64::from(floor0_bark_map_size) / bark(0.5 * rate);
370
371    // Compute 0 to N-1 elements.
372    for i in 0..n {
373        let foobar = (bark(rate_by_2n * f64::from(i)) * c).floor() as i32;
374        map.push(foobar.min(foobar_min));
375    }
376
377    map
378}

380/// Calculate the linear floor value as per Vorbis I specification, section 6.2.3.
381#[inline(always)]
382fn linear_floor0_value(
383    p: f32,
384    q: f32,
385    amplitude: u64,
386    amplitude_bits: u8,
387    amplitude_offset: u8,
388) -> f32 {
389    // Amplitude could be up-to 63-bits, and amplitude offset 8-bits. Therefore, 71-bits is required
390    // in total. This is unreasonable since even an f64 can't represent more than 56-bits. Since
391    // large values like this shouldn't happen in practice, use wrapping arithmetic to prevent
392    // panics and truncate to 32-bits in the f32 conversion.
393    let a = amplitude.wrapping_mul(u64::from(amplitude_offset)) as f32;
394    let b = (p + q).sqrt() * ((1u64 << amplitude_bits) - 1) as f32;
395
396    (0.11512925 * ((a / b) - f32::from(amplitude_offset))).exp()
397}

440impl Floor1 {
441    pub fn try_read(bs: &mut BitReaderRtl<'_>, max_codebook: u8) -> Result<Box<dyn Floor>> {
442        let setup = Self::read_setup(bs, max_codebook)?;
443
444        let x_list_len = setup.floor1_x_list.len();
445
446        Ok(Box::new(Floor1 {
447            setup,
448            is_unused: false,

454
455    fn read_setup(bs: &mut BitReaderRtl<'_>, max_codebook: u8) -> Result<Floor1Setup> {
456        // The number of partitions. 5-bit value, 0..31 range.
457        let floor1_partitions = bs.read_bits_leq32(5)? as usize;
458
459        // Parition list of up-to 32 partitions (floor1_partitions), with each partition indicating
460        // a 4-bit class (0..16) identifier.
461        let mut floor1_partition_class_list = [0; 32];
462
463        let mut floor1_classes: [Floor1Class; 16] = Default::default();
464
465        if floor1_partitions > 0 {
466            let mut max_class = 0; // 4-bits, 0..15
467
468            // Read the partition class list.
469            for class_idx in &mut floor1_partition_class_list[..floor1_partitions] {
470                *class_idx = bs.read_bits_leq32(4)? as u8;
471
472                // Find the maximum class value.
473                max_class = max_class.max(*class_idx);
474            }
475
476            let num_classes = usize::from(1 + max_class);
477
478            for class in floor1_classes[..num_classes].iter_mut() {
479                class.dimensions = bs.read_bits_leq32(3)? as u8 + 1;
480                class.subclass_bits = bs.read_bits_leq32(2)? as u8;
481
482                if class.subclass_bits != 0 {
483                    let masterbook = bs.read_bits_leq32(8)? as u8;
484
485                    if masterbook >= max_codebook {
486                        return decode_error("vorbis: floor1, invalid codebook for class");
487                    }
488
489                    class.mainbook = masterbook;
490                }
491
492                let num_subclasses = 1 << class.subclass_bits;
493
494                for (i, book) in class.subbooks[..num_subclasses].iter_mut().enumerate() {
495                    // Read the codebook number.
496                    *book = bs.read_bits_leq32(8)? as u8;
497
498                    // A codebook number > 0 indicates a codebook is used.
499                    if *book > 0 {
500                        // The actual codebook used is the number read minus one.
501                        *book -= 1;
502
503                        if *book >= max_codebook {
504                            return decode_error("vorbis: floor1, invalid codebook for subclass");
505                        }
506
507                        class.is_subbook_used |= 1 << i;
508                    }
509                }
510            }
511        }
512
513        let floor1_multiplier = bs.read_bits_leq32(2)? as u8 + 1;
514
515        let rangebits = bs.read_bits_leq32(4)?;
516
517        let mut floor1_x_list = Vec::new();
518        let mut floor1_x_list_unique = HashSet::new();
519
520        floor1_x_list.push(0);
521        floor1_x_list.push(1 << rangebits);
522
523        for &class_idx in &floor1_partition_class_list[..floor1_partitions] {
524            let class = &floor1_classes[class_idx as usize];
525
526            // No more than 65 elements are allowed.
527            if floor1_x_list.len() + usize::from(class.dimensions) > 65 {
528                return decode_error("vorbis: floor1, x_list too long");
529            }
530
531            for _ in 0..class.dimensions {
532                let x = bs.read_bits_leq32(rangebits)?;
533
534                // All elements in the x list must be unique.
535                if !floor1_x_list_unique.insert(x) {
536                    return decode_error("vorbis: floor1, x_list is not unique");
537                }
538
539                floor1_x_list.push(x);
540            }
541        }
542
543        let mut floor1_x_list_neighbors = Vec::with_capacity(floor1_x_list.len());
544        let mut floor1_x_list_sort_order = Vec::with_capacity(floor1_x_list.len());
545
546        // Precompute neighbors.
547        for i in 0..floor1_x_list.len() {
548            floor1_x_list_neighbors.push(find_neighbors(&floor1_x_list, i));
549            floor1_x_list_sort_order.push(i as u8);
550        }
551
552        // Precompute sort-order.
553        floor1_x_list_sort_order.sort_by_key(|&i| floor1_x_list[i as usize]);
554
555        let floor_type1 = Floor1Setup {
556            floor1_partitions,
557            floor1_partition_class_list,
558            floor1_classes,
559            floor1_multiplier,
560            floor1_x_list,
561            floor1_x_list_neighbors,
562            floor1_x_list_sort_order,
563        };
564
565        Ok(floor_type1)
566    }
567
568    fn synthesis_step1(&mut self) {
569        // Step 1.
570        let range = get_range(self.setup.floor1_multiplier);
571
572        self.floor_step2_flag[0] = true;
573        self.floor_step2_flag[1] = true;
574
575        self.floor_final_y[0] = self.floor_y[0] as i32;
576        self.floor_final_y[1] = self.floor_y[1] as i32;
577
578        for i in 2..self.setup.floor1_x_list.len() {
579            // Find the neighbours.
580            let (low_neighbor_offset, high_neighbor_offset) = self.setup.floor1_x_list_neighbors[i];
581
582            let predicted = render_point(
583                self.setup.floor1_x_list[low_neighbor_offset],
584                self.floor_final_y[low_neighbor_offset],
585                self.setup.floor1_x_list[high_neighbor_offset],
586                self.floor_final_y[high_neighbor_offset],
587                self.setup.floor1_x_list[i],
588            );
589
590            let val = self.floor_y[i] as i32;
591            let highroom = range as i32 - predicted;
592            let lowroom = predicted;
593
594            if val != 0 {
595                let room = 2 * if highroom < lowroom { highroom } else { lowroom };
596
597                self.floor_step2_flag[low_neighbor_offset] = true;
598                self.floor_step2_flag[high_neighbor_offset] = true;
599                self.floor_step2_flag[i] = true;
600
601                self.floor_final_y[i] = if val >= room {
602                    if highroom > lowroom {
603                        val - lowroom + predicted
604                    }
605                    else {
606                        predicted - val + highroom - 1
607                    }
608                }
609                else {
610                    // If val is odd.
611                    if val & 1 == 1 {
612                        predicted - ((val + 1) / 2)
613                    }
614                    else {
615                        // If val is even.
616                        predicted + (val / 2)
617                    }
618                }
619            }
620            else {
621                self.floor_step2_flag[i] = false;
622                self.floor_final_y[i] = predicted;
623            }
624        }
625    }
626
627    fn synthesis_step2(&mut self, n: u32, floor: &mut [f32]) {
628        let multiplier = self.setup.floor1_multiplier as i32;
629
630        let floor_final_y0 = self.floor_final_y[self.setup.floor1_x_list_sort_order[0] as usize];
631
632        let mut hx = 0;
633        let mut hy = 0;
634        let mut lx = 0;
635        let mut ly = floor_final_y0 * multiplier;
636
637        // Iterate in sort-order.
638        for i in self.setup.floor1_x_list_sort_order[1..].iter().map(|i| *i as usize) {
639            if self.floor_step2_flag[i] {
640                hy = self.floor_final_y[i] * multiplier;
641                hx = self.setup.floor1_x_list[i];
642
643                render_line(lx, ly, hx, hy, n as usize, floor);
644
645                lx = hx;
646                ly = hy;
647            }
648        }
649
650        if hx < n {
651            render_line(hx, hy, n, hy, n as usize, floor);
652        }
653    }
654}
655
656impl Floor for Floor1 {
657    fn read_channel(
658        &mut self,
659        bs: &mut BitReaderRtl<'_>,
660        codebooks: &[VorbisCodebook],
661    ) -> Result<()> {
662        // Assume the floor is unused until it is decoded successfully.
663        self.is_unused = true;
664
665        // First bit marks if this floor is used. Exit early if it is not.
666        let is_used = io_try_or_ret!(bs.read_bool());
667
668        if !is_used {
669            return Ok(());
670        }
671
672        // Section 7.3.2
673        let range = get_range(self.setup.floor1_multiplier);
674
675        // The number of bits required to represent range.
676        let range_bits = ilog(range - 1);
677
678        self.floor_y[0] = io_try_or_ret!(bs.read_bits_leq32(range_bits));
679        self.floor_y[1] = io_try_or_ret!(bs.read_bits_leq32(range_bits));
680
681        let mut offset = 2;
682
683        for &class_idx in &self.setup.floor1_partition_class_list[0..self.setup.floor1_partitions] {
684            // The class.
685            let class = &self.setup.floor1_classes[class_idx as usize];
686
687            let cdim = class.dimensions as usize;
688            let cbits = class.subclass_bits;
689            let csub = (1 << cbits) - 1;
690
691            let mut cval = 0;
692
693            if cbits > 0 {
694                let mainbook_idx = class.mainbook as usize;
695                cval = try_or_ret!(codebooks[mainbook_idx].read_scalar(bs));
696            }
697
698            for floor_y in self.floor_y[offset..offset + cdim].iter_mut() {
699                let subclass_idx = cval & csub;
700
701                // Is the sub-book used for this sub-class.
702                let is_subbook_used = class.is_subbook_used & (1 << subclass_idx) != 0;
703
704                cval >>= cbits;
705
706                *floor_y = if is_subbook_used {
707                    let subbook_idx = class.subbooks[subclass_idx as usize] as usize;
708                    try_or_ret!(codebooks[subbook_idx].read_scalar(bs))
709                }
710                else {
711                    0
712                };
713            }
714
715            offset += cdim;
716        }
717
718        // If this point is reached then the floor is used.
719        self.is_unused = false;
720
721        Ok(())
722    }
723
724    fn is_unused(&self) -> bool {
725        self.is_unused
726    }
727
728    fn synthesis(&mut self, bs_exp: u8, floor: &mut [f32]) -> Result<()> {
729        debug_assert!(!self.is_unused);
730        self.synthesis_step1();
731        self.synthesis_step2((1 << bs_exp) >> 1, floor);
732        Ok(())
733    }
734}

736#[inline(always)]
737fn get_range(multiplier: u8) -> u32 {
738    match multiplier - 1 {
739        0 => 256,
740        1 => 128,
741        2 => 86,
742        3 => 64,
743        _ => unreachable!(),
744    }
745}

747#[inline(always)]
748fn find_neighbors(vec: &[u32], x: usize) -> (usize, usize) {
749    let bound = vec[x];
750
751    let mut low = u32::MIN; // TODO: Should be -1?
752    let mut high = u32::MAX;
753
754    let mut res: (usize, usize) = (0, 0);
755
756    // Sections 9.2.4 and 9.2.5
757    for (i, &xv) in vec[..x].iter().enumerate() {
758        // low_neighbor(v,x) finds the position n in vector [v] of the greatest value scalar element
759        // for which n is less than x and vector [v] element n is less than vector [v] element [x].
760        if xv > low && xv < bound {
761            low = xv;
762            res.0 = i;
763        }
764        // high_neighbor(v,x) finds the position n in vector [v] of the lowest value scalar element
765        // for which n is less than x and vector [v] element n is greater than vector [v] element [x].
766        if xv < high && xv > bound {
767            high = xv;
768            res.1 = i;
769        }
770    }
771
772    res
773}

775#[inline(always)]
776fn render_point(x0: u32, y0: i32, x1: u32, y1: i32, x: u32) -> i32 {
777    let dy = y1 - y0;
778    let adx = x1 - x0;
779    let err = dy.unsigned_abs() * (x - x0);
780    let off = err / adx;
781    if dy < 0 {
782        y0 - off as i32
783    }
784    else {
785        y0 + off as i32
786    }
787}

789#[inline(always)]
790fn render_line(x0: u32, y0: i32, x1: u32, y1: i32, n: usize, v: &mut [f32]) {
791    let dy = y1 - y0;
792    let adx = (x1 - x0) as i32;
793
794    let base = dy / adx;
795
796    let mut y = y0;
797
798    let sy = if dy < 0 { base - 1 } else { base + 1 };
799
800    let ady = dy.abs() - base.abs() * adx;
801
802    v[x0 as usize] = FLOOR1_INVERSE_DB_TABLE[y as usize];
803
804    let mut err = 0;
805
806    let x_begin = x0 as usize + 1;
807    let x_end = min(n, x1 as usize);
808
809    for v in v[x_begin..x_end].iter_mut() {
810        err += ady;
811
812        y += if err >= adx {
813            err -= adx;
814            sy
815        }
816        else {
817            base
818        };
819
820        *v = FLOOR1_INVERSE_DB_TABLE[y as usize];
821    }
