87    /// Reads a page.
88    pub fn read_page(&mut self, page: &Page<'_>) -> Result<Vec<SideData>> {
89        // Side data vector. This will not allocate unless data is pushed to it (normal case).
90        let mut side_data = Vec::new();
91
92        // If the last sequence number is available, detect non-monotonicity and discontinuities
93        // in the stream. In these cases, clear any partial packet data.
94        if let Some(last_ts) = &self.prev_page_info {
95            if page.header.sequence < last_ts.seq {
96                warn!("detected stream page non-monotonicity");
97                self.part_len = 0;
98            }
99            else if page.header.sequence - last_ts.seq > 1 {
100                warn!(
101                    "detected stream discontinuity of {} page(s)",
102                    page.header.sequence - last_ts.seq
103                );
104                self.part_len = 0;
105            }
106        }
107
108        self.prev_page_info =
109            Some(PageInfo { seq: page.header.sequence, absgp: page.header.absgp });
110
111        let mut iter = page.packets();
112
113        // If there is partial packet data buffered, a continuation page is expected.
114        if !page.header.is_continuation && self.part_len > 0 {
115            warn!("expected a continuation page");
116
117            // Clear partial packet data.
118            self.part_len = 0;
119        }
120
121        // If there is no partial packet data buffered, a continuation page is not expected.
122        if page.header.is_continuation && self.part_len == 0 {
123            // If the continuation page contains packets, drop the first packet since it would
124            // require partial packet data to be complete. Otherwise, ignore this page entirely.
125            if page.num_packets() > 0 {
126                warn!("unexpected continuation page, ignoring incomplete first packet");
127                iter.next();
128            }
129            else {
130                warn!("unexpected continuation page, ignoring page");
131                return Ok(side_data);
132            }
133        }
134
135        let num_prev_packets = self.packets.len();
136
137        for buf in &mut iter {
138            // Get a packet with data from the partial packet buffer, the page, or both.
139            let data = self.get_packet(buf);
140
141            // Perform packet mapping. If the packet contains stream data, queue it onto the packet
142            // queue. If it contains side data, then add it to the side data list. Ignore other
143            // types of packet data.
144            match self.mapper.map_packet(&data) {
145                Ok(MapResult::StreamData { dur }) => {
146                    // Create a packet.
147                    self.packets.push_back(Packet::new_from_boxed_slice(
148                        page.header.serial,
149                        0,
150                        dur,
151                        data,
152                    ));
153                }
154                Ok(MapResult::SideData { data }) => side_data.push(data),
155                Err(e) => {
156                    warn!("mapping packet failed ({}), skipping", e)
157                }
158                _ => (),
159            }
160        }
161
162        // If the page contains partial packet data, then save the partial packet data for later
163        // as the packet will be completed on a later page.
164        if let Some(buf) = iter.partial_packet() {
165            self.save_partial_packet(buf)?;
166        }
167
168        // The number of packets from this page that were queued.
169        let num_new_packets = self.packets.len() - num_prev_packets;
170
171        if num_new_packets > 0 {
172            // Get the start delay.
173            let start_delay = self.start_bound.as_ref().map_or(0, |b| b.delay);
174
175            // Assign timestamps by first calculating the timestamp of one past the last sample in
176            // in the last packet of this page, add the start delay.
177            let mut page_end_ts =
178                self.mapper.absgp_to_ts(page.header.absgp).saturating_add(start_delay);
179
180            // If this is the last page, then add the end delay to the timestamp.
181            if page.header.is_last_page {
182                let end_delay = self.end_bound.as_ref().map_or(0, |b| b.delay);
183                page_end_ts = page_end_ts.saturating_add(end_delay);
184            }
185
186            // Then, iterate over the newly added packets in reverse order and subtract their
187            // cumulative duration at each iteration to get the timestamp of the first sample
188            // in each packet.
189            let mut page_dur = 0u64;
190
191            for packet in self.packets.iter_mut().rev().take(num_new_packets) {
192                page_dur = page_dur.saturating_add(packet.dur);
193                packet.ts = page_end_ts.saturating_sub(page_dur);
194            }
195
196            if self.gapless {
197                for packet in self.packets.iter_mut().rev().take(num_new_packets) {
198                    symphonia_core::formats::util::trim_packet(
199                        packet,
200                        start_delay as u32,
201                        self.end_bound.as_ref().map(|b| b.ts),
202                    );
203                }
204            }
205        }
206
207        Ok(side_data)
208    }

230    /// Examine the first page of the non-setup codec bitstream to obtain the start time and start
231    /// delay parameters.
232    pub fn inspect_start_page(&mut self, page: &Page<'_>) {
233        if self.start_bound.is_some() {
234            debug!("start page already found");
235            return;
236        }
237
238        let mut parser = match self.mapper.make_parser() {
239            Some(parser) => parser,
240            _ => {
241                debug!("failed to make start bound packet parser");
242                return;
243            }
244        };
245
246        // Calculate the page duration.
247        let mut page_dur = 0u64;
248
249        for buf in page.packets() {
250            page_dur = page_dur.saturating_add(parser.parse_next_packet_dur(buf));
251        }
252
253        let page_end_ts = self.mapper.absgp_to_ts(page.header.absgp);
254
255        // If the page timestamp is >= the page duration, then the stream starts at timestamp 0 or
256        // a positive start time.
257        let bound = if page_end_ts >= page_dur {
258            Bound { seq: page.header.sequence, ts: page_end_ts - page_dur, delay: 0 }
259        }
260        else {
261            // If the page timestamp < the page duration, then the difference is the start delay.
262            Bound { seq: page.header.sequence, ts: 0, delay: page_dur - page_end_ts }
263        };
264
265        // Update codec parameters.
266        let codec_params = self.mapper.codec_params_mut();
267
268        codec_params.with_start_ts(bound.ts);
269
270        if bound.delay > 0 {
271            codec_params.with_delay(bound.delay as u32);
272        }
273
274        // Update start bound.
275        self.start_bound = Some(bound);
276    }

278    /// Examines one or more of the last pages of the codec bitstream to obtain the end time and
279    /// end delay parameters. To obtain the end delay, at a minimum, the last two pages are
280    /// required. The state returned by each iteration of this function should be passed into the
281    /// subsequent iteration.
282    pub fn inspect_end_page(&mut self, mut state: InspectState, page: &Page<'_>) -> InspectState {
283        if self.end_bound.is_some() {
284            debug!("end page already found");
285            return state;
286        }
287
288        // Get and/or create the sniffer state.
289        let parser = match &mut state.parser {
290            Some(parser) => parser,
291            None => {
292                state.parser = self.mapper.make_parser();
293
294                if let Some(parser) = &mut state.parser {
295                    parser
296                }
297                else {
298                    debug!("failed to make end bound packet parser");
299                    return state;
300                }
301            }
302        };
303
304        let start_delay = self.start_bound.as_ref().map_or(0, |b| b.delay);
305
306        // The actual page end timestamp is the absolute granule position + the start delay.
307        let page_end_ts = self
308            .mapper
309            .absgp_to_ts(page.header.absgp)
310            .saturating_add(if self.gapless { 0 } else { start_delay });
311
312        // Calculate the page duration. Note that even though only the last page uses this duration,
313        // it is important to feed the packet parser so that the first packet of the final page
314        // doesn't have a duration of 0 due to lapping on some codecs.
315        let mut page_dur = 0u64;
316
317        for buf in page.packets() {
318            page_dur = page_dur.saturating_add(parser.parse_next_packet_dur(buf));
319        }
320
321        // The end delay can only be determined if this is the last page, and the timstamp of the
322        // second last page is known.
323        let end_delay = if page.header.is_last_page {
324            if let Some(last_bound) = &state.bound {
325                // The real ending timestamp of the decoded data is the timestamp of the previous
326                // page plus the decoded duration of this page.
327                let actual_page_end_ts = last_bound.ts.saturating_add(page_dur);
328
329                // Any samples after the stated timestamp of this page are considered delay samples.
330                actual_page_end_ts.saturating_sub(page_end_ts)
331            }
332            else {
333                // Don't have the timestamp of the previous page so it is not possible to
334                // calculate the end delay.
335                0
336            }
337        }
338        else {
339            // Only the last page can have an end delay.
340            0
341        };
342
343        let bound = Bound { seq: page.header.sequence, ts: page_end_ts, delay: end_delay };
344
345        // If this is the last page, update the codec parameters.
346        if page.header.is_last_page {
347            let codec_params = self.mapper.codec_params_mut();
348
349            // Do not report the end delay if gapless is enabled.
350            let block_end_ts = bound.ts + if self.gapless { 0 } else { bound.delay };
351
352            if block_end_ts > codec_params.start_ts {
353                codec_params.with_n_frames(block_end_ts - codec_params.start_ts);
354            }
355
356            if bound.delay > 0 {
357                codec_params.with_padding(bound.delay as u32);
358            }
359
360            self.end_bound = Some(bound)
361        }
362
363        // Update the state's bound.
364        state.bound = Some(bound);
365
366        state
367    }

369    /// Examine a page and return the start and end timestamps as a tuple.
370    pub fn inspect_page(&mut self, page: &Page<'_>) -> (u64, u64) {
371        // Get the start delay.
372        let start_delay = self.start_bound.as_ref().map_or(0, |b| b.delay);
373
374        // Get the cumulative duration of all packets within this page.
375        let mut page_dur = 0u64;
376
377        if let Some(mut parser) = self.mapper.make_parser() {
378            for buf in page.packets() {
379                page_dur = page_dur.saturating_add(parser.parse_next_packet_dur(buf));
380            }
381        }
382
383        // If this is the final page, get the end delay.
384        let end_delay = if page.header.is_last_page {
385            self.end_bound.as_ref().map_or(0, |b| b.delay)
386        }
387        else {
388            0
389        };
390
391        // The total delay.
392        let delay = start_delay + end_delay;
393
394        // Add the total delay to the page end timestamp.
395        let page_end_ts = self.mapper.absgp_to_ts(page.header.absgp).saturating_add(delay);
396
397        // Get the page start timestamp of the page by subtracting the cumulative packet duration.
398        let page_start_ts = page_end_ts.saturating_sub(page_dur);
399
400        if !self.gapless {
401            // If gapless playback is disabled, then report the start and end timestamps with the
402            // delays incorporated.
403            (page_start_ts, page_end_ts)
404        }
405        else {
406            // If gapless playback is enabled, report the start and end timestamps without the
407            // delays.
408            (page_start_ts.saturating_sub(delay), page_end_ts.saturating_sub(delay))
409        }
410    }

412    fn get_packet(&mut self, packet_buf: &[u8]) -> Box<[u8]> {
413        if self.part_len == 0 {
414            Box::from(packet_buf)
415        }
416        else {
417            let mut buf = vec![0u8; self.part_len + packet_buf.len()];
418
419            // Split packet buffer into two portions: saved and new.
420            let (vec0, vec1) = buf.split_at_mut(self.part_len);
421
422            // Copy and consume the saved partial packet.
423            vec0.copy_from_slice(&self.part_buf[..self.part_len]);
424            self.part_len = 0;
425
426            // Read the remainder of the partial packet from the page.
427            vec1.copy_from_slice(packet_buf);
428
429            buf.into_boxed_slice()
430        }
431    }
432
433    fn save_partial_packet(&mut self, buf: &[u8]) -> Result<()> {
434        let new_part_len = self.part_len + buf.len();
435
436        if new_part_len > self.part_buf.len() {
437            // Do not exceed an a certain limit to prevent unbounded memory growth.
438            if new_part_len > LogicalStream::MAX_PACKET_LEN {
439                return decode_error("ogg: packet buffer would exceed max size");
440            }
441
442            // New partial packet buffer size, rounded up to the nearest 8K block.
443            let new_buf_len = (new_part_len + (8 * 1024 - 1)) & !(8 * 1024 - 1);
444            debug!("grow packet buffer to {} bytes", new_buf_len);
445
446            self.part_buf.resize(new_buf_len, Default::default());
447        }
448
449        self.part_buf[self.part_len..new_part_len].copy_from_slice(buf);
450        self.part_len = new_part_len;
451
452        Ok(())
453    }
