14const OGG_PAGE_MARKER: [u8; 4] = *b"OggS";
15const OGG_PAGE_HEADER_SIZE: usize = 27;
16
17pub const OGG_PAGE_MAX_SIZE: usize = OGG_PAGE_HEADER_SIZE + 255 + 255 * 255;
18
19#[derive(Copy, Clone, Default)]
20pub struct PageHeader {
21    pub version: u8,
22    pub absgp: u64,
23    pub serial: u32,
24    pub sequence: u32,
25    pub crc: u32,
26    pub n_segments: u8,
27    pub is_continuation: bool,
28    pub is_first_page: bool,
29    pub is_last_page: bool,
30}
31
32/// Reads a `PageHeader` from the the provided reader.
33fn read_page_header<B: ReadBytes>(reader: &mut B) -> Result<PageHeader> {
34    // The OggS marker should be present.
35    let marker = reader.read_quad_bytes()?;
36
37    if marker != OGG_PAGE_MARKER {
38        return decode_error("ogg: missing ogg stream marker");
39    }
40
41    let version = reader.read_byte()?;
42
43    // There is only one OGG version, and that is version 0.
44    if version != 0 {
45        return decode_error("ogg: invalid ogg version");
46    }
47
48    let flags = reader.read_byte()?;
49
50    // Only the first 3 least-significant bits are used for flags.
51    if flags & 0xf8 != 0 {
52        return decode_error("ogg: invalid flag bits set");
53    }
54
55    let ts = reader.read_u64()?;
56    let serial = reader.read_u32()?;
57    let sequence = reader.read_u32()?;
58    let crc = reader.read_u32()?;
59    let n_segments = reader.read_byte()?;
60
61    Ok(PageHeader {
62        version,
63        absgp: ts,
64        serial,
65        sequence,
66        crc,
67        n_segments,
68        is_continuation: (flags & 0x01) != 0,
69        is_first_page: (flags & 0x02) != 0,
70        is_last_page: (flags & 0x04) != 0,
71    })
72}
73
74/// Quickly synchronizes the provided reader to the next OGG page capture pattern, but does not
75/// perform any further verification.
76fn sync_page<B: ReadBytes>(reader: &mut B) -> Result<()> {
77    let mut marker = u32::from_be_bytes(reader.read_quad_bytes()?);
78
79    while marker.to_be_bytes() != OGG_PAGE_MARKER {
80        marker <<= 8;
81        marker |= u32::from(reader.read_u8()?);
82    }
83
84    Ok(())
85}

87/// An iterator over packets within a `Page`.
88pub struct PagePackets<'a> {
89    lens: core::slice::Iter<'a, u16>,
90    data: &'a [u8],
91}
92
93impl<'a> PagePackets<'a> {
94    /// If this page ends with an incomplete (partial) packet, get a slice to the data associated
95    /// with the partial packet.
96    pub fn partial_packet(self) -> Option<&'a [u8]> {
97        // Consume the rest of the packets.
98        let discard = usize::from(self.lens.sum::<u16>());
99
100        if self.data.len() > discard {
101            Some(&self.data[discard..])
102        }
103        else {
104            None
105        }
106    }
107}
108
109impl<'a> Iterator for PagePackets<'a> {
110    type Item = &'a [u8];
111
112    fn next(&mut self) -> Option<Self::Item> {
113        match self.lens.next() {
114            Some(len) => {
115                let (packet, rem) = self.data.split_at(usize::from(*len));
116                self.data = rem;
117                Some(packet)
118            }
119            _ => None,
120        }
121    }
122}

124/// An OGG page.
125pub struct Page<'a> {
126    /// The page header.
127    pub header: PageHeader,
128    packet_lens: &'a [u16],
129    page_buf: &'a [u8],
130}
131
132impl<'a> Page<'a> {
133    /// Returns an iterator over all complete packets within the page.
134    ///
135    /// If this page contains a partial packet, then the partial packet data may be retrieved using
136    /// the `partial_packet` function of the iterator.
137    pub fn packets(&self) -> PagePackets<'_> {
138        PagePackets { lens: self.packet_lens.iter(), data: self.page_buf }
139    }
140
141    /// Gets the number of packets completed on this page.
142    pub fn num_packets(&self) -> usize {
143        self.packet_lens.len()
144    }
145}

147/// A reader of OGG pages.
148pub struct PageReader {
149    header: PageHeader,
150    packet_lens: Vec<u16>,
151    page_buf: Vec<u8>,
152    page_buf_len: usize,
153}
154
155impl PageReader {
156    pub fn try_new<B>(reader: &mut B) -> Result<Self>
157    where
158        B: ReadBytes + SeekBuffered,
159    {
160        let mut page_reader = PageReader {
161            header: Default::default(),
162            packet_lens: Vec::new(),
163            page_buf: Vec::new(),
164            page_buf_len: 0,
165        };
166
167        page_reader.try_next_page(reader)?;
168
169        Ok(page_reader)
170    }
171
172    /// Attempts to read the next page. If the page is corrupted or invalid, returns an error.
173    pub fn try_next_page<B>(&mut self, reader: &mut B) -> Result<()>
174    where
175        B: ReadBytes + SeekBuffered,
176    {
177        let mut header_buf = [0u8; OGG_PAGE_HEADER_SIZE];
178        header_buf[..4].copy_from_slice(&OGG_PAGE_MARKER);
179
180        // Synchronize to an OGG page capture pattern.
181        sync_page(reader)?;
182
183        // Record the position immediately after synchronization. If the page is found corrupt the
184        // reader will need to seek back here to try to regain synchronization.
185        let sync_pos = reader.pos();
186
187        // Read the part of the page header after the capture pattern into a buffer.
188        reader.read_buf_exact(&mut header_buf[4..])?;
189
190        // Parse the page header buffer.
191        let header = read_page_header(&mut BufReader::new(&header_buf))?;
192
193        // debug!(
194        //     "page {{ version={}, absgp={}, serial={}, sequence={}, crc={:#x}, n_segments={}, \
195        //         is_first={}, is_last={}, is_continuation={} }}",
196        //     header.version,
197        //     header.absgp,
198        //     header.serial,
199        //     header.sequence,
200        //     header.crc,
201        //     header.n_segments,
202        //     header.is_first_page,
203        //     header.is_last_page,
204        //     header.is_continuation,
205        // );
206
207        // The CRC of the OGG page requires the page checksum bytes to be zeroed.
208        header_buf[22..26].copy_from_slice(&[0u8; 4]);
209
210        // Instantiate a Crc32, initialize it with 0, and feed it the page header buffer.
211        let mut crc32 = Crc32::new(0);
212
213        crc32.process_buf_bytes(&header_buf);
214
215        // The remainder of the page will be checksummed as it is read.
216        let mut crc32_reader = MonitorStream::new(reader, crc32);
217
218        // Read segment table.
219        let mut page_body_len = 0;
220        let mut packet_len = 0;
221
222        // TODO: Can this be transactional? A corrupt page causes the PageReader's state not
223        // to change.
224        self.packet_lens.clear();
225
226        for _ in 0..header.n_segments {
227            let seg_len = crc32_reader.read_byte()?;
228
229            page_body_len += usize::from(seg_len);
230            packet_len += u16::from(seg_len);
231
232            // A segment with a length < 255 indicates that the segment is the end of a packet.
233            // Push the packet length into the packet queue for the stream.
234            if seg_len < 255 {
235                self.packet_lens.push(packet_len);
236                packet_len = 0;
237            }
238        }
239
240        self.read_page_body(&mut crc32_reader, page_body_len)?;
241
242        let calculated_crc = crc32_reader.monitor().crc();
243
244        // If the CRC for the page is incorrect, then the page is corrupt.
245        if header.crc != calculated_crc {
246            warn!("crc mismatch: expected {:#x}, got {:#x}", header.crc, calculated_crc);
247
248            // Clear packet buffer.
249            self.packet_lens.clear();
250            self.page_buf_len = 0;
251
252            // Seek back to the immediately after the previous sync position.
253            crc32_reader.into_inner().seek_buffered(sync_pos);
254
255            return decode_error("ogg: crc mismatch");
256        }
257
258        self.header = header;
259
260        Ok(())
261    }
262
263    /// Reads the next page. If the next page is corrupted or invalid, the page is discarded and
264    /// the reader tries again until a valid page is read or end-of-stream.
265    pub fn next_page<B>(&mut self, reader: &mut B) -> Result<()>
266    where
267        B: ReadBytes + SeekBuffered,
268    {
269        loop {
270            match self.try_next_page(reader) {
271                Ok(_) => break,
272                Err(Error::IoError(e)) => return Err(Error::from(e)),
273                _ => (),
274            }
275        }
276        Ok(())
277    }
278
279    /// Reads the next page with a specific serial. If the next page is corrupted or invalid, the
280    /// page is discarded and the reader tries again until a valid page is read or end-of-stream.
281    pub fn next_page_for_serial<B>(&mut self, reader: &mut B, serial: u32) -> Result<()>
282    where
283        B: ReadBytes + SeekBuffered,
284    {
285        loop {
286            match self.try_next_page(reader) {
287                Ok(_) => {
288                    // Exit if a page with the specific serial is found.
289                    if self.header.serial == serial && !self.header.is_continuation {
290                        break;
291                    }
292                }
293                Err(Error::IoError(e)) => return Err(Error::from(e)),
294                _ => (),
295            }
296        }
297        Ok(())
298    }
299
300    /// Gets a buffer to the first packet, if it exists.
301    pub fn first_packet(&self) -> Option<&[u8]> {
302        self.packet_lens.first().map(|&len| &self.page_buf[..usize::from(len)])
303    }
304
305    /// Gets the current page header.
306    pub fn header(&self) -> PageHeader {
307        self.header
308    }
309
310    /// Gets a reference to the current page.
311    pub fn page(&self) -> Page<'_> {
312        assert!(self.page_buf_len <= 255 * 255, "ogg pages are <= 65025 bytes");
313
314        Page {
315            header: self.header,
316            packet_lens: &self.packet_lens,
317            page_buf: &self.page_buf[..self.page_buf_len],
318        }
319    }
320
321    fn read_page_body<B: ReadBytes>(&mut self, reader: &mut B, len: usize) -> Result<()> {
322        // This is precondition.
323        assert!(len <= 255 * 255);
324
325        if len > self.page_buf.len() {
326            // New page buffer size, rounded up to the nearest 8K block.
327            let new_buf_len = (len + (8 * 1024 - 1)) & !(8 * 1024 - 1);
328            debug!("grow page buffer to {} bytes", new_buf_len);
329
330            self.page_buf.resize(new_buf_len, Default::default());
331        }
332
333        self.page_buf_len = len;
334
335        reader.read_buf_exact(&mut self.page_buf[..len])?;
336
337        Ok(())
338    }
