23/// The minimum expected size of an Opus identification packet.
24const OGG_OPUS_MIN_IDENTIFICATION_PACKET_SIZE: usize = 19;
25
26/// The signature for an Opus identification packet.
27const OGG_OPUS_MAGIC_SIGNATURE: &[u8] = b"OpusHead";
28
29/// The signature for an Opus metadata packet.
30const OGG_OPUS_COMMENT_SIGNATURE: &[u8] = b"OpusTags";
31
32/// The maximum support Opus OGG mapping version.
33const OGG_OPUS_MAPPING_VERSION_MAX: u8 = 0x0f;
34
35pub fn detect(buf: &[u8]) -> Result<Option<Box<dyn Mapper>>> {
36    // The identification packet for Opus must be a minimum size.
37    if buf.len() < OGG_OPUS_MIN_IDENTIFICATION_PACKET_SIZE {
38        return Ok(None);
39    }
40
41    let mut reader = BufReader::new(buf);
42
43    // The first 8 bytes are the magic signature ASCII bytes.
44    let mut magic = [0; 8];
45    reader.read_buf_exact(&mut magic)?;
46
47    if magic != *OGG_OPUS_MAGIC_SIGNATURE {
48        return Ok(None);
49    }
50
51    // The next byte is the OGG Opus encapsulation version. The version is split into two
52    // sub-fields: major and minor. These fields are stored in the upper and lower 4-bit,
53    // respectively.
54    if reader.read_byte()? > OGG_OPUS_MAPPING_VERSION_MAX {
55        return Ok(None);
56    }
57
58    // The next byte is the number of channels and must not be 0.
59    let channel_count = reader.read_byte()?;
60
61    if channel_count == 0 {
62        return Ok(None);
63    }
64
65    // The next 16-bit integer is the pre-skip padding (# of samples at 48kHz to subtract from the
66    // OGG granule position to obtain the PCM sample position).
67    let pre_skip = reader.read_u16()?;
68
69    // The next 32-bit integer is the sample rate of the original audio.
70    let _ = reader.read_u32()?;
71
72    // Next, the 16-bit gain value.
73    let _ = reader.read_u16()?;
74
75    // The next byte indicates the channel mapping. Most of these values are reserved.
76    let channel_mapping = reader.read_byte()?;
77
78    let channels = match channel_mapping {
79        // RTP Mapping
80        0 if channel_count == 1 => Channels::FRONT_LEFT,
81        0 if channel_count == 2 => Channels::FRONT_LEFT | Channels::FRONT_RIGHT,
82        // Vorbis Mapping
83        1 => match channel_count {
84            1 => Channels::FRONT_LEFT,
85            2 => Channels::FRONT_LEFT | Channels::FRONT_RIGHT,
86            3 => Channels::FRONT_LEFT | Channels::FRONT_CENTRE | Channels::FRONT_RIGHT,
87            4 => {
88                Channels::FRONT_LEFT
89                    | Channels::FRONT_RIGHT
90                    | Channels::REAR_LEFT
91                    | Channels::REAR_RIGHT
92            }
93            5 => {
94                Channels::FRONT_LEFT
95                    | Channels::FRONT_CENTRE
96                    | Channels::FRONT_RIGHT
97                    | Channels::REAR_LEFT
98                    | Channels::REAR_RIGHT
99            }
100            6 => {
101                Channels::FRONT_LEFT
102                    | Channels::FRONT_CENTRE
103                    | Channels::FRONT_RIGHT
104                    | Channels::REAR_LEFT
105                    | Channels::REAR_RIGHT
106                    | Channels::LFE1
107            }
108            7 => {
109                Channels::FRONT_LEFT
110                    | Channels::FRONT_CENTRE
111                    | Channels::FRONT_RIGHT
112                    | Channels::SIDE_LEFT
113                    | Channels::SIDE_RIGHT
114                    | Channels::REAR_CENTRE
115                    | Channels::LFE1
116            }
117            8 => {
118                Channels::FRONT_LEFT
119                    | Channels::FRONT_CENTRE
120                    | Channels::FRONT_RIGHT
121                    | Channels::SIDE_LEFT
122                    | Channels::SIDE_RIGHT
123                    | Channels::REAR_LEFT
124                    | Channels::REAR_RIGHT
125                    | Channels::LFE1
126            }
127            _ => return Ok(None),
128        },
129        // Reserved, and should NOT be supported for playback.
130        _ => return Ok(None),
131    };
132
133    // Populate the codec parameters with the information read from identification header.
134    let mut codec_params = CodecParameters::new();
135
136    codec_params
137        .for_codec(CODEC_TYPE_OPUS)
138        .with_delay(u32::from(pre_skip))
139        .with_sample_rate(48_000)
140        .with_time_base(TimeBase::new(1, 48_000))
141        .with_channels(channels)
142        .with_extra_data(Box::from(buf));
143
144    // Instantiate the Opus mapper.
145    let mapper = Box::new(OpusMapper { codec_params, need_comment: true });
146
147    Ok(Some(mapper))
148}

150pub struct OpusPacketParser {}
151
152impl PacketParser for OpusPacketParser {
153    fn parse_next_packet_dur(&mut self, packet: &[u8]) -> u64 {
154        // See https://www.rfc-editor.org/rfc/rfc6716
155        // Read TOC (Table Of Contents) byte which is the first byte in the opus data.
156        let toc_byte = match packet.first() {
157            Some(b) => b,
158            None => {
159                warn!("opus packet empty");
160                return 0;
161            }
162        };
163        // The configuration number is the 5 most significant bits. Shift out 3 least significant
164        // bits.
165        let configuration_number = toc_byte >> 3; // max 2^5-1 = 31
166
167        // The configuration number maps to packet length according to this lookup table.
168        // See https://www.rfc-editor.org/rfc/rfc6716 top half of page 14.
169        // Numbers are in milliseconds in the rfc. Down below they are in TimeBase units, so
170        // 10ms = 10*48.
171        #[rustfmt::skip]
172        const CONFIGURATION_NUMBER_TO_FRAME_DURATION: [u32; 32] = [
173            10*48, 20*48, 40*48, 60*48,
174            10*48, 20*48, 40*48, 60*48,
175            10*48, 20*48, 40*48, 60*48,
176            10*48, 20*48,
177            10*48, 20*48,
178            (2.5*48.0) as u32, 5*48, 10*48, 20*48,
179            (2.5*48.0) as u32, 5*48, 10*48, 20*48,
180            (2.5*48.0) as u32, 5*48, 10*48, 20*48,
181            (2.5*48.0) as u32, 5*48, 10*48, 20*48,
182        ];
183        // Look up the frame length.
184        let frame_duration =
185            CONFIGURATION_NUMBER_TO_FRAME_DURATION[configuration_number as usize] as u64;
186
187        // Look up the number of frames in the packet.
188        // See https://www.rfc-editor.org/rfc/rfc6716 bottom half of page 14.
189        let c = toc_byte & 0b11; // Note: it's actually called "c" in the rfc.
190        let num_frames = match c {
191            0 => 1,
192            1 | 2 => 2,
193            3 => match packet.get(1) {
194                Some(byte) => {
195                    // TOC byte is followed by number of frames. See page 18 section 3.2.5 code 3
196                    let m = byte & 0b11111; // Note: it's actually called "M" in the rfc.
197                    m as u64
198                }
199                None => {
200                    // What to do here? I'd like to return an error but this is an infalliable
201                    // trait.
202                    warn!("opus code 3 packet with no following byte containing number of frames");
203                    return 0;
204                }
205            },
206            _ => unreachable!("masked 2 bits"),
207        };
208        // Look up the packet length and return it.
209        frame_duration * num_frames
210    }
211}

213struct OpusMapper {
214    codec_params: CodecParameters,
215    need_comment: bool,
216}
217
218impl Mapper for OpusMapper {
219    fn name(&self) -> &'static str {
220        "opus"
221    }
222
223    fn reset(&mut self) {
224        // Nothing to do.
225    }
226
227    fn codec_params(&self) -> &CodecParameters {
228        &self.codec_params
229    }
230
231    fn codec_params_mut(&mut self) -> &mut CodecParameters {
232        &mut self.codec_params
233    }
234
235    fn make_parser(&self) -> Option<Box<dyn super::PacketParser>> {
236        Some(Box::new(OpusPacketParser {}))
237    }
238
239    fn map_packet(&mut self, packet: &[u8]) -> Result<MapResult> {
240        if !self.need_comment {
241            Ok(MapResult::StreamData { dur: OpusPacketParser {}.parse_next_packet_dur(packet) })
242        }
243        else {
244            let mut reader = BufReader::new(packet);
245
246            // Read the header signature.
247            let mut sig = [0; 8];
248            reader.read_buf_exact(&mut sig)?;
249
250            if sig == *OGG_OPUS_COMMENT_SIGNATURE {
251                // This packet should be a metadata packet containing a Vorbis Comment.
252                let mut builder = MetadataBuilder::new();
253
254                vorbis::read_comment_no_framing(&mut reader, &mut builder)?;
255
256                self.need_comment = false;
257
258                Ok(MapResult::SideData { data: SideData::Metadata(builder.metadata()) })
259            }
260            else {
261                warn!("ogg (opus): invalid packet type");
262                Ok(MapResult::Unknown)
263            }
264        }
265    }
266}
