22/// The identification header packet size.
23const VORBIS_IDENTIFICATION_HEADER_SIZE: usize = 30;
24
25/// The packet type for an identification header.
26const VORBIS_PACKET_TYPE_IDENTIFICATION: u8 = 1;
27/// The packet type for a comment header.
28const VORBIS_PACKET_TYPE_COMMENT: u8 = 3;
29/// The packet type for a setup header.
30const VORBIS_PACKET_TYPE_SETUP: u8 = 5;
31
32/// The common header packet signature.
33const VORBIS_HEADER_PACKET_SIGNATURE: &[u8] = b"vorbis";
34
35/// The Vorbis version supported by this mapper.
36const VORBIS_VERSION: u32 = 0;
37
38/// The minimum block size (64) expressed as a power-of-2 exponent.
39const VORBIS_BLOCKSIZE_MIN: u8 = 6;
40/// The maximum block size (8192) expressed as a power-of-2 exponent.
41const VORBIS_BLOCKSIZE_MAX: u8 = 13;
42
43struct VorbisPacketParser {
44    modes_block_flags: u64,
45    num_modes: u8,
46    bs0_exp: u8,
47    bs1_exp: u8,
48    prev_bs_exp: Option<u8>,
49}
50
51impl VorbisPacketParser {
52    fn new(bs0_exp: u8, bs1_exp: u8, num_modes: u8, modes_block_flags: u64) -> Self {
53        Self { bs0_exp, bs1_exp, num_modes, modes_block_flags, prev_bs_exp: None }
54    }
55
56    fn reset(&mut self) {
57        self.prev_bs_exp = None;
58    }
59}
60
61impl PacketParser for VorbisPacketParser {
62    fn parse_next_packet_dur(&mut self, packet: &[u8]) -> u64 {
63        let mut bs = BitReaderRtl::new(packet);
64
65        // First bit must be 0 to indicate audio packet.
66        match bs.read_bool() {
67            Ok(bit) if !bit => (),
68            _ => return 0,
69        }
70
71        // Number of bits for the mode number.
72        let mode_num_bits = ilog(u32::from(self.num_modes) - 1);
73
74        // Read the mode number.
75        let mode_num = match bs.read_bits_leq32(mode_num_bits) {
76            Ok(mode_num) => mode_num as u8,
77            _ => return 0,
78        };
79
80        // Determine the current block size.
81        let cur_bs_exp = if mode_num < self.num_modes {
82            let block_flag = (self.modes_block_flags >> mode_num) & 1;
83            if block_flag == 1 {
84                self.bs1_exp
85            }
86            else {
87                self.bs0_exp
88            }
89        }
90        else {
91            return 0;
92        };
93
94        // Calculate the duration if the previous block size is available. Otherwise return 0.
95        let dur = if let Some(prev_bs_exp) = self.prev_bs_exp {
96            ((1 << prev_bs_exp) >> 2) + ((1 << cur_bs_exp) >> 2)
97        }
98        else {
99            0
100        };
101
102        self.prev_bs_exp = Some(cur_bs_exp);
103
104        dur
105    }
106}

108pub fn detect(buf: &[u8]) -> Result<Option<Box<dyn Mapper>>> {
109    // The identification header packet must be the correct size.
110    if buf.len() != VORBIS_IDENTIFICATION_HEADER_SIZE {
111        return Ok(None);
112    }
113
114    // Read the identification header. Any errors cause detection to fail.
115    let ident = match read_ident_header(&mut BufReader::new(buf)) {
116        Ok(ident) => ident,
117        _ => return Ok(None),
118    };
119
120    // Populate the codec parameters with the information above.
121    let mut codec_params = CodecParameters::new();
122
123    codec_params
124        .for_codec(CODEC_TYPE_VORBIS)
125        .with_sample_rate(ident.sample_rate)
126        .with_time_base(TimeBase::new(1, ident.sample_rate))
127        .with_extra_data(Box::from(buf));
128
129    if let Some(channels) = vorbis_channels_to_channels(ident.n_channels) {
130        codec_params.with_channels(channels);
131    }
132
133    // Instantiate the Vorbis mapper.
134    let mapper =
135        Box::new(VorbisMapper { codec_params, ident, parser: None, has_setup_header: false });
136
137    Ok(Some(mapper))
138}

140struct VorbisMapper {
141    codec_params: CodecParameters,
142    ident: IdentHeader,
143    parser: Option<VorbisPacketParser>,
144    has_setup_header: bool,
145}
146
147impl Mapper for VorbisMapper {
148    fn name(&self) -> &'static str {
149        "vorbis"
150    }
151
152    fn codec_params(&self) -> &CodecParameters {
153        &self.codec_params
154    }
155
156    fn codec_params_mut(&mut self) -> &mut CodecParameters {
157        &mut self.codec_params
158    }
159
160    fn reset(&mut self) {
161        if let Some(parser) = &mut self.parser {
162            parser.reset()
163        }
164    }
165
166    fn make_parser(&self) -> Option<Box<dyn PacketParser>> {
167        match &self.parser {
168            Some(base_parser) => {
169                let parser = Box::new(VorbisPacketParser::new(
170                    base_parser.bs0_exp,
171                    base_parser.bs1_exp,
172                    base_parser.num_modes,
173                    base_parser.modes_block_flags,
174                ));
175                Some(parser)
176            }
177            _ => None,
178        }
179    }
180
181    fn map_packet(&mut self, packet: &[u8]) -> Result<MapResult> {
182        let mut reader = BufReader::new(packet);
183
184        // All Vorbis packets indicate the packet type in the first byte.
185        let packet_type = reader.read_u8()?;
186
187        // An even numbered packet type is an audio packet.
188        if packet_type & 1 == 0 {
189            let dur = match &mut self.parser {
190                Some(parser) => parser.parse_next_packet_dur(packet),
191                _ => 0,
192            };
193
194            Ok(MapResult::StreamData { dur })
195        }
196        else {
197            // Odd numbered packet types are header packets.
198            let mut sig = [0; 6];
199            reader.read_buf_exact(&mut sig)?;
200
201            // Check if the presumed header packet has the common header packet signature.
202            if sig != VORBIS_HEADER_PACKET_SIGNATURE {
203                return decode_error("ogg (vorbis): header packet signature invalid");
204            }
205
206            // Handle each header packet type specifically.
207            match packet_type {
208                VORBIS_PACKET_TYPE_COMMENT => {
209                    let mut builder = MetadataBuilder::new();
210
211                    read_comment_no_framing(&mut reader, &mut builder)?;
212
213                    Ok(MapResult::SideData { data: SideData::Metadata(builder.metadata()) })
214                }
215                VORBIS_PACKET_TYPE_SETUP => {
216                    // Append the setup headers to the extra data.
217                    let mut extra_data = self.codec_params.extra_data.take().unwrap().to_vec();
218                    extra_data.extend_from_slice(packet);
219
220                    // Try to read the setup header.
221                    if let Ok(modes) = read_setup(&mut BufReader::new(packet), &self.ident) {
222                        let num_modes = modes.len();
223                        let mut modes_block_flags = 0;
224
225                        assert!(num_modes <= 64);
226
227                        for (i, mode) in modes.iter().enumerate() {
228                            if mode.block_flag {
229                                modes_block_flags |= 1 << i;
230                            }
231                        }
232
233                        let parser = VorbisPacketParser::new(
234                            self.ident.bs0_exp,
235                            self.ident.bs1_exp,
236                            num_modes as u8,
237                            modes_block_flags,
238                        );
239
240                        self.parser.replace(parser);
241                    }
242
243                    self.codec_params.with_extra_data(extra_data.into_boxed_slice());
244                    self.has_setup_header = true;
245
246                    Ok(MapResult::Setup)
247                }
248                _ => {
249                    warn!("ogg (vorbis): packet type {} unexpected", packet_type);
250                    Ok(MapResult::Unknown)
251                }
252            }
253        }
254    }
255
256    fn is_ready(&self) -> bool {
257        self.has_setup_header
258    }
259}

261struct IdentHeader {
262    n_channels: u8,
263    sample_rate: u32,
264    bs0_exp: u8,
265    bs1_exp: u8,
266}

268fn read_ident_header<B: ReadBytes>(reader: &mut B) -> Result<IdentHeader> {
269    // The packet type must be an identification header.
270    let packet_type = reader.read_u8()?;
271
272    if packet_type != VORBIS_PACKET_TYPE_IDENTIFICATION {
273        return decode_error("ogg (vorbis): invalid packet type for identification header");
274    }
275
276    // Next, the header packet signature must be correct.
277    let mut packet_sig_buf = [0; 6];
278    reader.read_buf_exact(&mut packet_sig_buf)?;
279
280    if packet_sig_buf != VORBIS_HEADER_PACKET_SIGNATURE {
281        return decode_error("ogg (vorbis): invalid header signature");
282    }
283
284    // Next, the Vorbis version must be 0.
285    let version = reader.read_u32()?;
286
287    if version != VORBIS_VERSION {
288        return unsupported_error("ogg (vorbis): only vorbis 1 is supported");
289    }
290
291    // Next, the number of channels and sample rate must be non-zero.
292    let n_channels = reader.read_u8()?;
293
294    if n_channels == 0 {
295        return decode_error("ogg (vorbis): number of channels cannot be 0");
296    }
297
298    let sample_rate = reader.read_u32()?;
299
300    if sample_rate == 0 {
301        return decode_error("ogg (vorbis): sample rate cannot be 0");
302    }
303
304    // Read the bitrate range.
305    let _bitrate_max = reader.read_u32()?;
306    let _bitrate_nom = reader.read_u32()?;
307    let _bitrate_min = reader.read_u32()?;
308
309    // Next, blocksize_0 and blocksize_1 are packed into a single byte.
310    let block_sizes = reader.read_u8()?;
311
312    let bs0_exp = (block_sizes & 0x0f) >> 0;
313    let bs1_exp = (block_sizes & 0xf0) >> 4;
314
315    // The block sizes must not exceed the bounds.
316    if bs0_exp < VORBIS_BLOCKSIZE_MIN || bs0_exp > VORBIS_BLOCKSIZE_MAX {
317        return decode_error("ogg (vorbis): blocksize_0 out-of-bounds");
318    }
319
320    if bs1_exp < VORBIS_BLOCKSIZE_MIN || bs1_exp > VORBIS_BLOCKSIZE_MAX {
321        return decode_error("ogg (vorbis): blocksize_1 out-of-bounds");
322    }
323
324    // Blocksize_0 must be >= blocksize_1
325    if bs0_exp > bs1_exp {
326        return decode_error("ogg (vorbis): blocksize_0 exceeds blocksize_1");
327    }
328
329    // Framing flag must be set.
330    if reader.read_u8()? != 0x1 {
331        return decode_error("ogg (vorbis): ident header framing flag unset");
332    }
333
334    Ok(IdentHeader { n_channels, sample_rate, bs0_exp, bs1_exp })
335}

337fn read_setup(reader: &mut BufReader<'_>, ident: &IdentHeader) -> Result<Vec<Mode>> {
338    // The packet type must be an setup header.
339    let packet_type = reader.read_u8()?;
340
341    if packet_type != VORBIS_PACKET_TYPE_SETUP {
342        return decode_error("ogg (vorbis): invalid packet type for setup header");
343    }
344
345    // Next, the setup packet signature must be correct.
346    let mut packet_sig_buf = [0; 6];
347    reader.read_buf_exact(&mut packet_sig_buf)?;
348
349    if packet_sig_buf != VORBIS_HEADER_PACKET_SIGNATURE {
350        return decode_error("ogg (vorbis): invalid setup header signature");
351    }
352
353    // The remaining portion of the setup header packet is read bitwise.
354    let mut bs = BitReaderRtl::new(reader.read_buf_bytes_available_ref());
355
356    // Skip the codebooks.
357    skip_codebooks(&mut bs)?;
358
359    // Skip the time-domain transforms (placeholders in Vorbis 1).
360    skip_time_domain_transforms(&mut bs)?;
361
362    // Skip the floors.
363    skip_floors(&mut bs)?;
364
365    // Skip the residues.
366    skip_residues(&mut bs)?;
367
368    // Skip the channel mappings.
369    skip_mappings(&mut bs, ident.n_channels)?;
370
371    // Read modes.
372    let modes = read_modes(&mut bs)?;
373
374    // Framing flag must be set.
375    if !bs.read_bool()? {
376        return decode_error("ogg (vorbis): setup header framing flag unset");
377    }
378
379    Ok(modes)
380}

382fn skip_codebooks(bs: &mut BitReaderRtl<'_>) -> Result<()> {
383    let count = bs.read_bits_leq32(8)? + 1;
384    for _ in 0..count {
385        skip_codebook(bs)?;
386    }
387    Ok(())
388}

390pub fn skip_codebook(bs: &mut BitReaderRtl<'_>) -> Result<()> {
391    // Verify codebook synchronization word.
392    let sync = bs.read_bits_leq32(24)?;
393
394    if sync != 0x564342 {
395        return decode_error("ogg (vorbis): invalid codebook sync");
396    }
397
398    // Read codebook number of dimensions and entries.
399    let codebook_dimensions = bs.read_bits_leq32(16)? as u16;
400    let codebook_entries = bs.read_bits_leq32(24)?;
401    let is_length_ordered = bs.read_bool()?;
402
403    if !is_length_ordered {
404        // Codeword list is not length ordered.
405        let is_sparse = bs.read_bool()?;
406
407        if is_sparse {
408            // Sparsely packed codeword entry list.
409            for _ in 0..codebook_entries {
410                if bs.read_bool()? {
411                    let _ = bs.read_bits_leq32(5)?;
412                }
413            }
414        }
415        else {
416            bs.ignore_bits(codebook_entries * 5)?;
417        }
418    }
419    else {
420        // Codeword list is length ordered.
421        let mut cur_entry = 0;
422        let mut _cur_len = bs.read_bits_leq32(5)? + 1;
423
424        loop {
425            let num_bits =
426                if codebook_entries > cur_entry { ilog(codebook_entries - cur_entry) } else { 0 };
427
428            let num = bs.read_bits_leq32(num_bits)?;
429
430            cur_entry += num;
431
432            if cur_entry > codebook_entries {
433                return decode_error("ogg (vorbis): invalid codebook");
434            }
435
436            if cur_entry == codebook_entries {
437                break;
438            }
439        }
440    }
441
442    // Read and unpack vector quantization (VQ) lookup table.
443    let lookup_type = bs.read_bits_leq32(4)?;
444
445    match lookup_type & 0xf {
446        0 => (),
447        1 | 2 => {
448            let _min_value = bs.read_bits_leq32(32)?;
449            let _delta_value = bs.read_bits_leq32(32)?;
450            let value_bits = bs.read_bits_leq32(4)? + 1;
451            let _sequence_p = bs.read_bool()?;
452
453            // Lookup type is either 1 or 2 as per outer match.
454            let lookup_values = match lookup_type {
455                1 => lookup1_values(codebook_entries, codebook_dimensions),
456                2 => codebook_entries * u32::from(codebook_dimensions),
457                _ => unreachable!(),
458            };
459
460            // Multiplicands
461            bs.ignore_bits(lookup_values * value_bits)?;
462        }
463        _ => return decode_error("ogg (vorbis): invalid codeword lookup type"),
464    }
465
466    Ok(())
467}

469fn skip_time_domain_transforms(bs: &mut BitReaderRtl<'_>) -> Result<()> {
470    let count = bs.read_bits_leq32(6)? + 1;
471
472    for _ in 0..count {
473        // All these values are placeholders and must be 0.
474        if bs.read_bits_leq32(16)? != 0 {
475            return decode_error("ogg (vorbis): invalid time domain tranform");
476        }
477    }
478
479    Ok(())
480}

490fn skip_floor(bs: &mut BitReaderRtl<'_>) -> Result<()> {
491    let floor_type = bs.read_bits_leq32(16)?;
492
493    match floor_type {
494        0 => skip_floor0_setup(bs),
495        1 => skip_floor1_setup(bs),
496        _ => decode_error("ogg (vorbis): invalid floor type"),
497    }
498}

500fn skip_floor0_setup(bs: &mut BitReaderRtl<'_>) -> Result<()> {
501    // floor0_order
502    // floor0_rate
503    // floor0_bark_map_size
504    // floor0_amplitude_bits
505    // floor0_amplitude_offset
506    bs.ignore_bits(8 + 16 + 16 + 6 + 8)?;
507    let floor0_number_of_books = bs.read_bits_leq32(4)? + 1;
508    bs.ignore_bits(floor0_number_of_books * 8)?;
509    Ok(())
510}

512fn skip_floor1_setup(bs: &mut BitReaderRtl<'_>) -> Result<()> {
513    // The number of partitions. 5-bit value, 0..31 range.
514    let floor1_partitions = bs.read_bits_leq32(5)? as usize;
515
516    // Parition list of up-to 32 partitions (floor1_partitions), with each partition indicating
517    // a 4-bit class (0..16) identifier.
518    let mut floor1_partition_class_list = [0; 32];
519    let mut floor1_classes_dimensions = [0; 16];
520
521    if floor1_partitions > 0 {
522        let mut max_class = 0; // 4-bits, 0..15
523
524        for class_idx in &mut floor1_partition_class_list[..floor1_partitions] {
525            *class_idx = bs.read_bits_leq32(4)? as u8;
526            max_class = max_class.max(*class_idx);
527        }
528
529        let num_classes = usize::from(1 + max_class);
530
531        for dimensions in floor1_classes_dimensions[..num_classes].iter_mut() {
532            *dimensions = bs.read_bits_leq32(3)? as u8 + 1;
533
534            let subclass_bits = bs.read_bits_leq32(2)?;
535
536            if subclass_bits != 0 {
537                let _main_book = bs.read_bits_leq32(8)?;
538            }
539
540            let num_subclasses = 1 << subclass_bits;
541
542            // Sub-class books
543            bs.ignore_bits(num_subclasses * 8)?;
544        }
545    }
546
547    let _floor1_multiplier = bs.read_bits_leq32(2)?;
548
549    let rangebits = bs.read_bits_leq32(4)?;
550
551    for &class_idx in &floor1_partition_class_list[..floor1_partitions] {
552        let class_dimensions = u32::from(floor1_classes_dimensions[class_idx as usize]);
553        // TODO? No more than 65 elements are allowed.
554        bs.ignore_bits(class_dimensions * rangebits)?;
555    }
556
557    Ok(())
558}

560fn skip_residues(bs: &mut BitReaderRtl<'_>) -> Result<()> {
561    let count = bs.read_bits_leq32(6)? + 1;
562    for _ in 0..count {
563        let _residue_type = bs.read_bits_leq32(16)?;
564        skip_residue_setup(bs)?
565    }
566    Ok(())
567}

569fn skip_residue_setup(bs: &mut BitReaderRtl<'_>) -> Result<()> {
570    // residue_begin
571    // residue_end
572    // residue_partition_size
573    bs.ignore_bits(24 + 24 + 24)?;
574    let residue_classifications = bs.read_bits_leq32(6)? as u8 + 1;
575
576    // residue_classbook
577    bs.ignore_bits(8)?;
578
579    let mut num_codebooks = 0;
580
581    for _ in 0..residue_classifications {
582        let low_bits = bs.read_bits_leq32(3)? as u8;
583        let high_bits = if bs.read_bool()? { bs.read_bits_leq32(5)? as u8 } else { 0 };
584        let is_used = (high_bits << 3) | low_bits;
585        num_codebooks += is_used.count_ones();
586    }
587
588    bs.ignore_bits(num_codebooks * 8)?;
589
590    Ok(())
591}

593fn skip_mappings(bs: &mut BitReaderRtl<'_>, audio_channels: u8) -> Result<()> {
594    let count = bs.read_bits_leq32(6)? + 1;
595    for _ in 0..count {
596        skip_mapping(bs, audio_channels)?
597    }
598    Ok(())

601fn skip_mapping(bs: &mut BitReaderRtl<'_>, audio_channels: u8) -> Result<()> {
602    let mapping_type = bs.read_bits_leq32(16)?;
603
604    match mapping_type {
605        0 => skip_mapping_type0_setup(bs, audio_channels),
606        _ => decode_error("ogg (vorbis): invalid mapping type"),
607    }
608}
609
610fn skip_mapping_type0_setup(bs: &mut BitReaderRtl<'_>, audio_channels: u8) -> Result<()> {
611    let num_submaps = if bs.read_bool()? { bs.read_bits_leq32(4)? + 1 } else { 1 };
612
613    if bs.read_bool()? {
614        // Number of channel couplings (up-to 256).
615        let coupling_steps = bs.read_bits_leq32(8)? as u16 + 1;
616
617        // The maximum channel number.
618        let max_ch = audio_channels - 1;
619
620        // The number of bits to read for the magnitude and angle channel numbers. Never exceeds 8.
621        let coupling_bits = ilog(u32::from(max_ch));
622        debug_assert!(coupling_bits <= 8);
623
624        // Read each channel coupling.
625        for _ in 0..coupling_steps {
626            let _magnitude_ch = bs.read_bits_leq32(coupling_bits)?;
627            let _angle_ch = bs.read_bits_leq32(coupling_bits)?;
628        }
629    }
630
631    if bs.read_bits_leq32(2)? != 0 {
632        return decode_error("ogg (vorbis): reserved mapping bits non-zero");
633    }
634
635    // If the number of submaps is > 1 read the multiplex numbers from the bitstream, otherwise
636    // they're all 0.
637    if num_submaps > 1 {
638        // Mux to use per channel.
639        bs.ignore_bits(u32::from(audio_channels) * 4)?;
640    }
641
642    // Reserved, floor, and residue to use per submap.
643    bs.ignore_bits(num_submaps * (8 + 8 + 8))?;
644
645    Ok(())
646}

648fn read_modes(bs: &mut BitReaderRtl<'_>) -> Result<Vec<Mode>> {
649    let count = bs.read_bits_leq32(6)? + 1;
650    (0..count).map(|_| read_mode(bs)).collect()
651}

653#[derive(Debug)]
654struct Mode {
655    block_flag: bool,
656}
657
658fn read_mode(bs: &mut BitReaderRtl<'_>) -> Result<Mode> {
659    let block_flag = bs.read_bool()?;
660    let window_type = bs.read_bits_leq32(16)? as u16;
661    let transform_type = bs.read_bits_leq32(16)? as u16;
662    let _mapping = bs.read_bits_leq32(8)? as u8;
663
664    // Only window type 0 is allowed in Vorbis 1 (section 4.2.4).
665    if window_type != 0 {
666        return decode_error("ogg (vorbis): invalid window type for mode");
667    }
668
669    // Only transform type 0 is allowed in Vorbis 1 (section 4.2.4).
670    if transform_type != 0 {
671        return decode_error("ogg (vorbis): invalid transform type for mode");
672    }
673
674    let mode = Mode { block_flag };
675
676    Ok(mode)
677}

679#[inline(always)]
680pub fn ilog(x: u32) -> u32 {
681    32 - x.leading_zeros()
682}

684#[inline(always)]
685fn lookup1_values(entries: u32, dimensions: u16) -> u32 {
686    // (value ^ dimensions) <= entries
687    // [(value ^ dimensions) ^ (1 / dimensions)] = lower[entries ^ (1 / dimensions)]
688    // value = lower[entries ^ (1 / dimensions)]
689    let value = (entries as f32).powf(1.0f32 / f32::from(dimensions)).floor() as u32;
690
691    assert!(value.pow(u32::from(dimensions)) <= entries);
692    assert!((value + 1).pow(u32::from(dimensions)) > entries);
693
694    value
695}
