24/// The expected size of the first FLAC header packet.
25const OGG_FLAC_HEADER_PACKET_SIZE: usize = 51;
26
27/// The major version number of the supported OGG-FLAC mapping.
28const OGG_FLAC_MAPPING_MAJOR_VERSION: u8 = 1;
29
30/// The OGG-FLAC header packet signature.
31const OGG_FLAC_HEADER_SIGNATURE: &[u8] = b"FLAC";
32
33/// The OGG-FLAC header packet type value.
34const OGG_FLAC_PACKET_TYPE: u8 = 0x7f;
35
36/// The native FLAC signature.
37const FLAC_SIGNATURE: &[u8] = b"fLaC";
38
39pub fn detect(buf: &[u8]) -> Result<Option<Box<dyn Mapper>>> {
40    // The packet shall be exactly the expected length.
41    if buf.len() != OGG_FLAC_HEADER_PACKET_SIZE {
42        return Ok(None);
43    }
44
45    let mut reader = BufReader::new(buf);
46
47    // The first byte indicates the packet type and must be 0x7f.
48    if reader.read_u8()? != OGG_FLAC_PACKET_TYPE {
49        return Ok(None);
50    }
51
52    // Next, the OGG FLAC signature, in ASCII, must be "FLAC".
53    if reader.read_quad_bytes()? != OGG_FLAC_HEADER_SIGNATURE {
54        return Ok(None);
55    }
56
57    // Next, a one-byte binary major version number for the mapping, only version 1 is supported.
58    if reader.read_u8()? != OGG_FLAC_MAPPING_MAJOR_VERSION {
59        return Ok(None);
60    }
61
62    // Next, a one-byte minor version number for the mapping. This is ignored because we support all
63    // version 1 features.
64    let _minor = reader.read_u8()?;
65
66    // Next, a two-byte, big-endian number signifying the number of header (non-audio) packets, not
67    // including the identification packet. This number may be 0 to signify it is unknown.
68    let _ = reader.read_be_u16()?;
69
70    // Last, the four-byte ASCII native FLAC signature "fLaC".
71    if reader.read_quad_bytes()? != FLAC_SIGNATURE {
72        return Ok(None);
73    }
74
75    // Following the previous OGG FLAC identification data is the stream information block as a
76    // native FLAC metadata block.
77    let header = MetadataBlockHeader::read(&mut reader)?;
78
79    if header.block_type != MetadataBlockType::StreamInfo {
80        return Ok(None);
81    }
82
83    // Ensure the block length is correct for a stream information block before allocating a buffer
84    // for it.
85    if !StreamInfo::is_valid_size(u64::from(header.block_len)) {
86        return Ok(None);
87    }
88
89    let extra_data = reader.read_boxed_slice_exact(header.block_len as usize)?;
90    let stream_info = StreamInfo::read(&mut BufReader::new(&extra_data))?;
91
92    // Populate the codec parameters with the information read from the stream information block.
93    let mut codec_params = CodecParameters::new();
94
95    codec_params
96        .for_codec(CODEC_TYPE_FLAC)
97        .with_packet_data_integrity(true)
98        .with_extra_data(extra_data)
99        .with_sample_rate(stream_info.sample_rate)
100        .with_time_base(TimeBase::new(1, stream_info.sample_rate))
101        .with_bits_per_sample(stream_info.bits_per_sample)
102        .with_channels(stream_info.channels);
103
104    if let Some(md5) = stream_info.md5 {
105        codec_params.with_verification_code(VerificationCheck::Md5(md5));
106    }
107
108    if let Some(n_frames) = stream_info.n_samples {
109        codec_params.with_n_frames(n_frames);
110    }
111
112    // Instantiate the FLAC mapper.
113    let mapper = Box::new(FlacMapper { codec_params });
114
115    Ok(Some(mapper))
116}

118/// Decodes a big-endian unsigned integer encoded via extended UTF8.
119fn utf8_decode_be_u64<B: ReadBytes>(src: &mut B) -> Result<Option<u64>> {
120    // NOTE: See the symphonia-bundle-flac crate for a detailed description of this function.
121    let mut state = u64::from(src.read_u8()?);
122
123    let mask: u8 = match state {
124        0x00..=0x7f => return Ok(Some(state)),
125        0xc0..=0xdf => 0x1f,
126        0xe0..=0xef => 0x0f,
127        0xf0..=0xf7 => 0x07,
128        0xf8..=0xfb => 0x03,
129        0xfc..=0xfd => 0x01,
130        0xfe => 0x00,
131        _ => return Ok(None),
132    };
133
134    state &= u64::from(mask);
135
136    for _ in 2..mask.leading_zeros() {
137        state = (state << 6) | u64::from(src.read_u8()? & 0x3f);
138    }
139
140    Ok(Some(state))
141}

143#[allow(dead_code)]
144struct FrameHeader {
145    ts: u64,
146    dur: u64,
147}
148
149/// Try to decode a FLAC frame header from the provided buffer.
150fn decode_frame_header(buf: &[u8]) -> Result<FrameHeader> {
151    // The FLAC frame header is checksummed with a CRC-8 hash.
152    let mut reader_crc8 = MonitorStream::new(BufReader::new(buf), Crc8Ccitt::new(0));
153
154    // Read the sync word.
155    let sync = reader_crc8.read_be_u16()?;
156
157    // Within an OGG packet the frame should be synchronized.
158    if sync & 0xfffc != 0xfff8 {
159        return decode_error("ogg (flac): header is not synchronized");
160    }
161
162    // Read all the standard frame description fields as one 16-bit value and extract the fields.
163    let desc = reader_crc8.read_be_u16()?;
164
165    // Reserved bit field.
166    if desc & 0x0001 == 1 {
167        return decode_error("ogg (flac): frame header reserved bit is not set to 1");
168    }
169
170    // Extract the blocking strategy from the sync word.
171    let is_fixed_block_size = sync & 0x1 == 0;
172
173    let block_sequence = if is_fixed_block_size {
174        // Fixed block size stream sequence blocks by a frame number.
175        let frame = match utf8_decode_be_u64(&mut reader_crc8)? {
176            Some(frame) => frame,
177            None => return decode_error("ogg (flac): frame sequence number is not valid"),
178        };
179
180        // The frame number should only be 31-bits.
181        if frame > 0x7fff_ffff {
182            return decode_error("ogg (flac): frame sequence number exceeds 31-bits");
183        }
184
185        frame
186    }
187    else {
188        // Variable block size streams sequence blocks by a sample number.
189        let sample = match utf8_decode_be_u64(&mut reader_crc8)? {
190            Some(sample) => sample,
191            None => return decode_error("ogg: sample sequence number is not valid"),
192        };
193
194        // The sample number should only be 36-bits.
195        if sample > 0xff_ffff_ffff {
196            return decode_error("ogg (flac): sample sequence number exceeds 36-bits");
197        }
198
199        sample
200    };
201
202    // The block size provides the duration in samples.
203    let block_size_enc = u32::from((desc & 0xf000) >> 12);
204
205    let block_size = match block_size_enc {
206        0x1 => 192,
207        0x2..=0x5 => 576 * (1 << (block_size_enc - 2)),
208        0x6 => u64::from(reader_crc8.read_u8()?) + 1,
209        0x7 => {
210            let block_size = reader_crc8.read_be_u16()?;
211            if block_size == 0xffff {
212                return decode_error("ogg (flac): block size not allowed to be greater than 65535");
213            }
214
215            u64::from(block_size) + 1
216        }
217        0x8..=0xf => 256 * (1 << (block_size_enc - 8)),
218        _ => return decode_error("ogg (flac): block size set to reserved value"),
219    };
220
221    // The sample rate is not required but should be read so checksum verification of the header
222    // can be performed.
223    let sample_rate_enc = u32::from((desc & 0x0f00) >> 8);
224
225    match sample_rate_enc {
226        0xc => {
227            reader_crc8.read_u8()?;
228        }
229        0xd => {
230            reader_crc8.read_be_u16()?;
231        }
232        0xe => {
233            reader_crc8.read_be_u16()?;
234        }
235        _ => (),
236    }
237
238    // End of frame header, get the computed CRC-8 checksum.
239    let crc8_computed = reader_crc8.monitor().crc();
240
241    // Read the expected CRC-8 checksum from the frame header.
242    let crc8_expected = reader_crc8.into_inner().read_u8()?;
243
244    if crc8_expected != crc8_computed && cfg!(not(fuzzing)) {
245        return decode_error("ogg (flac): computed frame header CRC does not match expected CRC");
246    }
247
248    let ts = if is_fixed_block_size { block_sequence * block_size } else { block_sequence };
249
250    Ok(FrameHeader { ts, dur: block_size })
251}

253struct FlacPacketParser {}
254
255impl PacketParser for FlacPacketParser {
256    fn parse_next_packet_dur(&mut self, packet: &[u8]) -> u64 {
257        match decode_frame_header(packet).ok() {
258            Some(header) => header.dur,
259            _ => 0,
260        }
261    }
262}

264struct FlacMapper {
265    codec_params: CodecParameters,
266}
267
268impl Mapper for FlacMapper {
269    fn name(&self) -> &'static str {
270        "flac"
271    }
272
273    fn codec_params(&self) -> &CodecParameters {
274        &self.codec_params
275    }
276
277    fn codec_params_mut(&mut self) -> &mut CodecParameters {
278        &mut self.codec_params
279    }
280
281    fn make_parser(&self) -> Option<Box<dyn super::PacketParser>> {
282        Some(Box::new(FlacPacketParser {}))
283    }
284
285    fn reset(&mut self) {
286        // Nothing to do.
287    }
288
289    fn map_packet(&mut self, packet: &[u8]) -> Result<MapResult> {
290        let packet_type = BufReader::new(packet).read_u8()?;
291
292        // A packet type of 0xff is an audio packet.
293        if packet_type == 0xff {
294            // Parse the packet duration.
295            let dur = match decode_frame_header(packet).ok() {
296                Some(header) => header.dur,
297                _ => 0,
298            };
299
300            Ok(MapResult::StreamData { dur })
301        }
302        else if packet_type == 0x00 || packet_type == 0x80 {
303            // Packet types 0x00 and 0x80 are invalid.
304            warn!("ogg (flac): flac packet type {} unexpected", packet_type);
305            Ok(MapResult::Unknown)
306        }
307        else {
308            let mut reader = BufReader::new(packet);
309
310            // Packet types in the range 0x01 thru 0x7f, and 0x81 thru 0xfe are metadata blocks.
311            let header = MetadataBlockHeader::read(&mut reader)?;
312
313            match header.block_type {
314                MetadataBlockType::VorbisComment => {
315                    let mut builder = MetadataBuilder::new();
316
317                    read_comment_block(&mut reader, &mut builder)?;
318
319                    Ok(MapResult::SideData { data: SideData::Metadata(builder.metadata()) })
320                }
321                MetadataBlockType::Picture => {
322                    let mut builder = MetadataBuilder::new();
323
324                    read_picture_block(&mut reader, &mut builder)?;
325
326                    Ok(MapResult::SideData { data: SideData::Metadata(builder.metadata()) })
327                }
328                _ => Ok(MapResult::Unknown),
329            }
330        }
331    }
332}
