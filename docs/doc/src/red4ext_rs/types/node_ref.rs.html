6#[derive(Debug, Clone, Copy)]
7#[repr(transparent)]
8pub struct NodeRef(red::NodeRef);
9
10unsafe impl NativeRepr for NodeRef {
11    const NAME: &'static str = "worldGlobalNodeRef";
12}
13
14impl NodeRef {
15    #[inline]
16    pub const fn is_defined(self) -> bool {
17        self.0.hash != 0
18    }
19
20    /// Creates a new `NodeRef` from the given string.
21    /// This function just calculates the hash of the string using the FNV-1a algorithm.
22    #[inline]
23    pub const fn new(name: &str) -> Self {
24        Self::from_bytes(name.as_bytes())
25    }
26
27    pub const fn from_bytes(mut tail: &[u8]) -> Self {
28        const SEED: u64 = 0xCBF2_9CE4_8422_2325;
29
30        let mut hash = SEED;
31        while let [byte, t @ ..] = tail {
32            tail = t;
33
34            if *byte == b'#' {
35                continue;
36            }
37
38            if *byte == b';' {
39                while let [next, t @ ..] = tail {
40                    if *next == b'/' {
41                        break;
42                    }
43                    tail = t;
44                }
45                continue;
46            }
47
48            hash = fnv1a64_step(hash, *byte);
49        }
50
51        if hash == SEED {
52            hash = 0;
53        }
54
55        Self(red::NodeRef { hash })
56    }
57}
