58#[derive(Debug, Default, Clone, Copy, PartialEq, PartialOrd, Eq)]
59#[repr(u8)]
60pub enum GamedataItemStructure {
61    #[default]
62    BlueprintStackable = 0,
63    Stackable = 1,
64    Unique = 2,
65    Count = 3,
66    Invalid = 4,
67}
68
69impl From<u8> for GamedataItemStructure {
70    fn from(value: u8) -> Self {
71        match value {
72            v if v == Self::BlueprintStackable as u8 => Self::BlueprintStackable,
73            v if v == Self::Stackable as u8 => Self::Stackable,
74            v if v == Self::Unique as u8 => Self::Unique,
75            v if v == Self::Count as u8 => Self::Count,
76            _ => Self::Invalid,
77        }
78    }
79}

81/// see [gameEItemIDFlag](https://nativedb.red4ext.com/gameEItemIDFlag)

88    Preview = 1,

91impl From<u8> for GameEItemIdFlag {
92    fn from(value: u8) -> Self {

95            _ => Self::None,
