31/// Native game instance.
32///
33/// Please note that it differs from Redscript and Lua's `GameInstance`,
34/// see [`GameInstance`].
35#[derive(Default)]
36#[repr(transparent)]
37pub struct NativeGameInstance(red::GameInstance);
38
39impl NativeGameInstance {
40    #[inline]
41    pub fn get_system(&self, ty: &Type) -> Ref<ScriptableSystem> {
42        let instance = unsafe { (self.vft().get_system)(self, ty) };
43        if instance.is_null() {
44            return Ref::default();
45        }
46        let instance: &WeakRef<ScriptableSystem> =
47            unsafe { mem::transmute(&(*instance)._base.ref_) };
48        instance.clone().upgrade().unwrap_or_default()
49    }
50
51    #[inline]
52    fn vft(&self) -> &GameInstanceVft {
53        unsafe { &*(self.0.vtable_ as *const GameInstanceVft) }
54    }
55}
56
57impl Drop for NativeGameInstance {
58    #[inline]
59    fn drop(&mut self) {
60        unsafe { (self.vft().destroy)(self) };
61    }
62}
