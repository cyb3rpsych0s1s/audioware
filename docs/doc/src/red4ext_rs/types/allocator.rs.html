12/// An interface for allocating and freeing memory.
13#[derive(Debug)]
14#[repr(transparent)]
15pub struct IAllocator(red::Memory::IAllocator);
16
17impl IAllocator {
18    /// Frees the memory pointed by `memory`.
19    #[inline]
20    pub unsafe fn free<T>(&self, memory: *mut T) {
21        let mut alloc = AllocationResult {
22            memory: memory as VoidPtr,
23            size: 0,
24        };
25        unsafe {
26            ((*self.0.vtable_).IAllocator_Free)(
27                &self.0 as *const _ as *mut red::Memory::IAllocator,
28                &mut alloc,
29            )
30        }
31    }
32
33    /// Allocates `size` bytes of memory with `alignment` bytes alignment.
34    #[inline]
35    pub unsafe fn alloc_aligned<T>(&self, size: u32, alignment: u32) -> *mut T {
36        unsafe {
37            let result = ((*self.0.vtable_).IAllocator_GetHandle)(
38                &self.0 as *const _ as *mut red::Memory::IAllocator,
39            );
40            let vault = vault_get(result);
41            vault_alloc_aligned(vault, size, alignment).unwrap_or(ptr::null_mut()) as _
42        }
43    }
44}

122/// A trait with operations for types that can be stored in a pool.
123#[sealed]
124pub trait PoolableOps: Poolable + Sized {
125    /// Allocates memory for `Self`. The resulting value must be initialized before use.
126    fn alloc() -> Option<PoolRef<mem::MaybeUninit<Self>>>;
127    /// Frees memory pointed by `ptr`.
128    fn free(ptr: &mut PoolRef<Self>);
129}
130
131#[sealed]
132impl<T: Poolable> PoolableOps for T {
133    fn alloc() -> Option<PoolRef<mem::MaybeUninit<Self>>> {
134        let result = unsafe { vault_alloc(T::Pool::vault(), mem::size_of::<T>() as u32)? };
135        (!result.is_null()).then(|| PoolRef(result.cast::<mem::MaybeUninit<Self>>()))
136    }
137
138    fn free(ptr: &mut PoolRef<Self>) {
139        let mut alloc = AllocationResult {
140            memory: ptr.0 as VoidPtr,
141            size: 0,
142        };
143        unsafe {
144            let free = crate::fn_from_hash!(
145                Memory_Vault_Free,
146                unsafe extern "C" fn(*mut red::Memory::Vault, *mut AllocationResult)
147            );
148            free(T::Pool::vault(), &mut alloc);
149        };
150    }
151}

153/// A trait for different types of pools.
154#[sealed]
155pub trait Pool {
156    const NAME: &'static str;
157
158    fn vault() -> *mut red::Memory::Vault {
159        static VAULT: OnceNonZeroUsize = OnceNonZeroUsize::new();
160        VAULT
161            .get_or_try_init(|| {
162                NonZeroUsize::new(unsafe { vault_get(fnv1a32(Self::NAME)) as _ }).ok_or(())
163            })
164            .expect("should resolve vault")
165            .get() as _
166    }
167}

196pub(super) unsafe fn vault_alloc(vault: *mut red::Memory::Vault, size: u32) -> Option<VoidPtr> {
197    let mut result = AllocationResult::default();
198    unsafe {
199        let alloc = crate::fn_from_hash!(
200            Memory_Vault_Alloc,
201            unsafe extern "C" fn(*mut red::Memory::Vault, *mut AllocationResult, u32)
202        );
203        alloc(vault, &mut result, size as _);
204    };
205    (!result.memory.is_null()).then_some(result.memory)
206}
207
208pub(super) unsafe fn vault_alloc_aligned(
209    vault: *mut red::Memory::Vault,
210    size: u32,
211    alignment: u32,
212) -> Option<VoidPtr> {
213    let mut result = AllocationResult::default();
214    unsafe {
215        let alloc_aligned = crate::fn_from_hash!(
216            Memory_Vault_AllocAligned,
217            unsafe extern "C" fn(*mut red::Memory::Vault, *mut AllocationResult, u32, u32)
218        );
219        alloc_aligned(vault, &mut result, size as _, alignment as _);
220    };
221    (!result.memory.is_null()).then_some(result.memory)
222}

224#[cold]
225pub(super) unsafe fn vault_get(handle: u32) -> *mut red::Memory::Vault {
226    unsafe {
227        let vault = &mut *red::Memory::Vault::Get();
228
229        vault.poolRegistry.nodesLock.lock_shared();
230        let Some(info) = vault
231            .poolRegistry
232            .nodes
233            .iter()
234            .find(|node| node.handle == handle)
235        else {
236            return ptr::null_mut();
237        };
238        let storage = (*info.storage).allocatorStorage & !7;
239        vault.poolRegistry.nodesLock.unlock_shared();
240
241        storage as _
242    }
