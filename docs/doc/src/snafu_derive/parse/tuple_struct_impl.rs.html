18impl Attributes {
19    fn from_syn(attrs: &[syn::Attribute]) -> syn::Result<Self> {
20        let location = ErrorLocation::OnTupleStruct;
21        let mut errors = SynErrors::default();
22
23        let mut crate_roots = AtMostOne::attribute(attr::CrateRoot, location);
24        let mut provide_expressions = Vec::new();
25        let mut source_froms = AtMostOne::attribute(attr::SourceFrom, location);
26
27        parse::syn_attrs(attrs, &mut errors, |errors, attr| {
28            use Attribute::*;
29
30            match attr {
31                Backtrace(a) => errors.push_invalid_flag(a, location),
32                ContextFlag(a) => errors.push_invalid_flag(a, location),
33                ContextName(a) => errors.push_invalid(a, location),
34                ContextSuffix(a) => errors.push_invalid(a, location),
35                CrateRoot(a) => crate_roots.push(a),
36                Display(a) => errors.push_invalid(a, location),
37                DocComment(_a) => { /* no-op */ }
38                Implicit(a) => errors.push_invalid_flag(a, location),
39                Module(a) => errors.push_invalid(a, location),
40                ProvideFlag(a) => errors.push_invalid_flag(a, location),
41                ProvideExpression(a) => provide_expressions.push(a),
42                SourceFlag(a) => errors.push_invalid_flag(a, location),
43                SourceFrom(a) => source_froms.push(a),
44                Transparent(a) => errors.push_invalid_flag(a, location),
45                Visibility(a) => errors.push_invalid(a, location),
46                Whatever(a) => errors.push_invalid(a, location),
47            }
48        });
49
50        let crate_root = crate_roots.finish_default(&mut errors);
51        let source_from = source_froms.finish_default(&mut errors);
52
53        errors.finish(Self {
54            crate_root,
55            provide_expressions,
56            source_from,
57        })
58    }
59}

61pub(crate) fn parse_tuple_struct(
62    fields: &syn::FieldsUnnamed,
63    name: &syn::Ident,
64    generics: &syn::Generics,
65    attrs: &[syn::Attribute],
66    span: impl quote::ToTokens,
67) -> syn::Result<TupleStructInfo> {
68    let attrs = Attributes::from_syn(attrs);
69    let field_ty = parse_tuple_struct_field(fields, span);
70
71    let (attrs, field_ty) = join_syn_error!(attrs, field_ty)?;
72
73    let Attributes {
74        crate_root,
75        provide_expressions,
76        source_from,
77    } = attrs;
78
79    let crate_root = into_crate_root(crate_root);
80    let generics = generics.clone();
81    let name = name.clone();
82    let provides = provide_expressions
83        .into_iter()
84        .map(|p| p.into_provide())
85        .collect();
86    let transformation = into_transformation(source_from, field_ty, true);
87
88    Ok(TupleStructInfo {
89        crate_root,
90        generics,
91        name,
92        provides,
93        transformation,
94    })
