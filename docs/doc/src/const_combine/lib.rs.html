2macro_rules! const_combine {
3    ($A:expr, $B:expr) => {
4        unsafe {
5            std::str::from_utf8_unchecked(
6                $crate::internal::combine::<{ $A.len() + $B.len() }>($A, $B).as_slice(),
7            )
8        }
9    };
10}

12/// useful when above doesn't work due to const expression limitations

40pub mod internal {
41    // based on https://users.rust-lang.org/t/concatenate-const-strings/51712/5
42    pub const fn combine<const L: usize>(a: &'static str, b: &'static str) -> [u8; L] {
43        let mut out = [0u8; L];
44        out = copy_slice(a.as_bytes(), out, 0);
45        out = copy_slice(b.as_bytes(), out, a.len());
46        out
47    }
48
49    pub const fn copy_slice<const L: usize>(
50        input: &[u8],
51        mut output: [u8; L],
52        offset: usize,
53    ) -> [u8; L] {
54        let mut index = 0;
55        loop {
56            output[offset + index] = input[index];
57            index += 1;
58            if index == input.len() {
59                break;
60            }
61        }
62        output
63    }
64
65    pub const fn take_n(input: &[u8], n: usize) -> &[u8] {
66        let mut index = input.len();
67        let mut slice = input;
68        loop {
69            match slice.split_last() {
70                None => return slice,
71                Some(_) if index == n => return slice,
72                Some((_, tail)) => {
73                    slice = tail;
74                    index -= 1;
75                }
76            }
77        }
78    }
79}
