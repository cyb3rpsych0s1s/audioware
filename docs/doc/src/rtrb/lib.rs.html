1//! A realtime-safe single-producer single-consumer (SPSC) ring buffer.
2//!
3//! A [`RingBuffer`] consists of two parts:
4//! a [`Producer`] for writing into the ring buffer and
5//! a [`Consumer`] for reading from the ring buffer.
6//!
7//! A fixed-capacity buffer is allocated on construction.
8//! After that, no more memory is allocated (unless the type `T` does that internally).
9//! Reading from and writing into the ring buffer is *lock-free* and *wait-free*.
10//! All reading and writing functions return immediately.
11//! Attempts to write to a full buffer return an error;
12//! values inside the buffer are *not* overwritten.
13//! Attempts to read from an empty buffer return an error as well.
14//! Only a single thread can write into the ring buffer and a single thread
15//! (typically a different one) can read from the ring buffer.
16//! If the queue is empty, there is no way for the reading thread to wait
17//! for new data, other than trying repeatedly until reading succeeds.
18//! Similarly, if the queue is full, there is no way for the writing thread
19//! to wait for newly available space to write to, other than trying repeatedly.
20//!
21//! # Examples
22//!
23//! Moving single elements into and out of a queue with
24//! [`Producer::push()`] and [`Consumer::pop()`], respectively:
25//!
26//! ```
27//! use rtrb::{RingBuffer, PushError, PopError};
28//!
29//! let (mut producer, mut consumer) = RingBuffer::new(2);
30//!
31//! assert_eq!(producer.push(10), Ok(()));
32//! assert_eq!(producer.push(20), Ok(()));
33//! assert_eq!(producer.push(30), Err(PushError::Full(30)));
34//!
35//! std::thread::spawn(move || {
36//!     assert_eq!(consumer.pop(), Ok(10));
37//!     assert_eq!(consumer.pop(), Ok(20));
38//!     assert_eq!(consumer.pop(), Err(PopError::Empty));
39//! }).join().unwrap();
40//! ```
41//!
42//! See the documentation of the [`chunks#examples`] module
43//! for examples that write multiple items at once with ...
44//!
45//!   * [`Producer::write_chunk()`]
46//!   * [`Producer::write_chunk_uninit()`]
47//!   * [`Producer::push_partial_slice()`] (if `T: Copy`)
48//!
49//! ... and read multiple items at once with ...
50//!
51//!   * [`Consumer::read_chunk()`]
52//!   * [`Consumer::pop_partial_slice()`] (if `T: Copy`)
53//!   * [`Consumer::pop_partial_slice_uninit()`] (if `T: Copy`)

80/// A bounded single-producer single-consumer (SPSC) queue.
81///
82/// Elements can be written with a [`Producer`] and read with a [`Consumer`],
83/// both of which can be obtained with [`RingBuffer::new()`].
84///
85/// *See also the [crate-level documentation](crate).*
86#[derive(Debug)]
87pub struct RingBuffer<T> {
88    /// The head of the queue.
89    ///
90    /// This integer is in range `0 .. 2 * capacity`.
91    head: CachePadded<AtomicUsize>,
92
93    /// The tail of the queue.
94    ///
95    /// This integer is in range `0 .. 2 * capacity`.
96    tail: CachePadded<AtomicUsize>,
97
98    /// The buffer holding slots.
99    data_ptr: *mut T,
100
101    /// The queue capacity.
102    capacity: usize,
103
104    /// Indicates that dropping a `RingBuffer<T>` may drop elements of type `T`.
105    _marker: PhantomData<T>,
106}
107
108impl<T> RingBuffer<T> {
109    /// Creates a `RingBuffer` with the given `capacity` and returns [`Producer`] and [`Consumer`].
110    ///
111    /// # Examples
112    ///
113    /// ```
114    /// use rtrb::RingBuffer;
115    ///
116    /// let (producer, consumer) = RingBuffer::<f32>::new(100);
117    /// ```
118    ///
119    /// Specifying an explicit type with the [turbofish](https://turbo.fish/)
120    /// is is only necessary if it cannot be deduced by the compiler.
121    ///
122    /// ```
123    /// use rtrb::RingBuffer;
124    ///
125    /// let (mut producer, consumer) = RingBuffer::new(100);
126    /// assert_eq!(producer.push(0.0f32), Ok(()));
127    /// ```
128    #[allow(clippy::new_ret_no_self)]
129    #[must_use]
130    pub fn new(capacity: usize) -> (Producer<T>, Consumer<T>) {
131        let buffer = Arc::new(RingBuffer {
132            head: CachePadded::new(AtomicUsize::new(0)),
133            tail: CachePadded::new(AtomicUsize::new(0)),
134            data_ptr: ManuallyDrop::new(Vec::with_capacity(capacity)).as_mut_ptr(),
135            capacity,
136            _marker: PhantomData,
137        });
138        let p = Producer {
139            buffer: buffer.clone(),
140            cached_head: Cell::new(0),
141            cached_tail: Cell::new(0),
142        };
143        let c = Consumer {
144            buffer,
145            cached_head: Cell::new(0),
146            cached_tail: Cell::new(0),
147        };
148        (p, c)
149    }
150
151    /// Returns the capacity of the queue.
152    ///
153    /// # Examples
154    ///
155    /// ```
156    /// use rtrb::RingBuffer;
157    ///
158    /// let (producer, consumer) = RingBuffer::<f32>::new(100);
159    /// assert_eq!(producer.buffer().capacity(), 100);
160    /// assert_eq!(consumer.buffer().capacity(), 100);
161    /// // Both producer and consumer of course refer to the same ring buffer:
162    /// assert_eq!(producer.buffer(), consumer.buffer());
163    /// ```
164    pub fn capacity(&self) -> usize {
165        self.capacity
166    }
167
168    /// Wraps a position from the range `0 .. 2 * capacity` to `0 .. capacity`.
169    fn collapse_position(&self, pos: usize) -> usize {
170        debug_assert!(pos == 0 || pos < 2 * self.capacity);
171        if pos < self.capacity {
172            pos
173        } else {
174            pos - self.capacity
175        }
176    }
177
178    /// Returns a pointer to the slot at position `pos`.
179    ///
180    /// If `pos == 0 && capacity == 0`, the returned pointer must not be dereferenced!
181    unsafe fn slot_ptr(&self, pos: usize) -> *mut T {
182        debug_assert!(pos == 0 || pos < 2 * self.capacity);
183        let pos = self.collapse_position(pos);
184        // SAFETY: The caller must ensure a valid pos.
185        unsafe { self.data_ptr.add(pos) }
186    }
187
188    /// Increments a position by going `n` slots forward.
189    fn increment(&self, pos: usize, n: usize) -> usize {
190        debug_assert!(pos == 0 || pos < 2 * self.capacity);
191        debug_assert!(n <= self.capacity);
192        let threshold = 2 * self.capacity - n;
193        if pos < threshold {
194            pos + n
195        } else {
196            pos - threshold
197        }
198    }
199
200    /// Increments a position by going one slot forward.
201    ///
202    /// This is more efficient than self.increment(..., 1).
203    fn increment1(&self, pos: usize) -> usize {
204        debug_assert_ne!(self.capacity, 0);
205        debug_assert!(pos < 2 * self.capacity);
206        if pos < 2 * self.capacity - 1 {
207            pos + 1
208        } else {
209            0
210        }
211    }
212
213    /// Returns the distance between two positions.
214    fn distance(&self, a: usize, b: usize) -> usize {
215        debug_assert!(a == 0 || a < 2 * self.capacity);
216        debug_assert!(b == 0 || b < 2 * self.capacity);
217        if a <= b {
218            b - a
219        } else {
220            2 * self.capacity - a + b
221        }
222    }
223}

225impl<T> Drop for RingBuffer<T> {
226    /// Drops all non-empty slots.
227    fn drop(&mut self) {
228        let mut head = self.head.load(Ordering::Relaxed);
229        let tail = self.tail.load(Ordering::Relaxed);
230
231        // Loop over all slots that hold a value and drop them.
232        while head != tail {
233            // SAFETY: All slots between head and tail have been initialized.
234            unsafe { self.slot_ptr(head).drop_in_place() };
235            head = self.increment1(head);
236        }
237
238        // Finally, deallocate the buffer, but don't run any destructors.
239        // SAFETY: data_ptr and capacity are still valid from the original initialization.
240        unsafe { Vec::from_raw_parts(self.data_ptr, 0, self.capacity) };
241    }
242}

265/// The producer side of a [`RingBuffer`].
266///
267/// Can be moved between threads,
268/// but references from different threads are not allowed
269/// (i.e. it is [`Send`] but not [`Sync`]).
270///
271/// Can only be created with [`RingBuffer::new()`]
272/// (together with its counterpart, the [`Consumer`]).
273///
274/// Individual elements can be moved into the ring buffer with [`Producer::push()`],
275/// multiple elements at once can be written with [`Producer::write_chunk()`],
276/// [`Producer::write_chunk_uninit()`] and [`Producer::push_partial_slice()`].
277///
278/// The number of free slots currently available for writing can be obtained with
279/// [`Producer::slots()`].
280///
281/// When the `Producer` is dropped, [`Consumer::is_abandoned()`] will return `true`.
282/// This can be used as a crude way to communicate to the receiving thread
283/// that no more data will be produced.
284/// When the `Producer` is dropped after the [`Consumer`] has already been dropped,
285/// [`RingBuffer::drop()`] will be called, freeing the allocated memory.
286#[derive(Debug, PartialEq, Eq)]
287pub struct Producer<T> {
288    /// A reference to the ring buffer.
289    buffer: Arc<RingBuffer<T>>,
290
291    /// A copy of `buffer.head` for quick access.
292    ///
293    /// This value can be stale and sometimes needs to be resynchronized with `buffer.head`.
294    cached_head: Cell<usize>,
295
296    /// A copy of `buffer.tail` for quick access.
297    ///
298    /// This value is always in sync with `buffer.tail`.
299    // NB: Caching the tail seems to have little effect on Intel CPUs, but it seems to
300    //     improve performance on AMD CPUs, see https://github.com/mgeier/rtrb/pull/132
301    cached_tail: Cell<usize>,
302}
303
304// SAFETY: After moving a Producer to another thread, there is still only a single thread
305// that can access the producer side of the queue.
306unsafe impl<T: Send> Send for Producer<T> {}
307
308impl<T> Producer<T> {
309    /// Attempts to push an element into the queue.
310    ///
311    /// The element is *moved* into the ring buffer and its slot
312    /// is made available to be read by the [`Consumer`].
313    ///
314    /// # Errors
315    ///
316    /// If the queue is full, the element is returned back as an error.
317    ///
318    /// # Examples
319    ///
320    /// ```
321    /// use rtrb::{RingBuffer, PushError};
322    ///
323    /// let (mut p, c) = RingBuffer::new(1);
324    ///
325    /// assert_eq!(p.push(10), Ok(()));
326    /// assert_eq!(p.push(20), Err(PushError::Full(20)));
327    /// ```
328    pub fn push(&mut self, value: T) -> Result<(), PushError<T>> {
329        if let Some(tail) = self.next_tail() {
330            // SAFETY: tail points to an empty slot.
331            unsafe { self.buffer.slot_ptr(tail).write(value) };
332            let tail = self.buffer.increment1(tail);
333            self.buffer.tail.store(tail, Ordering::Release);
334            self.cached_tail.set(tail);
335            Ok(())
336        } else {
337            Err(PushError::Full(value))
338        }
339    }
340
341    /// Returns the number of slots available for writing.
342    ///
343    /// Since items can be concurrently consumed on another thread, the actual number
344    /// of available slots may increase at any time (up to the [`RingBuffer::capacity()`]).
345    ///
346    /// To check for a single available slot,
347    /// using [`Producer::is_full()`] is often quicker
348    /// (because it might not have to check an atomic variable).
349    ///
350    /// # Examples
351    ///
352    /// ```
353    /// use rtrb::RingBuffer;
354    ///
355    /// let (p, c) = RingBuffer::<f32>::new(1024);
356    ///
357    /// assert_eq!(p.slots(), 1024);
358    /// ```
359    pub fn slots(&self) -> usize {
360        let head = self.buffer.head.load(Ordering::Acquire);
361        self.cached_head.set(head);
362        self.buffer.capacity - self.buffer.distance(head, self.cached_tail.get())
363    }
364
365    /// Returns the number of cached slots.
366    ///
367    /// In many cases, this will not provide all available slots,
368    /// but it might be marginally faster than [`Producer::slots()`]
369    /// because it doesn't access the atomic read index.
370    pub fn cached_slots(&self) -> usize {
371        let head = self.cached_head.get();
372        let tail = self.cached_tail.get();
373        self.buffer.capacity - self.buffer.distance(head, tail)
374    }
375
376    /// Returns `true` if there are currently no slots available for writing.
377    ///
378    /// A full ring buffer might cease to be full at any time
379    /// if the corresponding [`Consumer`] is consuming items in another thread.
380    ///
381    /// # Examples
382    ///
383    /// ```
384    /// use rtrb::RingBuffer;
385    ///
386    /// let (p, c) = RingBuffer::<f32>::new(1);
387    ///
388    /// assert!(!p.is_full());
389    /// ```
390    ///
391    /// Since items can be concurrently consumed on another thread, the ring buffer
392    /// might not be full for long:
393    ///
394    /// ```
395    /// # use rtrb::RingBuffer;
396    /// # let (p, c) = RingBuffer::<f32>::new(1);
397    /// if p.is_full() {
398    ///     // The buffer might be full, but it might as well not be
399    ///     // if an item was just consumed on another thread.
400    /// }
401    /// ```
402    ///
403    /// However, if it's not full, another thread cannot change that:
404    ///
405    /// ```
406    /// # use rtrb::RingBuffer;
407    /// # let (p, c) = RingBuffer::<f32>::new(1);
408    /// if !p.is_full() {
409    ///     // At least one slot is guaranteed to be available for writing.
410    /// }
411    /// ```
412    pub fn is_full(&self) -> bool {
413        self.next_tail().is_none()
414    }
415
416    /// Returns `true` if the corresponding [`Consumer`] has been destroyed.
417    ///
418    /// Note that since Rust version 1.74.0, this is not synchronizing with the consumer thread
419    /// anymore, see <https://github.com/mgeier/rtrb/issues/114>.
420    /// In a future version of `rtrb`, the synchronizing behavior might be restored.
421    ///
422    /// # Examples
423    ///
424    /// ```
425    /// use rtrb::RingBuffer;
426    ///
427    /// let (mut p, c) = RingBuffer::new(7);
428    /// assert!(!p.is_abandoned());
429    /// assert_eq!(p.push(10), Ok(()));
430    /// drop(c);
431    /// // The items that are still in the ring buffer are not accessible anymore.
432    /// assert!(p.is_abandoned());
433    /// // Even though it's futile, items can still be written:
434    /// assert_eq!(p.push(11), Ok(()));
435    /// ```
436    ///
437    /// Since the consumer can be concurrently dropped on another thread,
438    /// the producer might become abandoned at any time:
439    ///
440    /// ```
441    /// # use rtrb::RingBuffer;
442    /// # let (p, c) = RingBuffer::<i32>::new(1);
443    /// if !p.is_abandoned() {
444    ///     // Right now, the consumer might still be alive, but it might as well not be
445    ///     // if another thread has just dropped it.
446    /// }
447    /// ```
448    ///
449    /// However, if it already is abandoned, it will stay that way:
450    ///
451    /// ```
452    /// # use rtrb::RingBuffer;
453    /// # let (p, c) = RingBuffer::<i32>::new(1);
454    /// if p.is_abandoned() {
455    ///     // This is needed since Rust 1.74.0, see https://github.com/mgeier/rtrb/issues/114:
456    ///     std::sync::atomic::fence(std::sync::atomic::Ordering::Acquire);
457    ///     // The consumer does definitely not exist anymore.
458    /// }
459    /// ```
460    pub fn is_abandoned(&self) -> bool {
461        Arc::strong_count(&self.buffer) < 2
462    }
463
464    /// Returns a read-only reference to the ring buffer.
465    pub fn buffer(&self) -> &RingBuffer<T> {
466        &self.buffer
467    }
468
469    /// Get the tail position for writing the next slot, if available.
470    ///
471    /// This is a strict subset of the functionality implemented in `write_chunk_uninit()`.
472    /// For performance, this special case is immplemented separately.
473    fn next_tail(&self) -> Option<usize> {
474        let tail = self.cached_tail.get();
475
476        // Check if the queue is *possibly* full.
477        if self.buffer.distance(self.cached_head.get(), tail) == self.buffer.capacity {
478            // Refresh the head ...
479            let head = self.buffer.head.load(Ordering::Acquire);
480            self.cached_head.set(head);
481
482            // ... and check if it's *really* full.
483            if self.buffer.distance(head, tail) == self.buffer.capacity {
484                return None;
485            }
486        }
487        Some(tail)
488    }
489}

491/// The consumer side of a [`RingBuffer`].
492///
493/// Can be moved between threads,
494/// but references from different threads are not allowed
495/// (i.e. it is [`Send`] but not [`Sync`]).
496///
497/// Can only be created with [`RingBuffer::new()`]
498/// (together with its counterpart, the [`Producer`]).
499///
500/// Individual elements can be moved out of the ring buffer with [`Consumer::pop()`],
501/// multiple elements at once can be read with [`Consumer::read_chunk()`],
502/// [`Consumer::pop_partial_slice()`] and [`Consumer::pop_partial_slice_uninit()`].
503///
504/// The number of slots currently available for reading can be obtained with
505/// [`Consumer::slots()`].
506///
507/// When the `Consumer` is dropped, [`Producer::is_abandoned()`] will return `true`.
508/// This can be used as a crude way to communicate to the sending thread
509/// that no more data will be consumed.
510/// When the `Consumer` is dropped after the [`Producer`] has already been dropped,
511/// [`RingBuffer::drop()`] will be called, freeing the allocated memory.
512#[derive(Debug, PartialEq, Eq)]
513pub struct Consumer<T> {
514    /// A reference to the ring buffer.
515    buffer: Arc<RingBuffer<T>>,
516
517    /// A copy of `buffer.head` for quick access.
518    ///
519    /// This value is always in sync with `buffer.head`.
520    // NB: Caching the head seems to have little effect on Intel CPUs, but it seems to
521    //     improve performance on AMD CPUs, see https://github.com/mgeier/rtrb/pull/132
522    cached_head: Cell<usize>,
523
524    /// A copy of `buffer.tail` for quick access.
525    ///
526    /// This value can be stale and sometimes needs to be resynchronized with `buffer.tail`.
527    cached_tail: Cell<usize>,
528}

534impl<T> Consumer<T> {
535    /// Attempts to pop an element from the queue.
536    ///
537    /// The element is *moved* out of the ring buffer and its slot
538    /// is made available to be filled by the [`Producer`] again.
539    ///
540    /// # Errors
541    ///
542    /// If the queue is empty, an error is returned.
543    ///
544    /// # Examples
545    ///
546    /// ```
547    /// use rtrb::{PopError, RingBuffer};
548    ///
549    /// let (mut p, mut c) = RingBuffer::new(1);
550    ///
551    /// assert_eq!(p.push(10), Ok(()));
552    /// assert_eq!(c.pop(), Ok(10));
553    /// assert_eq!(c.pop(), Err(PopError::Empty));
554    /// ```
555    ///
556    /// To obtain an [`Option<T>`](Option), use [`.ok()`](Result::ok) on the result.
557    ///
558    /// ```
559    /// # use rtrb::RingBuffer;
560    /// # let (mut p, mut c) = RingBuffer::new(1);
561    /// assert_eq!(p.push(20), Ok(()));
562    /// assert_eq!(c.pop().ok(), Some(20));
563    /// ```
564    pub fn pop(&mut self) -> Result<T, PopError> {
565        if let Some(head) = self.next_head() {
566            // SAFETY: head points to an initialized slot.
567            let value = unsafe { self.buffer.slot_ptr(head).read() };
568            let head = self.buffer.increment1(head);
569            self.buffer.head.store(head, Ordering::Release);
570            self.cached_head.set(head);
571            Ok(value)
572        } else {
573            Err(PopError::Empty)
574        }
575    }
576
577    /// Attempts to read an element from the queue without removing it.
578    ///
579    /// # Errors
580    ///
581    /// If the queue is empty, an error is returned.
582    ///
583    /// # Examples
584    ///
585    /// ```
586    /// use rtrb::{PeekError, RingBuffer};
587    ///
588    /// let (mut p, c) = RingBuffer::new(1);
589    ///
590    /// assert_eq!(c.peek(), Err(PeekError::Empty));
591    /// assert_eq!(p.push(10), Ok(()));
592    /// assert_eq!(c.peek(), Ok(&10));
593    /// assert_eq!(c.peek(), Ok(&10));
594    /// ```
595    ///
596    /// Note that `peek()` takes a shared reference to `self`,
597    /// which means that other methods that take `&self` can be called
598    /// while the returned reference is still in use.
599    /// However, calling methods that take `&mut self`
600    /// (like [`Consumer::pop()`] and [`Consumer::read_chunk()`]) leads to a compiler error:
601    ///
602    /// ```compile_fail
603    /// use rtrb::RingBuffer;
604    ///
605    /// let (mut p, mut c) = RingBuffer::new(8);
606    /// p.push(10).unwrap();
607    /// let shared_ref = c.peek().unwrap();
608    /// let value = c.pop().unwrap();
609    /// assert_eq!(shared_ref, &10);
610    /// ```
611    pub fn peek(&self) -> Result<&T, PeekError> {
612        if let Some(head) = self.next_head() {
613            // SAFETY: head points to an initialized slot.
614            Ok(unsafe { &*self.buffer.slot_ptr(head) })
615        } else {
616            Err(PeekError::Empty)
617        }
618    }
619
620    /// Returns the number of slots available for reading.
621    ///
622    /// Since items can be concurrently produced on another thread, the actual number
623    /// of available slots may increase at any time (up to the [`RingBuffer::capacity()`]).
624    ///
625    /// To check for a single available slot,
626    /// using [`Consumer::is_empty()`] is often quicker
627    /// (because it might not have to check an atomic variable).
628    ///
629    /// # Examples
630    ///
631    /// ```
632    /// use rtrb::RingBuffer;
633    ///
634    /// let (p, c) = RingBuffer::<f32>::new(1024);
635    ///
636    /// assert_eq!(c.slots(), 0);
637    /// ```
638    pub fn slots(&self) -> usize {
639        let tail = self.buffer.tail.load(Ordering::Acquire);
640        self.cached_tail.set(tail);
641        self.buffer.distance(self.cached_head.get(), tail)
642    }
643
644    /// Returns the number of cached slots.
645    ///
646    /// In many cases, this will not provide all available slots,
647    /// but it might be marginally faster than [`Consumer::slots()`]
648    /// because it doesn't access the atomic write index.
649    pub fn cached_slots(&self) -> usize {
650        let head = self.cached_head.get();
651        let tail = self.cached_tail.get();
652        self.buffer.distance(head, tail)
653    }

655    /// Returns `true` if there are currently no slots available for reading.
656    ///
657    /// An empty ring buffer might cease to be empty at any time
658    /// if the corresponding [`Producer`] is producing items in another thread.
659    ///
660    /// # Examples
661    ///
662    /// ```
663    /// use rtrb::RingBuffer;
664    ///
665    /// let (p, c) = RingBuffer::<f32>::new(1);
666    ///
667    /// assert!(c.is_empty());
668    /// ```
669    ///
670    /// Since items can be concurrently produced on another thread, the ring buffer
671    /// might not be empty for long:
672    ///
673    /// ```
674    /// # use rtrb::RingBuffer;
675    /// # let (p, c) = RingBuffer::<f32>::new(1);
676    /// if c.is_empty() {
677    ///     // The buffer might be empty, but it might as well not be
678    ///     // if an item was just produced on another thread.
679    /// }
680    /// ```
681    ///
682    /// However, if it's not empty, another thread cannot change that:
683    ///
684    /// ```
685    /// # use rtrb::RingBuffer;
686    /// # let (p, c) = RingBuffer::<f32>::new(1);
687    /// if !c.is_empty() {
688    ///     // At least one slot is guaranteed to be available for reading.
689    /// }
690    /// ```
691    pub fn is_empty(&self) -> bool {
692        self.next_head().is_none()
693    }
694
695    /// Returns `true` if the corresponding [`Producer`] has been destroyed.
696    ///
697    /// Note that since Rust version 1.74.0, this is not synchronizing with the producer thread
698    /// anymore, see <https://github.com/mgeier/rtrb/issues/114>.
699    /// In a future version of `rtrb`, the synchronizing behavior might be restored.
700    ///
701    /// # Examples
702    ///
703    /// ```
704    /// use rtrb::RingBuffer;
705    ///
706    /// let (mut p, mut c) = RingBuffer::new(7);
707    /// assert!(!c.is_abandoned());
708    /// assert_eq!(p.push(10), Ok(()));
709    /// drop(p);
710    /// assert!(c.is_abandoned());
711    /// // The items that are left in the ring buffer can still be consumed:
712    /// assert_eq!(c.pop(), Ok(10));
713    /// ```
714    ///
715    /// Since the producer can be concurrently dropped on another thread,
716    /// the consumer might become abandoned at any time:
717    ///
718    /// ```
719    /// # use rtrb::RingBuffer;
720    /// # let (p, c) = RingBuffer::<i32>::new(1);
721    /// if !c.is_abandoned() {
722    ///     // Right now, the producer might still be alive, but it might as well not be
723    ///     // if another thread has just dropped it.
724    /// }
725    /// ```
726    ///
727    /// However, if it already is abandoned, it will stay that way:
728    ///
729    /// ```
730    /// # use rtrb::RingBuffer;
731    /// # let (p, c) = RingBuffer::<i32>::new(1);
732    /// if c.is_abandoned() {
733    ///     // This is needed since Rust 1.74.0, see https://github.com/mgeier/rtrb/issues/114:
734    ///     std::sync::atomic::fence(std::sync::atomic::Ordering::Acquire);
735    ///     // The producer does definitely not exist anymore.
736    /// }
737    /// ```
738    pub fn is_abandoned(&self) -> bool {
739        Arc::strong_count(&self.buffer) < 2
740    }
741
742    /// Returns a read-only reference to the ring buffer.
743    pub fn buffer(&self) -> &RingBuffer<T> {
744        &self.buffer
745    }
746
747    /// Get the head position for reading the next slot, if available.
748    ///
749    /// This is a strict subset of the functionality implemented in `read_chunk()`.
750    /// For performance, this special case is immplemented separately.
751    fn next_head(&self) -> Option<usize> {
752        let head = self.cached_head.get();
753
754        // Check if the queue is *possibly* empty.
755        if head == self.cached_tail.get() {
756            // Refresh the tail ...
757            let tail = self.buffer.tail.load(Ordering::Acquire);
758            self.cached_tail.set(tail);
759
760            // ... and check if it's *really* empty.
761            if head == tail {
762                return None;
763            }
764        }
765        Some(head)
766    }
767}

769/// Extension trait used to provide a [`copy_to_uninit()`](CopyToUninit::copy_to_uninit)
770/// method on built-in slices.
771///
772/// This can be used to safely copy data to the slices returned from
773/// [`WriteChunkUninit::as_mut_slices()`].
774///
775/// To use this, the trait has to be brought into scope, e.g. with:
776///
777/// ```
778/// use rtrb::CopyToUninit;
779/// ```
780pub trait CopyToUninit<T: Copy> {
781    /// Copies contents to a possibly uninitialized slice.
782    fn copy_to_uninit<'a>(&self, dst: &'a mut [MaybeUninit<T>]) -> &'a mut [T];
783}
784
785impl<T: Copy> CopyToUninit<T> for [T] {
786    /// Copies contents to a possibly uninitialized slice.
787    ///
788    /// # Panics
789    ///
790    /// This function will panic if the two slices have different lengths.
791    fn copy_to_uninit<'a>(&self, dst: &'a mut [MaybeUninit<T>]) -> &'a mut [T] {
792        assert_eq!(
793            self.len(),
794            dst.len(),
795            "source slice length does not match destination slice length"
796        );
797        let dst_ptr = dst.as_mut_ptr().cast();
798        // SAFETY: The lengths have been checked to be equal and
799        // the mutable reference makes sure that there is no overlap.
800        unsafe {
801            self.as_ptr().copy_to_nonoverlapping(dst_ptr, self.len());
802            core::slice::from_raw_parts_mut(dst_ptr, self.len())
803        }
804    }
805}
